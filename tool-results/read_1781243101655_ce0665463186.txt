   153→// Relative time helper
   154→function getRelativeTime(dateStr?: string) {
   155→  if (!dateStr) return '';
   156→  const now = new Date();
   157→  const date = new Date(dateStr);
   158→  const diffMs = now.getTime() - date.getTime();
   159→  const diffMins = Math.floor(diffMs / 60000);
   160→  const diffHours = Math.floor(diffMins / 60);
   161→  const diffDays = Math.floor(diffHours / 24);
   162→
   163→  if (diffMins < 1) return 'Just now';
   164→  if (diffMins < 60) return `${diffMins}m ago`;
   165→  if (diffHours < 24) return `${diffHours}h ago`;
   166→  if (diffDays < 7) return `${diffDays}d ago`;
   167→  return date.toLocaleDateString();
   168→}

   221→// ─── Animated Counter ─────────────────────────────
   222→function AnimatedCounter({ value, duration = 1.5 }: { value: string; duration?: number }) {
   223→  const displayRef = useRef<HTMLSpanElement>(null);
   224→  const hasAnimated = useRef(false);
   225→
   226→  useEffect(() => {
   227→    if (hasAnimated.current) return;
   228→    hasAnimated.current = true;
   229→
   230→    const el = displayRef.current;
   231→    if (!el) return;
   232→
   233→    const numericMatch = value.match(/[\d,]+/);
   234→    if (!numericMatch) {
   235→      el.textContent = value;
   236→      return;
   237→    }
   238→    const numericStr = numericMatch[0].replace(/,/g, '');
   239→    const target = parseInt(numericStr, 10);
   240→    if (isNaN(target)) {
   241→      el.textContent = value;
   242→      return;
   243→    }
   244→
   245→    const prefix = value.substring(0, value.indexOf(numericMatch[0]));
   246→    const suffix = value.substring(value.indexOf(numericMatch[0]) + numericMatch[0].length);
   247→    const startTime = performance.now();
   248→
   249→    const animate = (currentTime: number) => {
   250→      const elapsed = currentTime - startTime;
   251→      const progress = Math.min(elapsed / (duration * 1000), 1);
   252→      const eased = 1 - Math.pow(1 - progress, 3);
   253→      const current = Math.round(eased * target);
   254→      const formatted = current.toLocaleString();
   255→      if (displayRef.current) {
   256→        displayRef.current.textContent = `${prefix}${formatted}${suffix}`;
   257→      }
   258→      if (progress < 1) {
   259→        requestAnimationFrame(animate);
   260→      }
   261→    };
   262→    requestAnimationFrame(animate);
   263→  }, [value, duration]);
   264→
   265→  return <span ref={displayRef}>0</span>;
   266→}

   282→    <svg width={size} height={size} className="-rotate-90">

   284→        cx={size / 2}
   285→        cy={size / 2}
   286→        r={radius}

   330→  const shapeClass = cn(

   342→      animate={{
   343→        y: [-10, 10, -10],
   344→        rotate: type === 'square' ? [12, 24, 12] : type === 'diamond' ? [45, 55, 45] : [0, 5, 0],

   347→        duration: duration || 6,
   348→        repeat: Infinity,
   349→        ease: 'easeInOut',

   483→// ─── Parallax Tilt Card ──────────────────────────
   484→function TiltCard({ children, className }: { children: React.ReactNode; className?: string }) {
   485→  const cardRef = useRef<HTMLDivElement>(null);
   486→  const [rotateX, setRotateX] = useState(0);
   487→  const [rotateY, setRotateY] = useState(0);
   488→
   489→  const handleMouseMove = (e: React.MouseEvent<HTMLDivElement>) => {
   490→    if (!cardRef.current) return;
   491→    const rect = cardRef.current.getBoundingClientRect();
   492→    const x = e.clientX - rect.left;
   493→    const y = e.clientY - rect.top;
   494→    const centerX = rect.width / 2;
   495→    const centerY = rect.height / 2;
   496→    const maxTilt = 5;
   497→    const newRotateX = ((y - centerY) / centerY) * -maxTilt;
   498→    const newRotateY = ((x - centerX) / centerX) * maxTilt;
   499→    setRotateX(newRotateX);
   500→    setRotateY(newRotateY);
   501→  };
   502→
   503→  const handleMouseLeave = () => {
   504→    setRotateX(0);
   505→    setRotateY(0);
   506→  };
   507→
   508→  return (
   509→    <div
   510→      ref={cardRef}
   511→      onMouseMove={handleMouseMove}
   512→      onMouseLeave={handleMouseLeave}
   513→      className={className}
   514→      style={{
   515→        transform: `perspective(800px) rotateX(${rotateX}deg) rotateY(${rotateY}deg)`,
   516→        transition: 'transform 0.15s ease-out',
   517→      }}
   518→    >
   519→      {children}
   520→    </div>
   521→  );
   522→}

   601→function KPICircularProgress({ value, color, size = 48, strokeWidth = 4 }: { value: number; color: string; size?: number; strokeWidth?: number }) {
   602→  const [animatedValue, setAnimatedValue] = useState(0);
   603→
   604→  useEffect(() => {
   605→    const timer = setTimeout(() => setAnimatedValue(value), 150);
   606→    return () => clearTimeout(timer);
   607→  }, [value]);
   608→
   609→  const radius = (size - strokeWidth) / 2;
   610→  const circumference = 2 * Math.PI * radius;
   611→  const strokeDashoffset = circumference - (animatedValue / 100) * circumference;
   612→
   613→  return (
   614→    <svg width={size} height={size} className="-rotate-90">
   615→      <circle
   616→        cx={size / 2}
   617→        cy={size / 2}
   618→        r={radius}
   619→        fill="none"
   620→        stroke="currentColor"
   621→        strokeWidth={strokeWidth}
   622→        className="text-slate-200/50 dark:text-slate-700/50"
   623→      />
   624→      <motion.circle
   625→        cx={size / 2}
   626→        cy={size / 2}
   627→        r={radius}
   628→        fill="none"
   629→        stroke={color}
   630→        strokeWidth={strokeWidth}
   631→        strokeLinecap="round"
   632→        strokeDasharray={circumference}
   633→        initial={{ strokeDashoffset: circumference }}
   634→        animate={{ strokeDashoffset }}
   635→        transition={{ duration: 1.2, ease: 'easeOut', delay: 0.2 }}
   636→      />
   637→    </svg>
