   281тЖТ   281тЖТfunction AnimatedCounter({ value, duration = 1500 }: { value: string; duration?: number }) {
   282тЖТ   282тЖТ  const [displayed, setDisplayed] = useState('0');
   283тЖТ   283тЖТ  const hasAnimated = useRef(false);
   284тЖТ   284тЖТ
   285тЖТ   285тЖТ  useEffect(() => {
   286тЖТ   286тЖТ    if (hasAnimated.current) return;
   287тЖТ   287тЖТ    hasAnimated.current = true;
   288тЖТ   288тЖТ
   289тЖТ   289тЖТ    const numericMatch = value.match(/[\d,.]+/);
   290тЖТ   290тЖТ    if (!numericMatch) {
   291тЖТ   291тЖТ      return;
   292тЖТ   292тЖТ    }
   293тЖТ   293тЖТ
   294тЖТ   294тЖТ    const numericStr = numericMatch[0];
   295тЖТ   295тЖТ    const prefix = value.slice(0, value.indexOf(numericStr));
   296тЖТ   296тЖТ    const suffix = value.slice(value.indexOf(numericStr) + numericStr.length);
   297тЖТ   297тЖТ    const cleanNumber = parseFloat(numericStr.replace(/,/g, ''));
   298тЖТ   298тЖТ    const hasDecimals = numericStr.includes('.');
   299тЖТ   299тЖТ    const decimalPlaces = hasDecimals ? numericStr.split('.')[1].length : 0;
   300тЖТ   300тЖТ
   301тЖТ   301тЖТ    const startTime = Date.now();
   302тЖТ   302тЖТ    const animate = () => {
   303тЖТ   303тЖТ      const elapsed = Date.now() - startTime;
   304тЖТ   304тЖТ      const progress = Math.min(elapsed / duration, 1);
   305тЖТ   305тЖТ      const eased = 1 - Math.pow(1 - progress, 3);
   306тЖТ   306тЖТ      const current = cleanNumber * eased;
   307тЖТ   307тЖТ
   308тЖТ   308тЖТ      let formatted: string;
   309тЖТ   309тЖТ      if (hasDecimals) {
   310тЖТ   310тЖТ        formatted = current.toFixed(decimalPlaces);
   311тЖТ   311тЖТ      } else {
   312тЖТ   312тЖТ        formatted = Math.round(current).toLocaleString('en-US');
   313тЖТ   313тЖТ      }
   314тЖТ   314тЖТ
   315тЖТ   315тЖТ      setDisplayed(`${prefix}${formatted}${suffix}`);
   316тЖТ   316тЖТ
   317тЖТ   317тЖТ      if (progress < 1) {
   318тЖТ   318тЖТ        requestAnimationFrame(animate);
   319тЖТ   319тЖТ      }
   320тЖТ   320тЖТ    };
   321тЖТ   321тЖТ
   322тЖТ   322тЖТ    requestAnimationFrame(animate);
   323тЖТ   323тЖТ  }, [value, duration]);
   324тЖТ   324тЖТ
   325тЖТ   325тЖТ  const numericMatch = value.match(/[\d,.]+/);
   326тЖТ   326тЖТ  if (!numericMatch) {
   327тЖТ   327тЖТ    return <span>{value}</span>;
   328тЖТ   328тЖТ  }
   329тЖТ   329тЖТ
   330тЖТ   330тЖТ  return <span>{displayed}</span>;
   331тЖТ   331тЖТ}
