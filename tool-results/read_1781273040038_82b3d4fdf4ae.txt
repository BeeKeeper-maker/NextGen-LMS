   141→   141→function timeAgo(dateStr: string) {
   142→   142→  const now = new Date();
   143→   143→  const date = new Date(dateStr);
   144→   144→  const diffMs = now.getTime() - date.getTime();
   145→   145→  const diffMins = Math.floor(diffMs / 60000);
   146→   146→  if (diffMins < 60) return `${diffMins}m ago`;
   147→   147→  const diffHours = Math.floor(diffMins / 60);
   148→   148→  if (diffHours < 24) return `${diffHours}h ago`;
   149→   149→  const diffDays = Math.floor(diffHours / 24);
   150→   150→  if (diffDays < 30) return `${diffDays}d ago`;
   151→   151→  return `${Math.floor(diffDays / 30)}mo ago`;
   152→   152→}

   172→   172→function AnimatedCounter({ target, duration = 1.5 }: { target: number; duration?: number }) {
   173→   173→  const [count, setCount] = useState(0);
   174→   174→  useEffect(() => {
   175→   175→    let start = 0;
   176→   176→    const increment = target / (duration * 60);
   177→   177→    const timer = setInterval(() => {
   178→   178→      start += increment;
   179→   179→      if (start >= target) {
   180→   180→        setCount(target);
   181→   181→        clearInterval(timer);
   182→   182→      } else {
   183→   183→        setCount(Math.floor(start));
   184→   184→      }
   185→   185→    }, 1000 / 60);
   186→   186→    return () => clearInterval(timer);
   187→   187→  }, [target, duration]);
   188→   188→
   189→   189→  return <span>{count.toLocaleString()}</span>;
   190→   190→}

   194→   194→function SparklineChart({ data, color = '#10B981', width = 48, height = 20 }: { data: number[]; color?: string; width?: number; height?: number }) {
   195→   195→  if (data.length < 2) return null;
   196→   196→  const max = Math.max(...data);
   197→   197→  const min = Math.min(...data);
   198→   198→  const range = max - min || 1;
   199→   199→  const padding = 2;
   200→   200→
   201→   201→  const points = data.map((val, i) => {
   202→   202→    const x = padding + (i / (data.length - 1)) * (width - padding * 2);
   203→   203→    const y = height - padding - ((val - min) / range) * (height - padding * 2);
   204→   204→    return `${x},${y}`;
   205→   205→  }).join(' ');
   206→   206→
   207→   207→  const areaPoints = `${padding},${height - padding} ${points} ${width - padding},${height - padding}`;
   208→   208→
   209→   209→  return (
   210→   210→    <svg width={width} height={height} className="shrink-0">
   211→   211→      <defs>
   212→   212→        <linearGradient id={`grad-${color.replace('#', '')}`} x1="0" y1="0" x2="0" y2="1">
   213→   213→          <stop offset="0%" stopColor={color} stopOpacity="0.3" />
   214→   214→          <stop offset="100%" stopColor={color} stopOpacity="0.05" />
   215→   215→        </linearGradient>
   216→   216→      </defs>
   217→   217→      <polygon points={areaPoints} fill={`url(#grad-${color.replace('#', '')})`} />
   218→   218→      <polyline points={points} fill="none" stroke={color} strokeWidth="1.5" strokeLinecap="round" strokeLinejoin="round" />
   219→   219→    </svg>
   220→   220→  );
   221→   221→}
