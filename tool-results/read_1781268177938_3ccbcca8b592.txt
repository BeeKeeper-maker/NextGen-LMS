    89→// ---- Helper to map API path data to LearnerPath ----
    90→function mapApiPathToLearnerPath(apiPath: any, userEnrollment: any): LearnerPath {
    91→  const progress = userEnrollment?.progress || 0;
    92→  const courses: LearnerCourse[] = (apiPath.courses || []).map((pc: any, idx: number) => {
    93→    const course = pc.course || {};
    94→    let status: CourseStatus = 'locked';
    95→    if (progress >= 100 || (userEnrollment?.status === 'completed')) {
    96→      status = 'completed';
    97→    } else if (idx === 0 || progress > (idx / apiPath.courses.length) * 100) {
    98→      const courseProgress = Math.min(100, Math.max(0, Math.round((progress - (idx / apiPath.courses.length) * 100) / (1 / apiPath.courses.length) * 100)));
    99→      if (courseProgress >= 100) status = 'completed';
   100→      else status = 'current';
   101→    }
   102→    return {
   103→      id: pc.id || course.id || `c-${idx}`,
   104→      title: course.title || 'Untitled Course',
   105→      duration: Math.round(course.durationHours || 0),
   106→      level: (course.level || 'beginner') as LearnerCourse['level'],
   107→      status,
   108→      isRequired: pc.isRequired ?? true,
   109→      prerequisiteIds: pc.prerequisiteIds ? pc.prerequisiteIds.split(',') : [],
   110→      milestone: pc.milestone || undefined,
   111→      progress: status === 'completed' ? 100 : status === 'current' ? Math.round((progress % (100 / apiPath.courses.length)) / (100 / apiPath.courses.length) * 100) : undefined,
   112→    };
   113→  });
   114→  const currentCourse = courses.find(c => c.status === 'current');
   115→  return {
   116→    id: apiPath.id,
   117→    name: apiPath.title,
   118→    description: apiPath.description || '',
   119→    category: apiPath.category || 'General',
   120→    difficulty: (apiPath.level || 'beginner') as LearnerPath['difficulty'],
   121→    courses,
   122→    enrolledAt: userEnrollment?.startedAt || new Date().toISOString(),
   123→    estimatedDuration: apiPath.estimatedDuration || 0,
   124→    overallProgress: Math.round(progress),
   125→    certificateEarned: userEnrollment?.status === 'completed',
   126→    nextCourse: currentCourse?.title,
   127→  };
   128→}

   146→function ProgressRing({ progress, size = 48, strokeWidth = 4 }: { progress: number; size?: number; strokeWidth?: number }) {
   147→  const radius = (size - strokeWidth) / 2;
   148→  const circumference = radius * 2 * Math.PI;
   149→  const offset = circumference - (progress / 100) * circumference;
   150→
   151→  return (
   152→    <div className="relative inline-flex items-center justify-center" style={{ width: size, height: size }}>
   153→      <svg className="transform -rotate-90" width={size} height={size}>
   154→        <circle
   155→          cx={size / 2}
   156→          cy={size / 2}
   157→          r={radius}
   158→          stroke="currentColor"
   159→          strokeWidth={strokeWidth}
   160→          fill="none"
   161→          className="text-muted/30"
   162→        />
   163→        <circle
   164→          cx={size / 2}
   165→          cy={size / 2}
   166→          r={radius}
   167→          stroke="url(#progressGradient)"
   168→          strokeWidth={strokeWidth}
   169→          fill="none"
   170→          strokeDasharray={circumference}
   171→          strokeDashoffset={offset}
   172→          strokeLinecap="round"
   173→          className="transition-all duration-700"
   174→        />
   175→        <defs>
   176→          <linearGradient id="progressGradient" x1="0%" y1="0%" x2="100%" y2="0%">
   177→            <stop offset="0%" stopColor="#10b981" />
   178→            <stop offset="100%" stopColor="#14b8a6" />
   179→          </linearGradient>
   180→        </defs>
   181→      </svg>
   182→      <span className="absolute text-xs font-bold">{progress}%</span>
   183→    </div>
   184→  );
   185→}
