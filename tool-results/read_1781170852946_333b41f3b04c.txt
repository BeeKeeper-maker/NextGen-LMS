    94→function formatDuration(seconds: number): string {
    95→  const mins = Math.floor(seconds / 60);
    96→  if (mins < 60) return `${mins}m`;
    97→  const hrs = Math.floor(mins / 60);
    98→  const rem = mins % 60;
    99→  return rem > 0 ? `${hrs}h ${rem}m` : `${hrs}h`;
   100→}

   399→function CourseBuilderTab() {
   400→  const [selectedCourseId, setSelectedCourseId] = useState<string>(demoCourses[0].id);
   401→  const [selectedLessonId, setSelectedLessonId] = useState<string | null>(null);
   402→  const [expandedModules, setExpandedModules] = useState<Set<string>>(
   403→    new Set(demoCourses[0].modules?.map((m) => m.id) || [])
   404→  );
   405→
   406→  const selectedCourse = useMemo(
   407→    () => demoCourses.find((c) => c.id === selectedCourseId) || demoCourses[0],
   408→    [selectedCourseId]
   409→  );
   410→
   411→  const selectedLesson = useMemo(() => {
   412→    if (!selectedLessonId) return null;
   413→    for (const mod of selectedCourse.modules || []) {
   414→      const found = mod.lessons?.find((l) => l.id === selectedLessonId);
   415→      if (found) return found;
   416→    }
   417→    return null;
   418→  }, [selectedLessonId, selectedCourse]);
   419→
   420→  const toggleModule = (moduleId: string) => {
   421→    setExpandedModules((prev) => {
   422→      const next = new Set(prev);
   423→      if (next.has(moduleId)) next.delete(moduleId);
   424→      else next.add(moduleId);
   425→      return next;
   426→    });
   427→  };
   428→
   429→  const totalLessons = useMemo(
   430→    () =>
   431→      (selectedCourse.modules || []).reduce(
   432→        (acc, m) => acc + (m.lessons?.length || 0),
   433→        0
   434→      ),
   435→    [selectedCourse]
   436→  );
   437→
   438→  const totalDurationSec = useMemo(
   439→    () =>
   440→      (selectedCourse.modules || []).reduce(
   441→        (acc, m) =>
   442→          acc +
   443→          (m.lessons || []).reduce(
   444→            (a, l) => a + (l.videoDuration || 0),
   445→            0
   446→          ),
   447→        0
   448→      ),
   449→    [selectedCourse]
   450→  );
   451→
   452→  return (
   453→    <div className="space-y-4">
   454→      {/* Course selector */}
   455→      <div className="flex items-center gap-3">
   456→        <Label className="text-sm font-medium shrink-0">Course:</Label>
   457→        <Select value={selectedCourseId} onValueChange={(v) => { setSelectedCourseId(v); setSelectedLessonId(null); }}>
   458→          <SelectTrigger className="w-full sm:w-[320px]">
   459→            <SelectValue />
   460→          </SelectTrigger>
   461→          <SelectContent>
   462→            {demoCourses.map((c) => (
   463→              <SelectItem key={c.id} value={c.id}>
   464→                {c.title}
   465→              </SelectItem>
   466→            ))}
   467→          </SelectContent>
   468→        </Select>
   469→      </div>
   470→
   471→      {/* Two-panel layout */}
   472→      <div className="grid grid-cols-1 lg:grid-cols-[340px_1fr] gap-4 min-h-[600px]">
   473→        {/* Left panel: Module & Lesson Tree */}
   474→        <Card className="overflow-hidden">
   475→          <div className="p-4 border-b bg-muted/30">
   476→            <div className="flex items-center justify-between">
   477→              <h3 className="text-sm font-semibold text-foreground">Curriculum</h3>
   478→              <div className="flex items-center gap-1">
   479→                <Button variant="ghost" size="sm" className="h-7 text-xs gap-1 text-emerald-600 dark:text-emerald-400 hover:text-emerald-700 dark:hover:text-emerald-300">
   480→                  <Plus className="h-3 w-3" /> Module
   481→                </Button>
   482→              </div>
   483→            </div>
   484→            <div className="flex items-center gap-3 mt-2 text-xs text-muted-foreground">
   485→              <span>{(selectedCourse.modules || []).length} modules</span>
   486→              <span>{totalLessons} lessons</span>
   487→              <span>{formatDuration(totalDurationSec)}</span>
   488→            </div>
   489→          </div>
   490→
   491→          <ScrollArea className="h-[540px]">
   492→            <div className="p-2">
   493→              {(selectedCourse.modules || []).length === 0 ? (
   494→                <div className="flex flex-col items-center justify-center py-12 text-muted-foreground">
   495→                  <Layers className="h-8 w-8 mb-2 opacity-40" />
   496→                  <p className="text-sm">No modules yet</p>
   497→                  <p className="text-xs">Add a module to start building</p>
   498→                </div>
   499→              ) : (
   500→                (selectedCourse.modules || [])
   501→                  .sort((a, b) => a.orderIndex - b.orderIndex)
   502→                  .map((mod) => (
   503→                    <ModuleTreeItem
   504→                      key={mod.id}
   505→                      module={mod}
   506→                      isExpanded={expandedModules.has(mod.id)}
   507→                      onToggle={() => toggleModule(mod.id)}
   508→                      selectedLessonId={selectedLessonId}
   509→                      onSelectLesson={setSelectedLessonId}
   510→                    />
   511→                  ))
   512→              )}
   513→            </div>
   514→          </ScrollArea>
   515→        </Card>
   516→
   517→        {/* Right panel: Content Editor */}
   518→        <Card className="overflow-hidden">
   519→          {selectedLesson ? (
   520→            <LessonEditor lesson={selectedLesson} />
   521→          ) : (
   522→            <div className="flex flex-col items-center justify-center h-full min-h-[400px] text-muted-foreground">
   523→              <BookOpen className="h-12 w-12 mb-4 opacity-30" />
   524→              <p className="text-lg font-medium">Select a lesson</p>
   525→              <p className="text-sm">Choose a lesson from the curriculum tree to edit its content</p>
   526→            </div>
   527→          )}
   528→        </Card>
   529→      </div>
   530→    </div>
   531→  );
   532→}

   534→function ModuleTreeItem({
   535→  module,
   536→  isExpanded,
   537→  onToggle,
   538→  selectedLessonId,
   539→  onSelectLesson,
   540→}: {
   541→  module: Module;
   542→  isExpanded: boolean;
   543→  onToggle: () => void;
   544→  selectedLessonId: string | null;
   545→  onSelectLesson: (id: string) => void;
   546→}) {
   547→  const lessonCount = module.lessons?.length || 0;
