   453→// ─── Profile Completion Ring Component ───────────────────
   454→function ProfileCompletionRing({ percentage, size = 64 }: { percentage: number; size?: number }) {
   455→  const strokeWidth = 4;
   456→  const radius = (size - strokeWidth) / 2;
   457→  const circumference = 2 * Math.PI * radius;
   458→  const offset = circumference - (percentage / 100) * circumference;
   459→
   460→  return (
   461→    <div className="relative" style={{ width: size, height: size }}>
   462→      <svg width={size} height={size} className="-rotate-90">
   463→        <circle
   464→          cx={size / 2}
   465→          cy={size / 2}
   466→          r={radius}
   467→          fill="none"
   468→          stroke="hsl(var(--muted))"
   469→          strokeWidth={strokeWidth}
   470→        />
   471→        <motion.circle
   472→          cx={size / 2}
   473→          cy={size / 2}
   474→          r={radius}
   475→          fill="none"
   476→          stroke="url(#completionGradient)"
   477→          strokeWidth={strokeWidth}
   478→          strokeLinecap="round"
   479→          strokeDasharray={circumference}
   480→          initial={{ strokeDashoffset: circumference }}
   481→          animate={{ strokeDashoffset: offset }}
   482→          transition={{ duration: 1.5, ease: 'easeOut', delay: 0.5 }}
   483→        />
   484→        <defs>
   485→          <linearGradient id="completionGradient" x1="0%" y1="0%" x2="100%" y2="0%">
   486→            <stop offset="0%" stopColor="#10b981" />
   487→            <stop offset="100%" stopColor="#06b6d4" />
   488→          </linearGradient>
