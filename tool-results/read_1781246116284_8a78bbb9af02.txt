   201→function timeAgo(dateStr: string) {
   202→  const now = new Date();
   203→  const date = new Date(dateStr);
   204→  const diffMs = now.getTime() - date.getTime();
   205→  const diffMins = Math.floor(diffMs / 60000);
   206→  if (diffMins < 60) return `${diffMins}m ago`;
   207→  const diffHours = Math.floor(diffMins / 60);
   208→  if (diffHours < 24) return `${diffHours}h ago`;
   209→  const diffDays = Math.floor(diffHours / 24);
   210→  if (diffDays < 30) return `${diffDays}d ago`;
   211→  return `${Math.floor(diffDays / 30)}mo ago`;
   212→}
