   235→// ============================================================
   236→// Typing Animation Component
   237→// ============================================================
   238→function TypingText({ text, speed = 35, delay = 0 }: { text: string; speed?: number; delay?: number }) {
   239→  const [displayed, setDisplayed] = useState('');
   240→  const [started, setStarted] = useState(false);
   241→
   242→  useEffect(() => {
   243→    const timer = setTimeout(() => setStarted(true), delay);
   244→    return () => clearTimeout(timer);
   245→  }, [delay]);
   246→
   247→  useEffect(() => {
   248→    if (!started) return;
   249→    if (displayed.length < text.length) {
   250→      const timer = setTimeout(() => {
   251→        setDisplayed(text.slice(0, displayed.length + 1));
   252→      }, speed);
   253→      return () => clearTimeout(timer);
   254→    }
   255→  }, [displayed, text, speed, started]);
   256→
   257→  return (
   258→    <span>
   259→      {displayed}
   260→      {displayed.length < text.length && started && (
   261→        <span className="inline-block w-0.5 h-5 bg-emerald-500 ml-0.5 animate-pulse align-middle" />
   262→      )}
   263→    </span>
   264→  );
   265→}

   270→function RotatingText({ phrases, interval = 3000 }: { phrases: string[]; interval?: number }) {
   271→  const [index, setIndex] = useState(0);
   272→  const [direction, setDirection] = useState(1);
   273→
   274→  useEffect(() => {
   275→    const timer = setInterval(() => {
   276→      setDirection(1);
   277→      setIndex((prev) => (prev + 1) % phrases.length);
   278→    }, interval);
   279→    return () => clearInterval(timer);
   280→  }, [phrases.length, interval]);
   281→
   282→  return (
   283→    <AnimatePresence mode="wait" custom={direction}>
   284→      <motion.span
   285→        key={phrases[index]}
   286→        custom={direction}
   287→        initial={{ y: 20, opacity: 0 }}
   288→        animate={{ y: 0, opacity: 1 }}
   289→        exit={{ y: -20, opacity: 0 }}
   290→        transition={{ duration: 0.4, ease: 'easeInOut' }}
   291→        className="inline-block bg-gradient-to-r from-emerald-600 via-emerald-500 to-teal-500 bg-clip-text text-transparent font-semibold"
   292→      >
   293→        {phrases[index]}

   352→    const dist = Math.sqrt(dx * dx + dy * dy);
   353→    if (dist < 25) {
   354→      return [{ x1: p1.x, y1: p1.y, x2: p2.x, y2: p2.y, opacity: Math.max(0, 1 - dist / 25) * 0.15 }];
   355→    }
   356→    return [];

   476→// ============================================================
   477→// Floating Dashboard Mockup (Hero) — with 3D perspective tilt on mouse move
   478→// ============================================================
   479→function DashboardMockup() {
   480→  const mockupRef = useRef<HTMLDivElement>(null);
   481→  const [tilt, setTilt] = useState({ rotateX: 0, rotateY: 0 });
   482→
   483→  const handleMouseMove = useCallback((e: React.MouseEvent) => {
   484→    if (!mockupRef.current) return;
   485→    const rect = mockupRef.current.getBoundingClientRect();
   486→    const x = (e.clientX - rect.left) / rect.width - 0.5;
   487→    const y = (e.clientY - rect.top) / rect.height - 0.5;
   488→    setTilt({ rotateX: -y * 6, rotateY: x * 6 });
   489→  }, []);
   490→
   491→  const handleMouseLeave = useCallback(() => {
   492→    setTilt({ rotateX: 0, rotateY: 0 });
   493→  }, []);
   494→
   495→  return (
   496→    <motion.div
   497→      className="relative mx-auto mt-12 max-w-4xl"
   498→      initial={{ opacity: 0, y: 40 }}
   499→      animate={{ opacity: 1, y: 0 }}
   500→      transition={{ duration: 1, delay: 0.6, ease: 'easeOut' }}
   501→    >
   502→      <motion.div
   503→        animate={{ y: [0, -8, 0] }}
   504→        transition={{ duration: 5, repeat: Infinity, ease: 'easeInOut' }}
   505→        className="relative"
   506→      >
   507→        <div
   508→          ref={mockupRef}
   509→          onMouseMove={handleMouseMove}
   510→          onMouseLeave={handleMouseLeave}
   511→          style={{
   512→            perspective: 1200,
   513→          }}
   514→        >
   515→          <motion.div
   516→            animate={{
   517→              rotateX: tilt.rotateX,
   518→              rotateY: tilt.rotateY,
   519→            }}
   520→            transition={{ type: 'spring', stiffness: 200, damping: 20 }}
   521→            style={{ transformStyle: 'preserve-3d' }}
   522→          >
   523→            {/* Glow effect behind mockup */}
   524→            <div className="absolute -inset-4 bg-gradient-to-r from-emerald-500/20 via-violet-500/20 to-emerald-500/20 rounded-2xl blur-2xl" />
   525→
   526→            {/* Mockup Card */}
   527→            <div className="relative rounded-xl border border-border/60 bg-card/90 backdrop-blur-sm shadow-2xl overflow-hidden">
   528→              {/* Window chrome */}

   683→// ============================================================
   684→// Animated Counter Hook
   685→// ============================================================
   686→function useAnimatedCounter(target: number, duration: number = 800) {
   687→  const [count, setCount] = useState(0);
   688→  const prevTarget = useRef(0);
   689→
   690→  useEffect(() => {
   691→    if (target === prevTarget.current) return;
   692→    prevTarget.current = target;
   693→
   694→    const start = count;
   695→    const diff = target - start;
   696→    const startTime = performance.now();
   697→
   698→    const animate = (now: number) => {
   699→      const elapsed = now - startTime;
   700→      const progress = Math.min(elapsed / duration, 1);
   701→      // easeOutCubic
   702→      const eased = 1 - Math.pow(1 - progress, 3);
   703→      setCount(Math.round(start + diff * eased));
   704→      if (progress < 1) {
   705→        requestAnimationFrame(animate);
   706→      }
   707→    };
   708→
   709→    requestAnimationFrame(animate);
   710→  }, [target, duration]);
   711→
   712→  return count;
   713→}

  1807→  const [count, setCount] = useState(0);
  1808→  const hasAnimated = useRef(false);
  1809→
  1810→  useEffect(() => {
  1811→    if (!isInView || hasAnimated.current) return;
  1812→    hasAnimated.current = true;
  1813→
  1814→    const duration = 1200;
  1815→    const startTime = performance.now();
  1816→
  1817→    const animate = (now: number) => {
  1818→      const elapsed = now - startTime;
  1819→      const progress = Math.min(elapsed / duration, 1);
  1820→      const eased = 1 - Math.pow(1 - progress, 3);
  1821→      setCount(Math.round(target * eased));
  1822→      if (progress < 1) {
  1823→        requestAnimationFrame(animate);
  1824→      }
  1825→    };
  1826→
  1827→    requestAnimationFrame(animate);
  1828→  }, [target, isInView]);
  1829→
  1830→  return <span>{count}</span>;
  1831→}
