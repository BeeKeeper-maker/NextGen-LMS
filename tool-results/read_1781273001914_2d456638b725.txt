   116→/** Map API review data to the CourseReview display type */
   117→function mapApiReviewToDisplay(apiReview: any): CourseReview {
   118→  const authorName = apiReview.author?.name || 'Unknown User';
   119→  const initials = authorName.split(' ').map((n: string) => n[0]).join('').toUpperCase().slice(0, 2);
   120→  // Extract title from content: first sentence or first 60 chars
   121→  const firstSentence = apiReview.content?.split(/[.!\n]/)[0] || '';
   122→  const title = firstSentence.length > 60 ? firstSentence.slice(0, 57) + '...' : firstSentence || 'Course Review';
   123→  // Map API status to display status
   124→  let displayStatus: CourseReview['status'] = 'published';
   125→  if (apiReview.status === 'flagged' || apiReview.flagged) displayStatus = 'flagged';
   126→  else if (apiReview.status === 'rejected') displayStatus = 'hidden';
   127→  else if (apiReview.status === 'pending' || apiReview.status === 'approved') displayStatus = 'published';
   128→  return {
   129→    id: apiReview.id,
   130→    courseId: apiReview.courseId,
   131→    userId: apiReview.authorId,
   132→    userName: authorName,
   133→    userAvatar: initials,
   134→    rating: apiReview.rating,
   135→    title,
   136→    content: apiReview.content || '',
   137→    date: apiReview.createdAt,
   138→    helpfulCount: 0,
   139→    isHelpful: false,
   140→    courseProgress: 0,
   141→    isVerifiedPurchase: (apiReview.author?._count?.enrollments ?? 0) > 0,
   142→    tags: [],
   143→    status: displayStatus,
   144→    instructorReply: apiReview.adminResponse
   145→      ? { instructorName: 'Admin', content: apiReview.adminResponse, date: apiReview.updatedAt }
   146→      : undefined,
   147→    isOwnReview: false,
   148→  };
   149→}

   223→function formatDuration(seconds: number): string {
   224→  const mins = Math.floor(seconds / 60);
   225→  if (mins < 60) return `${mins}m`;
   226→  const hrs = Math.floor(mins / 60);
   227→  const rem = mins % 60;
   228→  return rem > 0 ? `${hrs}h ${rem}m` : `${hrs}h`;
   229→}

   480→function NewCourseDialog() {
   481→  const tenantId = useAppStore((s) => s.currentTenant?.id) || 'demo-tenant-1';
   482→  const createCourseMutation = useCreateCourse();
   483→  const [open, setOpen] = useState(false);
   484→  const [title, setTitle] = useState('');
   485→  const [description, setDescription] = useState('');
   486→  const [category, setCategory] = useState('');
   487→  const [level, setLevel] = useState('beginner');
   488→  const [price, setPrice] = useState('0');
   489→  const [compareAtPrice, setCompareAtPrice] = useState('');
   490→  const [errors, setErrors] = useState<Record<string, string>>({});
   491→  const [touched, setTouched] = useState<Record<string, boolean>>({});
   492→
   493→  const validate = (): boolean => {
   494→    const errs = validateFields({
   495→      title: [required(title, 'Title'), minLength(title, 3, 'Title'), maxLength(title, 200, 'Title')],
   496→      description: [required(description, 'Description'), minLength(description, 10, 'Description')],
   497→      category: [required(category, 'Category')],
   498→      level: [required(level, 'Level')],
   499→      price: [numeric(price, 'Price'), min(price, 0, 'Price')],
   500→    });
   501→    setErrors(errs);
   502→    setTouched({ title: true, description: true, category: true, level: true, price: true });
   503→    return Object.keys(errs).length === 0;
   504→  };
   505→
   506→  const handleBlur = (field: string) => {
   507→    setTouched(prev => ({ ...prev, [field]: true }));
   508→    // Validate just this field on blur
   509→    const errs = validateFields({
   510→      title: [required(title, 'Title'), minLength(title, 3, 'Title'), maxLength(title, 200, 'Title')],
   511→      description: [required(description, 'Description'), minLength(description, 10, 'Description')],
   512→      category: [required(category, 'Category')],
   513→      level: [required(level, 'Level')],
   514→      price: [numeric(price, 'Price'), min(price, 0, 'Price')],
   515→    });
   516→    // Only show error for touched field
   517→    setErrors(prev => {
   518→      const next = { ...prev };
   519→      if (touched[field] || errs[field]) {
   520→        next[field] = errs[field] || '';
   521→        if (!next[field]) delete next[field];
   522→      }
   523→      return next;
   524→    });
   525→  };
   526→
   527→  const handleCreate = () => {
   528→    if (!validate()) return;
   529→    createCourseMutation.mutate(
   530→      {
   531→        tenantId,
   532→        slug: slugify(title.trim()),
   533→        title: title.trim(),
   534→        description: description.trim(),
   535→        category: category || 'Web Development',
   536→        level: level || 'beginner',
   537→        price: parseFloat(price) || 0,
   538→        compareAtPrice: compareAtPrice ? parseFloat(compareAtPrice) : null,
   539→        isPublished: false,
   540→        isFeatured: false,
   541→      },
   542→      {
   543→        onSuccess: () => {
   544→          setOpen(false);
   545→          setTitle('');
   546→          setDescription('');
   547→          setCategory('');
   548→          setLevel('beginner');
   549→          setPrice('0');
   550→          setCompareAtPrice('');
   551→          setErrors({});
   552→          setTouched({});
   553→        },
   554→      }

   641→function CourseBuilderTab() {
   642→  const { courses } = useCoursesData();
   643→  const [selectedCourseId, setSelectedCourseId] = useState<string>(courses[0]?.id || '');
   644→  const [selectedLessonId, setSelectedLessonId] = useState<string | null>(null);
   645→  const [expandedModules, setExpandedModules] = useState<Set<string>>(
   646→    new Set(courses[0]?.modules?.map((m) => m.id) || [])
   647→  );
   648→
   649→  const selectedCourse = useMemo(
   650→    () => courses.find((c) => c.id === selectedCourseId) || courses[0],
   651→    [selectedCourseId, courses]
   652→  );
   653→
   654→  const selectedLesson = useMemo(() => {
   655→    if (!selectedLessonId) return null;
   656→    for (const mod of selectedCourse.modules || []) {
   657→      const found = mod.lessons?.find((l) => l.id === selectedLessonId);
   658→      if (found) return found;
   659→    }
   660→    return null;
   661→  }, [selectedLessonId, selectedCourse]);
   662→
   663→  const toggleModule = (moduleId: string) => {
   664→    setExpandedModules((prev) => {
   665→      const next = new Set(prev);
   666→      if (next.has(moduleId)) next.delete(moduleId);
   667→      else next.add(moduleId);
   668→      return next;
   669→    });
   670→  };
   671→
   672→  const totalLessons = useMemo(
   673→    () =>
   674→      (selectedCourse.modules || []).reduce(
   675→        (acc, m) => acc + (m.lessons?.length || 0),
   676→        0
   677→      ),
   678→    [selectedCourse]
   679→  );
   680→
   681→  const totalDurationSec = useMemo(
   682→    () =>
   683→      (selectedCourse.modules || []).reduce(
   684→        (acc, m) =>
   685→          acc +
   686→          (m.lessons || []).reduce(
   687→            (a, l) => a + (l.videoDuration || 0),
   688→            0
   689→          ),
   690→        0
   691→      ),
   692→    [selectedCourse]
   693→  );
   694→
   695→  return (
   696→    <div className="space-y-4">
   697→      {/* Course selector */}
   698→      <div className="flex items-center gap-3">
   699→        <Label className="text-sm font-medium shrink-0">Course:</Label>
   700→        <Select value={selectedCourseId} onValueChange={(v) => { setSelectedCourseId(v); setSelectedLessonId(null); }}>
   701→          <SelectTrigger className="w-full sm:w-[320px]">
   702→            <SelectValue />
   703→          </SelectTrigger>
   704→          <SelectContent>
   705→            {courses.map((c) => (
   706→              <SelectItem key={c.id} value={c.id}>
   707→                {c.title}
   708→              </SelectItem>
   709→            ))}
   710→          </SelectContent>
   711→        </Select>
   712→      </div>
   713→
   714→      {/* Two-panel layout */}
   715→      <div className="grid grid-cols-1 lg:grid-cols-[340px_1fr] gap-4 min-h-[600px]">
   716→        {/* Left panel: Module & Lesson Tree */}
   717→        <Card className="overflow-hidden">
   718→          <div className="p-4 border-b bg-muted/30">
   719→            <div className="flex items-center justify-between">
   720→              <h3 className="text-sm font-semibold text-foreground">Curriculum</h3>
   721→              <div className="flex items-center gap-1">
   722→                <Button variant="ghost" size="sm" className="h-7 text-xs gap-1 text-emerald-600 dark:text-emerald-400 hover:text-emerald-700 dark:hover:text-emerald-300">
   723→                  <Plus className="h-3 w-3" /> Module
   724→                </Button>
   725→              </div>
   726→            </div>
   727→            <div className="flex items-center gap-3 mt-2 text-xs text-muted-foreground">
   728→              <span>{(selectedCourse.modules || []).length} modules</span>

   737→                <div className="flex flex-col items-center justify-center py-12 text-muted-foreground">
