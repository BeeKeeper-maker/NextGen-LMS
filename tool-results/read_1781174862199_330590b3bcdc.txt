    91→    91→function timeAgo(dateStr: string) {
    92→    92→  const now = new Date();
    93→    93→  const date = new Date(dateStr);
    94→    94→  const diffMs = now.getTime() - date.getTime();
    95→    95→  const diffMins = Math.floor(diffMs / 60000);
    96→    96→  if (diffMins < 60) return `${diffMins}m ago`;
    97→    97→  const diffHours = Math.floor(diffMins / 60);
    98→    98→  if (diffHours < 24) return `${diffHours}h ago`;
    99→    99→  const diffDays = Math.floor(diffHours / 24);
   100→   100→  if (diffDays < 30) return `${diffDays}d ago`;
   101→   101→  return `${Math.floor(diffDays / 30)}mo ago`;
   102→   102→}
