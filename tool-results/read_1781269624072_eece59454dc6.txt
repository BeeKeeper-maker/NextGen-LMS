   126→// ─── Format duration ───────────────────────────────────────
   127→function formatDuration(seconds?: number): string {
   128→  if (!seconds) return '';
   129→  const mins = Math.floor(seconds / 60);
   130→  if (mins < 60) return `${mins} min`;
   131→  const hours = Math.floor(mins / 60);
   132→  const remainMins = mins % 60;
   133→  return `${hours}h ${remainMins}m`;
   134→}

   169→function useLessonNotes(courseId: string | undefined) {
   170→  const storageKey = `lms-notes-${courseId || 'default'}`;
   171→
   172→  const [notes, setNotes] = useState<Note[]>(() => {
   173→    if (typeof window === 'undefined') return [];
   174→    try {
   175→      const stored = localStorage.getItem(storageKey);
   176→      return stored ? JSON.parse(stored) : [];
   177→    } catch {
   178→      return [];
   179→    }
   180→  });
   181→
   182→  const persistNotes = useCallback((updated: Note[]) => {
   183→    setNotes(updated);
   184→    if (typeof window !== 'undefined') {
   185→      try {
   186→        localStorage.setItem(storageKey, JSON.stringify(updated));
   187→      } catch {
   188→        // ignore quota errors
   189→      }
   190→    }
   191→  }, [storageKey]);
   192→
   193→  const addNote = useCallback((note: Omit<Note, 'id' | 'createdAt' | 'updatedAt'>) => {
   194→    const newNote: Note = {
   195→      ...note,
   196→      id: `note-${Date.now()}-${Math.random().toString(36).slice(2, 7)}`,
   197→      createdAt: new Date().toISOString(),
   198→      updatedAt: new Date().toISOString(),
   199→    };
   200→    persistNotes([...notes, newNote]);
   201→    return newNote;
   202→  }, [notes, persistNotes]);
   203→
   204→  const updateNote = useCallback((id: string, updates: Partial<Note>) => {
   205→    const updated = notes.map(n =>
   206→      n.id === id ? { ...n, ...updates, updatedAt: new Date().toISOString() } : n
   207→    );
   208→    persistNotes(updated);
   209→  }, [notes, persistNotes]);
   210→
   211→  const deleteNote = useCallback((id: string) => {
   212→    persistNotes(notes.filter(n => n.id !== id));
   213→  }, [notes, persistNotes]);
   214→
   215→  return { notes, addNote, updateNote, deleteNote };
   216→}

   246→// ─── Time ago helper ──────────────────────────────────────
   247→function timeAgo(dateStr: string): string {
   248→  const seconds = Math.floor((Date.now() - new Date(dateStr).getTime()) / 1000);
   249→  if (seconds < 60) return 'just now';
   250→  const minutes = Math.floor(seconds / 60);
   251→  if (minutes < 60) return `${minutes}m ago`;
   252→  const hours = Math.floor(minutes / 60);
   253→  if (hours < 24) return `${hours}h ago`;
   254→  const days = Math.floor(hours / 24);
   255→  if (days < 30) return `${days}d ago`;
   256→  const months = Math.floor(days / 30);
   257→  return `${months}mo ago`;
   258→}

   473→  const { data: discussionsData, isLoading } = useLessonDiscussions({ courseId });

   488→  // Separate top-level questions from replies & build reply map
   489→  const { topLevelDiscussions, repliesMap } = useMemo(() => {
   490→    const top: any[] = [];
   491→    const map: Record<string, any[]> = {};
   492→    discussions.forEach((d: any) => {
   493→      if (d.parentId) {
   494→        if (!map[d.parentId]) map[d.parentId] = [];

   497→        top.push(d);

   502→
   503→  const unansweredCount = topLevelDiscussions.filter((q: any) => !(repliesMap[q.id]?.length > 0)).length;
   504→
   505→  const sortedQuestions = [...topLevelDiscussions].sort((a: any, b: any) => {
   506→    if (sortBy === 'recent') return new Date(b.createdAt).getTime() - new Date(a.createdAt).getTime();

   602→            placeholder="Search discussions..."

   662→                    </Button>

   664→                </div>
   665→              </CardContent>

   668→        )}

   671→      {/* Questions List */}

   693→                      <div className="flex flex-col items-center gap-1 pt-1">
   694→                        <button className="p-1 rounded hover:bg-muted/50 transition-colors text-muted-foreground hover:text-foreground">
   695→                          <ChevronUp className="h-4 w-4" />
   696→                        </button>
   697→                        <span className="text-sm font-semibold text-foreground">{qReplies.length}</span>
   698→                        <button className="p-1 rounded hover:bg-muted/50 transition-colors text-muted-foreground hover:text-foreground">
   699→                          <ChevronDown className="h-4 w-4" />
   700→                        </button>
   701→                      </div>
   702→
   703→                      {/* Content */}
   704→                      <div className="flex-1 min-w-0" onClick={() => setExpandedQuestion(expandedQuestion === question.id ? null : question.id)}>
   705→                        <div className="flex items-center gap-2 mb-1 flex-wrap">

   707→                            <Badge className="bg-emerald-100 text-emerald-700 dark:bg-emerald-900/30 dark:text-emerald-400 text-[10px] gap-0.5">
   708→                              <CheckCircle2 className="h-3 w-3" /> Answered
   709→                            </Badge>
   710→                          )}
   711→                          {question.lesson?.title && (
   712→                            <Badge variant="outline" className="text-[10px] gap-0.5">
   713→                              <BookOpen className="h-2.5 w-2.5" /> {question.lesson.title}
   714→                            </Badge>
   715→                          )}
   716→                          {question.isPinned && (

   718→                              <Pin className="h-2.5 w-2.5" /> Pinned
   719→                            </Badge>
   720→                          )}
   721→                        </div>
   722→                        <p className="text-sm text-muted-foreground line-clamp-3">{question.content}</p>
   723→                        <div className="flex items-center gap-3 mt-2">
   724→                          <span className="text-xs text-muted-foreground">· {timeAgo(question.createdAt)}</span>
   725→                          <span className="flex items-center gap-1 text-xs text-muted-foreground">

   728→                        </div>
   729→                      </div>
   730→                    </div>
   731→
   732→                    {/* Expanded Replies */}
   733→                    <AnimatePresence>
   734→                      {expandedQuestion === question.id && (
   735→                        <motion.div
   736→                          initial={{ height: 0, opacity: 0 }}
   737→                          animate={{ height: 'auto', opacity: 1 }}
   738→                          exit={{ height: 0, opacity: 0 }}
   739→                          transition={{ duration: 0.2 }}
   740→                          className="overflow-hidden"
   741→                        >
   742→                          <div className="mt-4 pt-4 border-t border-border/50 space-y-3">
   743→                            {qReplies.map((reply: any) => (
   744→                              <div key={reply.id} className={cn('ml-8 p-3 rounded-lg bg-muted/30')}>
   745→                                <p className="text-xs text-muted-foreground leading-relaxed">{reply.content}</p>
   746→                                <div className="flex items-center gap-3 mt-2">
   747→                                  <span className="text-[10px] text-muted-foreground">{timeAgo(reply.createdAt)}</span>
   748→                                  <button className="flex items-center gap-1 text-[10px] text-muted-foreground hover:text-foreground transition-colors">
   749→                                    <ThumbsUp className="h-3 w-3" />
   750→                                  </button>
   751→                                </div>
   752→                              </div>
   753→                            ))}
   754→
   755→                            {/* Reply input */}
   756→                            <div className="ml-8 flex gap-2">
   757→                              <div className="flex-1">
   758→                                <Input
   759→                                  placeholder="Write a reply..."
   760→                                  value={replyingTo === question.id ? replyContent : ''}
   761→                                  onChange={(e) => { setReplyingTo(question.id); setReplyContent(e.target.value); if (discussionErrors.replyContent) setDiscussionErrors({}); }}
   762→                                  className={`h-8 text-xs ${discussionErrors.replyContent && replyingTo === question.id ? 'border-destructive' : ''}`}
   763→                                  onFocus={() => setReplyingTo(question.id)}

   768→                                size="sm"
   769→                                className="h-8 gap-1 bg-emerald-600 hover:bg-emerald-700 shrink-0"

   774→                              </Button>
   775→                            </div>
   776→                          </div>
   777→                        </motion.div>
   778→                      )}
   779→                    </AnimatePresence>
   780→                  </CardContent>
   781→                </Card>
   782→              </motion.div>
   783→            );
   784→          })}
   785→        </AnimatePresence>
   786→      </div>
   787→
   788→      {filteredQuestions.length === 0 && (
   789→        <div className="text-center py-8">
   790→          <MessageCircleQuestion className="h-12 w-12 text-muted-foreground/30 mx-auto mb-3" />

   792→        </div>
   793→      )}
   794→
   795→      <Button variant="outline" className="w-full">View All Discussions</Button>
   796→    </div>
   797→  );

   800→// ─── Q&A Discussion Tab ────────────────────────────────────
   801→function QADiscussionTab({ courseId, userId, modules }: { courseId: string; userId: string; modules: Module[] }) {
   802→  const { data: discussionsData, isLoading } = useLessonDiscussions({ courseId });
   803→  const createDiscussion = useCreateDiscussion();
   804→
   805→  const [selectedLessonId, setSelectedLessonId] = useState<string>('all');
   806→  const [searchQuery, setSearchQuery] = useState('');
   807→  const [sortBy, setSortBy] = useState<'newest' | 'voted' | 'unresolved' | 'resolved'>('newest');
   808→  const [filterBy, setFilterBy] = useState<'all' | 'my' | 'unresolved' | 'resolved'>('all');
   809→  const [selectedQuestion, setSelectedQuestion] = useState<any | null>(null);
   810→  const [showAskDialog, setShowAskDialog] = useState(false);
   811→  const [askContent, setAskContent] = useState('');
   812→  const [askLessonId, setAskLessonId] = useState<string>(selectedLessonId === 'all' ? '' : selectedLessonId);
   813→  const [answerContent, setAnswerContent] = useState('');
   814→  const [qaErrors, setQaErrors] = useState<Record<string, string>>({});
   815→  const [expandedModules, setExpandedModules] = useState<Set<string>>(new Set(modules.map(m => m.id)));
   816→
   817→  const discussions = discussionsData?.discussions || [];
   818→
   819→  // Separate top-level from replies & build reply map
   820→  const { qaTopLevel, repliesMap } = useMemo(() => {
   821→    const top: any[] = [];
   822→    const map: Record<string, any[]> = {};
   823→    discussions.forEach((d: any) => {
   824→      if (d.parentId) {
   825→        if (!map[d.parentId]) map[d.parentId] = [];
   826→        map[d.parentId].push(d);
   827→      } else {
   828→        top.push(d);
   829→      }
   830→    });
   831→    return { qaTopLevel: top, repliesMap: map };
   832→  }, [discussions]);
   833→
   834→  // Build lesson map from modules
   835→  const lessonMap = new Map<string, { title: string; moduleId: string; moduleTitle: string }>();
   836→  modules.forEach(mod => {
   837→    (mod.lessons || []).forEach(lesson => {
   838→      lessonMap.set(lesson.id, { title: lesson.title, moduleId: mod.id, moduleTitle: mod.title });
   839→    });
   840→  });
   841→
   842→  // Filter questions
   843→  const filteredQuestions = qaTopLevel.filter((q: any) => {
   844→    if (selectedLessonId !== 'all' && q.lessonId !== selectedLessonId) return false;
   845→    if (filterBy === 'my' && q.userId !== userId) return false;
   846→    const qReplies = repliesMap[q.id] || [];
   847→    const isResolved = qReplies.length > 0;
   848→    if (filterBy === 'unresolved' && isResolved) return false;
   849→    if (filterBy === 'resolved' && !isResolved) return false;
   850→    if (searchQuery) {
   851→      const lq = searchQuery.toLowerCase();
   852→      if (!q.content.toLowerCase().includes(lq)) return false;
   853→    }
   854→    return true;
   855→  });
   856→
   857→  // Sort questions
   858→  const sortedQuestions = [...filteredQuestions].sort((a: any, b: any) => {
   859→    if (sortBy === 'newest') return new Date(b.createdAt).getTime() - new Date(a.createdAt).getTime();
   860→    if (sortBy === 'voted') return (repliesMap[b.id]?.length || 0) - (repliesMap[a.id]?.length || 0);
   861→    if (sortBy === 'unresolved') {
   862→      const aResolved = (repliesMap[a.id]?.length || 0) > 0;
   863→      const bResolved = (repliesMap[b.id]?.length || 0) > 0;
   864→      return aResolved === bResolved ? 0 : aResolved ? 1 : -1;
   865→    }
   866→    if (sortBy === 'resolved') {
   867→      const aResolved = (repliesMap[a.id]?.length || 0) > 0;
   868→      const bResolved = (repliesMap[b.id]?.length || 0) > 0;
   869→      return aResolved === bResolved ? 0 : aResolved ? -1 : 1;
   870→    }
   871→    return 0;
   872→  });
   873→
   874→  const toggleModule = (moduleId: string) => {
   875→    setExpandedModules(prev => {
   876→      const next = new Set(prev);
   877→      if (next.has(moduleId)) next.delete(moduleId);
   878→      else next.add(moduleId);
   879→      return next;
   880→    });
   881→  };
   882→
   883→  // Count questions per lesson
   884→  const questionCountsByLesson = new Map<string, number>();
   885→  const unresolvedCountsByLesson = new Map<string, number>();
   886→  qaTopLevel.forEach((q: any) => {
   887→    questionCountsByLesson.set(q.lessonId, (questionCountsByLesson.get(q.lessonId) || 0) + 1);
   888→    if (!(repliesMap[q.id]?.length > 0)) unresolvedCountsByLesson.set(q.lessonId, (unresolvedCountsByLesson.get(q.lessonId) || 0) + 1);
   889→  });
   890→
   891→  const unresolvedTotal = qaTopLevel.filter((q: any) => !(repliesMap[q.id]?.length > 0)).length;
   892→
   893→  // Render content with markdown
   894→  const renderContent = (content: string) => {
   895→    const parts = content.split(/(```[\s\S]*?```)/g);
   896→    return parts.map((part, i) => {
   897→      if (part.startsWith('```') && part.endsWith('```')) {
   898→        const codeContent = part.slice(3, -3).replace(/^\w+\n/, '');
   899→        return (
   900→          <pre key={i} className="bg-slate-900 dark:bg-slate-950 text-green-400 p-3 rounded-lg text-xs font-mono overflow-x-auto my-2 leading-relaxed">
   901→            <code>{codeContent}</code>
   902→          </pre>
   903→        );
   904→      }
   905→      const inlineParts = part.split(/(`[^`]+`)/g);
   906→      return (
   907→        <span key={i}>
   908→          {inlineParts.map((ip, j) => {
   909→            if (ip.startsWith('`') && ip.endsWith('`')) {
   910→              return <code key={j} className="bg-slate-100 dark:bg-slate-800 text-emerald-700 dark:text-emerald-400 px-1.5 py-0.5 rounded text-xs font-mono">{ip.slice(1, -1)}</code>;
   911→            }
   912→            const boldParts = ip.split(/(\*\*[^*]+\*\*)/g);
   913→            return boldParts.map((bp, k) => {
   914→              if (bp.startsWith('**') && bp.endsWith('**')) {
   915→                return <strong key={`${j}-${k}`} className="font-semibold text-foreground">{bp.slice(2, -2)}</strong>;
   916→              }
   917→              return <span key={`${j}-${k}`}>{bp}</span>;
   918→            });
   919→          })}
   920→        </span>
   921→      );
   922→    });
   923→  };
   924→
   925→  // Handle submit question
   926→  const handleSubmitQuestion = () => {
   927→    const errs = validateFields({
   928→      askContent: [required(askContent, 'Question'), minLength(askContent, 5, 'Question')],
   929→    });
   930→    if (!askLessonId) errs.askLessonId = 'Please select a lesson';
   931→    setQaErrors(errs);
   932→    if (Object.keys(errs).length > 0) return;
   933→    createDiscussion.mutate({
   934→      lessonId: askLessonId,
   935→      userId,
   936→      content: askContent,
   937→    }, {
   938→      onSuccess: () => {
   939→        setAskContent('');
   940→        setShowAskDialog(false);
   941→        setQaErrors({});
   942→      },
   943→    });
   944→  };
   945→
   946→  // Handle submit answer
   947→  const handleSubmitAnswer = () => {
   948→    const errs = validateFields({
   949→      answerContent: [required(answerContent, 'Answer'), minLength(answerContent, 5, 'Answer')],
   950→    });
   951→    setQaErrors(errs);
   952→    if (Object.keys(errs).length > 0) return;
   953→    if (!selectedQuestion) return;
   954→    createDiscussion.mutate({
   955→      lessonId: selectedQuestion.lessonId,
   956→      userId,
   957→      content: answerContent,
   958→      parentId: selectedQuestion.id,
   959→    }, {
   960→      onSuccess: () => {
   961→        setAnswerContent('');
   962→        setQaErrors({});
   963→      },
   964→    });
   965→  };
   966→
   967→  if (isLoading) {
   968→    return (
   969→      <div className="space-y-3">
   970→        {Array.from({ length: 3 }).map((_, i) => (
   971→          <div key={i} className="rounded-xl p-4 bg-muted/30 animate-pulse space-y-3">
   972→            <div className="h-4 bg-muted rounded w-1/2" />
   973→            <div className="h-3 bg-muted rounded w-3/4" />
   974→          </div>
   975→        ))}
   976→      </div>
   977→    );
   978→  }
   979→
   980→  // ─── Thread Detail View ─────────────────────────────────
   981→  if (selectedQuestion) {
   982→    const q = selectedQuestion;
   983→    const qReplies = repliesMap[q.id] || [];
   984→    const isResolved = qReplies.length > 0;
   985→

  1691→  // Derive timeline events from progress data
  1692→  const timelineEvents = useMemo(() => {
  1693→    if (!progressData || !course) return [];
  1694→    const events: { id: string; lessonTitle: string; moduleName: string; date: string; timeSpent: number; type: 'completed' | 'started' }[] = [];

  1699→      (mod.lessons || []).forEach((lesson: any) => {
  1700→        lessonLookup.set(lesson.id, { title: lesson.title, moduleName: mod.title });
  1701→      });

  1705→      const lessonInfo = lessonLookup.get(p.lessonId);
  1706→      if (!lessonInfo) return;
  1707→
  1708→      const date = p.completedAt
  1709→        ? new Date(p.completedAt).toLocaleDateString('en-US', { month: 'short', day: 'numeric', year: 'numeric' })
  1710→        : new Date(p.updatedAt || p.createdAt).toLocaleDateString('en-US', { month: 'short', day: 'numeric', year: 'numeric' });
  1711→
  1712→      events.push({
  1713→        id: p.id,
  1714→        lessonTitle: lessonInfo.title,
  1715→        moduleName: lessonInfo.moduleName,
  1716→        date,
  1717→        timeSpent: Math.round((p.timeSpent || 0) / 60),
  1718→        type: p.status === 'completed' ? 'completed' : 'started',
  1719→      });
  1720→    });
  1721→
  1722→    // Sort by date (most recent first)
