   210→function formatDuration(seconds: number): string {
   211→  const mins = Math.floor(seconds / 60);
   212→  if (mins < 60) return `${mins}m`;
   213→  const hrs = Math.floor(mins / 60);
   214→  const rem = mins % 60;
   215→  return rem > 0 ? `${hrs}h ${rem}m` : `${hrs}h`;
   216→}

   240→function CourseCatalogTab() {
   241→  const { courses } = useCoursesData();
   242→  const [search, setSearch] = useState('');
   243→  const [categoryFilter, setCategoryFilter] = useState('all');
   244→  const [levelFilter, setLevelFilter] = useState('all');
   245→
   246→  const categories = useMemo(
   247→    () => ['all', ...Array.from(new Set(courses.map((c) => c.category).filter(Boolean)))],
   248→    [courses]
   249→  );
   250→
   251→  const levels = ['all', 'beginner', 'intermediate', 'advanced', 'expert'];
   252→
   253→  const filtered = courses.filter((c) => {
   254→    const matchSearch =
   255→      !search ||
   256→      c.title.toLowerCase().includes(search.toLowerCase()) ||
   257→      c.description?.toLowerCase().includes(search.toLowerCase());
   258→    const matchCategory = categoryFilter === 'all' || c.category === categoryFilter;
   259→    const matchLevel = levelFilter === 'all' || c.level === levelFilter;
   260→    return matchSearch && matchCategory && matchLevel;
   261→  });
   262→
   263→  return (
   264→    <div className="space-y-6">
   265→      {/* Top bar */}
   266→      <div className="flex flex-col sm:flex-row items-start sm:items-center gap-3">

   279→            <SelectValue placeholder="Category" />

   628→function CourseBuilderTab() {

   633→    new Set(courses[0]?.modules?.map((m) => m.id) || [])
   634→  );

   636→  const selectedCourse = useMemo(
   637→    () => courses.find((c) => c.id === selectedCourseId) || courses[0],
   638→    [selectedCourseId, courses]
   639→  );

   642→    if (!selectedLessonId) return null;
   643→    for (const mod of selectedCourse.modules || []) {
   644→      const found = mod.lessons?.find((l) => l.id === selectedLessonId);
   645→      if (found) return found;
   646→    }
   647→    return null;
   648→  }, [selectedLessonId, selectedCourse]);
   649→
   650→  const toggleModule = (moduleId: string) => {
   651→    setExpandedModules((prev) => {
   652→      const next = new Set(prev);
   653→      if (next.has(moduleId)) next.delete(moduleId);
   654→      else next.add(moduleId);
   655→      return next;
   656→    });
   657→  };

   659→  const totalLessons = useMemo(
   660→    () =>
   661→      (selectedCourse.modules || []).reduce(
   662→        (acc, m) => acc + (m.lessons?.length || 0),
   663→        0
   664→      ),
   665→    [selectedCourse]
   666→  );
   667→
   668→  const totalDurationSec = useMemo(
   669→    () =>
   670→      (selectedCourse.modules || []).reduce(
   671→        (acc, m) =>
   672→          acc +
   673→          (m.lessons || []).reduce(
   674→            (a, l) => a + (l.videoDuration || 0),
   675→            0
   676→          ),
   677→        0
   678→      ),
   679→    [selectedCourse]
   680→  );

   682→  return (
   683→    <div className="space-y-4">
   684→      {/* Course selector */}

   688→          <SelectTrigger className="w-full sm:w-[320px]">
   689→            <SelectValue />
   690→          </SelectTrigger>
   691→          <SelectContent>
   692→            {courses.map((c) => (
   693→              <SelectItem key={c.id} value={c.id}>

   710→                  <Plus className="h-3 w-3" /> Module
