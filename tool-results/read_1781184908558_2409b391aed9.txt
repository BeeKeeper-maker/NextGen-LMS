   101→// ─── Format duration ───────────────────────────────────────
   102→function formatDuration(seconds?: number): string {
   103→  if (!seconds) return '';
   104→  const mins = Math.floor(seconds / 60);
   105→  if (mins < 60) return `${mins} min`;
   106→  const hours = Math.floor(mins / 60);
   107→  const remainMins = mins % 60;
   108→  return `${hours}h ${remainMins}m`;
   109→}
