   142→// ---- API Data Mappers ----
   143→function mapApiPathToLearningPath(apiPath: any): LearningPath {
   144→  return {
   145→    id: apiPath.id,
   146→    name: apiPath.title,
   147→    description: apiPath.description || '',
   148→    category: apiPath.category || '',
   149→    difficulty: (apiPath.level as 'beginner' | 'intermediate' | 'advanced') || 'beginner',
   150→    status: apiPath.isPublished ? 'published' : 'draft',
   151→    thumbnailUrl: apiPath.thumbnailUrl || undefined,
   152→    courseCount: apiPath.courseCount ?? apiPath.courses?.length ?? 0,
   153→    enrolledCount: apiPath.enrolledCount ?? apiPath.enrollments?.length ?? 0,
   154→    completionRate: apiPath.completionRate ?? 0,
   155→    estimatedDuration: apiPath.estimatedDuration ?? 0,
   156→    courses: (apiPath.courses || []).map((pc: any) => ({
   157→      id: pc.courseId || pc.course?.id,
   158→      title: pc.course?.title || '',
   159→      duration: pc.course?.durationHours || 0,
   160→      level: (pc.course?.level as 'beginner' | 'intermediate' | 'advanced' | 'expert') || 'beginner',
   161→      isRequired: pc.isRequired ?? true,
   162→      prerequisiteIds: pc.prerequisiteIds ? pc.prerequisiteIds.split(',').filter(Boolean) : [],
   163→      milestone: pc.milestone || undefined,
   164→    })),
   165→    createdAt: apiPath.createdAt,
   166→  };
   167→}

   339→function PathBuilderPanel({
   340→  editingPath,
   341→  onSave,
   342→  onCancel,
   343→  availableCourses,
   344→}: {
   345→  editingPath: LearningPath | null;
   346→  onSave: (path: LearningPath) => void;
   347→  onCancel: () => void;
   348→  availableCourses: { id: string; title: string; duration: number; level: 'beginner' | 'intermediate' | 'advanced' | 'expert' }[];
   349→}) {
   350→  const [name, setName] = useState(editingPath?.name || '');
   351→  const [description, setDescription] = useState(editingPath?.description || '');
   352→  const [category, setCategory] = useState(editingPath?.category || '');
   353→  const [difficulty, setDifficulty] = useState<'beginner' | 'intermediate' | 'advanced'>(editingPath?.difficulty || 'beginner');
   354→  const [status, setStatus] = useState<'published' | 'draft'>(editingPath?.status === 'archived' ? 'draft' : (editingPath?.status as 'published' | 'draft') || 'draft');
   355→  const [courses, setCourses] = useState<PathCourse[]>(editingPath?.courses || []);
   356→  const [courseSearch, setCourseSearch] = useState('');
   357→  const [showCourseDropdown, setShowCourseDropdown] = useState(false);
   358→  const [selectedCourseForPrereq, setSelectedCourseForPrereq] = useState<string | null>(null);
   359→
   360→  const filteredAvailableCourses = availableCourses.filter(
   361→    (c) => !courses.find((pc) => pc.id === c.id) &&
   362→      c.title.toLowerCase().includes(courseSearch.toLowerCase())
   363→  );
   364→
   365→  const addCourse = (courseId: string) => {
   366→    const course = availableCourses.find((c) => c.id === courseId);
   367→    if (!course) return;
   368→    const newCourse: PathCourse = {
   369→      id: course.id,
   370→      title: course.title,
   371→      duration: course.duration,
   372→      level: course.level,
   373→      isRequired: true,
   374→      prerequisiteIds: courses.length > 0 ? [courses[courses.length - 1].id] : [],
   375→    };
   376→    setCourses([...courses, newCourse]);
   377→    setCourseSearch('');
   378→    setShowCourseDropdown(false);
   379→  };
   380→
   381→  const removeCourse = (courseId: string) => {
   382→    setCourses(courses.filter((c) => c.id !== courseId).map((c) => ({
   383→      ...c,
   384→      prerequisiteIds: c.prerequisiteIds.filter((pid) => pid !== courseId),
   385→    })));
   386→  };
   387→
   388→  const toggleRequired = (courseId: string) => {
   389→    setCourses(courses.map((c) => c.id === courseId ? { ...c, isRequired: !c.isRequired } : c));
   390→  };
   391→
   392→  const setMilestone = (courseId: string, milestone: string) => {
   393→    setCourses(courses.map((c) => c.id === courseId ? { ...c, milestone: milestone || undefined } : c));
   394→  };
   395→
   396→  const togglePrerequisite = (courseId: string, prereqId: string) => {
   397→    setCourses(courses.map((c) => {
   398→      if (c.id !== courseId) return c;
   399→      const hasPrereq = c.prerequisiteIds.includes(prereqId);
   400→      return {
   401→        ...c,
   402→        prerequisiteIds: hasPrereq
   403→          ? c.prerequisiteIds.filter((p) => p !== prereqId)
   404→          : [...c.prerequisiteIds, prereqId],
   405→      };
   406→    }));
   407→  };
   408→
   409→  const totalDuration = courses.reduce((sum, c) => sum + c.duration, 0);
   410→
   411→  const handleSave = (publishStatus: 'published' | 'draft') => {
   412→    const path: LearningPath = {
   413→      id: editingPath?.id || `lp-${Date.now()}`,
   414→      name,
   415→      description,
   416→      category,
   417→      difficulty,
   418→      status: publishStatus,
   419→      courseCount: courses.length,
   420→      enrolledCount: editingPath?.enrolledCount || 0,
   421→      completionRate: editingPath?.completionRate || 0,
   422→      estimatedDuration: totalDuration,
   423→      courses,
   424→      createdAt: editingPath?.createdAt || new Date().toISOString(),
   425→    };
   426→    onSave(path);
   427→  };
   428→
   429→  return (
   430→    <div className="space-y-6">
   431→      {/* Header */}
   432→      <div className="flex items-center justify-between">
   433→        <div>
   434→          <h2 className="text-lg font-semibold">{editingPath ? 'Edit Learning Path' : 'Create Learning Path'}</h2>
   435→          <p className="text-sm text-muted-foreground">Design a guided course sequence for your learners</p>
   436→        </div>
   437→        <div className="flex items-center gap-2">
   438→          <Button variant="outline" size="sm" onClick={onCancel}>
   439→            <X className="h-4 w-4 mr-1" /> Cancel
   440→          </Button>
   441→          <Button variant="outline" size="sm" onClick={() => handleSave('draft')} className={cn(status === 'draft' && 'border-amber-500')}>
   442→            <Save className="h-4 w-4 mr-1" /> Save Draft
   443→          </Button>
   444→          <Button size="sm" onClick={() => handleSave('published')} className="bg-emerald-600 hover:bg-emerald-700">
   445→            <Globe className="h-4 w-4 mr-1" /> Publish
   446→          </Button>
   447→        </div>
   448→      </div>
   449→
   450→      <div className="grid grid-cols-1 lg:grid-cols-3 gap-6">
   451→        {/* Left: Form */}
   452→        <div className="lg:col-span-1 space-y-4">
   453→          <Card className="border-border/50 bg-card/80 backdrop-blur-sm">
   454→            <CardHeader className="pb-3">
   455→              <CardTitle className="text-sm">Path Details</CardTitle>
   456→            </CardHeader>
   457→            <CardContent className="space-y-4">
   458→              <div className="space-y-2">
   459→                <Label className="text-xs">Path Name</Label>
   460→                <Input
   461→                  value={name}
   462→                  onChange={(e) => setName(e.target.value)}
   463→                  placeholder="e.g., Full-Stack Developer"
   464→                  className="h-9"
   465→                />
   466→              </div>
   467→              <div className="space-y-2">
   468→                <Label className="text-xs">Description</Label>
   469→                <Textarea
   470→                  value={description}
   471→                  onChange={(e) => setDescription(e.target.value)}
   472→                  placeholder="Describe this learning path..."
   473→                  rows={3}
   474→                  className="text-sm"
   475→                />
   476→              </div>
   477→              <div className="space-y-2">
   478→                <Label className="text-xs">Category</Label>
   479→                <Select value={category} onValueChange={setCategory}>
   480→                  <SelectTrigger className="h-9">
   481→                    <SelectValue placeholder="Select category" />
   482→                  </SelectTrigger>
   483→                  <SelectContent>
   484→                    <SelectItem value="Web Development">Web Development</SelectItem>
   485→                    <SelectItem value="Data Science">Data Science</SelectItem>
   486→                    <SelectItem value="Backend">Backend</SelectItem>
   487→                    <SelectItem value="DevOps">DevOps</SelectItem>
   488→                    <SelectItem value="Mobile">Mobile</SelectItem>
   489→                    <SelectItem value="AI/ML">AI/ML</SelectItem>
   490→                  </SelectContent>
   491→                </Select>
   492→              </div>
   493→              <div className="space-y-2">
   494→                <Label className="text-xs">Difficulty Level</Label>
   495→                <Select value={difficulty} onValueChange={(v) => setDifficulty(v as 'beginner' | 'intermediate' | 'advanced')}>
   496→                  <SelectTrigger className="h-9">
   497→                    <SelectValue />
   498→                  </SelectTrigger>
   499→                  <SelectContent>
   500→                    <SelectItem value="beginner">Beginner</SelectItem>
   501→                    <SelectItem value="intermediate">Intermediate</SelectItem>
   502→                    <SelectItem value="advanced">Advanced</SelectItem>
   503→                  </SelectContent>
   504→                </Select>
   505→              </div>
   506→              <div className="flex items-center gap-3 pt-2">
   507→                <Label className="text-xs">Status</Label>
   508→                <div className="flex items-center gap-2">
   509→                  <Switch
   510→                    checked={status === 'published'}
   511→                    onCheckedChange={(checked) => setStatus(checked ? 'published' : 'draft')}
   512→                  />
   513→                  <span className="text-xs text-muted-foreground">
   514→                    {status === 'published' ? 'Published' : 'Draft'}
   515→                  </span>
   516→                </div>
   517→              </div>
   518→            </CardContent>
   519→          </Card>
   520→
   521→          <Card className="border-border/50 bg-card/80 backdrop-blur-sm">
   522→            <CardHeader className="pb-3">
   523→              <CardTitle className="text-sm">Summary</CardTitle>
