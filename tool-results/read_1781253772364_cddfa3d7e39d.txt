   273тЖТ   273тЖТ// Countdown Timer Component
   274тЖТ   274тЖТfunction CountdownTimer({ targetDate, color }: { targetDate: string; color: string }) {
   275тЖТ   275тЖТ  const [timeLeft, setTimeLeft] = useState('');
   276тЖТ   276тЖТ  const [isUrgent, setIsUrgent] = useState(false);
   277тЖТ   277тЖТ  const intervalRef = useRef<ReturnType<typeof setInterval> | null>(null);
   278тЖТ   278тЖТ
   279тЖТ   279тЖТ  useEffect(() => {
   280тЖТ   280тЖТ    const updateTimer = () => {
   281тЖТ   281тЖТ      const now = new Date();
   282тЖТ   282тЖТ      const target = parseISO(targetDate);
   283тЖТ   283тЖТ      const diffSeconds = differenceInSeconds(target, now);
   284тЖТ   284тЖТ
   285тЖТ   285тЖТ      if (diffSeconds <= 0) {
   286тЖТ   286тЖТ        setTimeLeft('Starting now');
   287тЖТ   287тЖТ        setIsUrgent(true);
   288тЖТ   288тЖТ        if (intervalRef.current) clearInterval(intervalRef.current);
   289тЖТ   289тЖТ        return;
   290тЖТ   290тЖТ      }
   291тЖТ   291тЖТ
   292тЖТ   292тЖТ      setIsUrgent(diffSeconds < 900); // 15 minutes
   293тЖТ   293тЖТ
   294тЖТ   294тЖТ      const days = Math.floor(diffSeconds / 86400);
   295тЖТ   295тЖТ      const hours = Math.floor((diffSeconds % 86400) / 3600);
   296тЖТ   296тЖТ      const minutes = Math.floor((diffSeconds % 3600) / 60);
   297тЖТ   297тЖТ      const seconds = diffSeconds % 60;
   298тЖТ   298тЖТ
   299тЖТ   299тЖТ      if (days > 0) {
   300тЖТ   300тЖТ        setTimeLeft(`${days}d ${hours}h ${minutes}m`);
   301тЖТ   301тЖТ      } else if (hours > 0) {
   302тЖТ   302тЖТ        setTimeLeft(`${hours}h ${minutes}m ${seconds}s`);
   303тЖТ   303тЖТ      } else if (minutes > 0) {
   304тЖТ   304тЖТ        setTimeLeft(`${minutes}m ${seconds}s`);
   305тЖТ   305тЖТ      } else {
   306тЖТ   306тЖТ        setTimeLeft(`${seconds}s`);
   307тЖТ   307тЖТ      }
   308тЖТ   308тЖТ    };
   309тЖТ   309тЖТ
   310тЖТ   310тЖТ    updateTimer();
   311тЖТ   311тЖТ    intervalRef.current = setInterval(updateTimer, 1000);
   312тЖТ   312тЖТ
   313тЖТ   313тЖТ    return () => {
   314тЖТ   314тЖТ      if (intervalRef.current) clearInterval(intervalRef.current);
   315тЖТ   315тЖТ    };
   316тЖТ   316тЖТ  }, [targetDate]);
   317тЖТ   317тЖТ
   318тЖТ   318тЖТ  return (
   319тЖТ   319тЖТ    <motion.div
   320тЖТ   320тЖТ      className="flex items-center gap-1.5 text-xs font-mono"
   321тЖТ   321тЖТ      animate={isUrgent ? { scale: [1, 1.02, 1] } : {}}
   322тЖТ   322тЖТ      transition={isUrgent ? { duration: 0.8, repeat: Infinity } : {}}
   323тЖТ   323тЖТ    >
   324тЖТ   324тЖТ      <Timer className="h-3.5 w-3.5" style={{ color }} />
   325тЖТ   325тЖТ      <span className={isUrgent ? 'text-red-500 font-bold' : 'text-muted-foreground'}>
   326тЖТ   326тЖТ        {timeLeft}
   327тЖТ   327тЖТ      </span>
   328тЖТ   328тЖТ    </motion.div>
   329тЖТ   329тЖТ  );
   330тЖТ   330тЖТ}
