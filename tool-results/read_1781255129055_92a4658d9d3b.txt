   176→function ProgressRing({ progress, size = 48, strokeWidth = 4 }: { progress: number; size?: number; strokeWidth?: number }) {
   177→  const radius = (size - strokeWidth) / 2;
   178→  const circumference = radius * 2 * Math.PI;
   179→  const offset = circumference - (progress / 100) * circumference;
   180→
   181→  return (
   182→    <div className="relative inline-flex items-center justify-center" style={{ width: size, height: size }}>
   183→      <svg className="transform -rotate-90" width={size} height={size}>
   184→        <circle
   185→          cx={size / 2}
   186→          cy={size / 2}
   187→          r={radius}
   188→          stroke="currentColor"
   189→          strokeWidth={strokeWidth}
   190→          fill="none"
   191→          className="text-muted/30"
   192→        />
   193→        <circle
   194→          cx={size / 2}
   195→          cy={size / 2}
   196→          r={radius}
   197→          stroke="url(#progressGradient)"
   198→          strokeWidth={strokeWidth}
   199→          fill="none"
   200→          strokeDasharray={circumference}
   201→          strokeDashoffset={offset}
   202→          strokeLinecap="round"
   203→          className="transition-all duration-700"
   204→        />
   205→        <defs>
   206→          <linearGradient id="progressGradient" x1="0%" y1="0%" x2="100%" y2="0%">
   207→            <stop offset="0%" stopColor="#10b981" />
   208→            <stop offset="100%" stopColor="#14b8a6" />
   209→          </linearGradient>
   210→        </defs>
   211→      </svg>
   212→      <span className="absolute text-xs font-bold">{progress}%</span>
   213→    </div>
   214→  );
   215→}
