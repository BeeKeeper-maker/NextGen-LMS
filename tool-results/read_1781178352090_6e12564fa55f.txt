    60→    60→// ─── Format duration ───────────────────────────────────────
    61→    61→function formatDuration(seconds?: number): string {
    62→    62→  if (!seconds) return '';
    63→    63→  const mins = Math.floor(seconds / 60);
    64→    64→  if (mins < 60) return `${mins} min`;
    65→    65→  const hours = Math.floor(mins / 60);
    66→    66→  const remainMins = mins % 60;
    67→    67→  return `${hours}h ${remainMins}m`;
    68→    68→}

   327→   327→export function LearnerCourse() {
   328→   328→  const { currentUser } = useAppStore();

   330→   330→  const enrollment = demoEnrollments[0]; // Active enrollment for this course
   331→   331→  const modules = course.modules || [];
   332→   332→  const [activeTab, setActiveTab] = useState('curriculum');
   333→   333→  const [activeLesson, setActiveLesson] = useState<Lesson | null>(null);
   334→   334→  const [activeLessonModuleName, setActiveLessonModuleName] = useState('');
   335→   335→  const [lessonProgressState, setLessonProgressState] = useState(lessonProgressMap);
   336→   336→
   337→   337→  // Calculate total lessons and completed
   338→   338→  const allLessons = modules.flatMap(m => m.lessons || []);
   339→   339→  const completedCount = allLessons.filter(l => lessonProgressState[l.id] === 'completed').length;
   340→   340→  const totalLessons = allLessons.length;
   341→   341→
   342→   342→  // Handle clicking a lesson
   343→   343→  const handleLessonClick = (lesson: Lesson) => {
   344→   344→    setActiveLesson(lesson);
   345→   345→    // Find the module name for this lesson
   346→   346→    const parentModule = modules.find(m => m.lessons?.some(l => l.id === lesson.id));
   347→   347→    setActiveLessonModuleName(parentModule?.title || '');
   348→   348→  };
   349→   349→
   350→   350→  // Handle marking a lesson as complete
   351→   351→  const handleMarkComplete = (lessonId: string) => {
   352→   352→    setLessonProgressState(prev => ({ ...prev, [lessonId]: 'completed' }));
   353→   353→  };
   354→   354→
   355→   355→  // Get next lessons for the sidebar
   356→   356→  const getNextLessons = (): { id: string; title: string; moduleName: string; duration: number; contentType: Lesson['contentType'] }[] => {
   357→   357→    if (!activeLesson) return [];
   358→   358→    const currentIdx = allLessons.findIndex(l => l.id === activeLesson.id);
   359→   359→    const next: { id: string; title: string; moduleName: string; duration: number; contentType: Lesson['contentType'] }[] = [];
   360→   360→    for (let i = currentIdx + 1; i < Math.min(currentIdx + 4, allLessons.length); i++) {
   361→   361→      const l = allLessons[i];
   362→   362→      const mod = modules.find(m => m.lessons?.some(ml => ml.id === l.id));
   363→   363→      next.push({ id: l.id, title: l.title, moduleName: mod?.title || '', duration: l.videoDuration || 0, contentType: l.contentType });
   364→   364→    }
   365→   365→    return next;
   366→   366→  };
   367→   367→
   368→   368→  // If a lesson is active, show the enhanced Video Player
   369→   369→  if (activeLesson) {
   370→   370→    return (
   371→   371→      <div className="p-4 md:p-6">
   372→   372→        <VideoPlayer
   373→   373→          title={activeLesson.title}
   374→   374→          lesson={activeLesson}
   375→   375→          moduleName={activeLessonModuleName}
   376→   376→          nextLessons={getNextLessons()}
   377→   377→          onMarkComplete={handleMarkComplete}
   378→   378→          onNextLesson={(lessonId) => {
   379→   379→            const found = allLessons.find(l => l.id === lessonId);
   380→   380→            if (found) handleLessonClick(found);
   381→   381→          }}
   382→   382→          onBack={() => setActiveLesson(null)}
   383→   383→          isCompleted={lessonProgressState[activeLesson.id] === 'completed'}
   384→   384→          initialPosition={resumePositions[activeLesson.id] || 0}
   385→   385→          chapters={demoChapters}
   386→   386→          onProgress={(progress, currentTime) => {
   387→   387→            // Store resume position
   388→   388→            resumePositions[activeLesson.id] = currentTime;
   389→   389→          }}
   390→   390→        />
   391→   391→      </div>
   392→   392→    );
   393→   393→  }
   394→   394→
   395→   395→  return (
   396→   396→    <motion.div
   397→   397→      className="p-4 md:p-6 space-y-6"
   398→   398→      variants={containerVariants}
   399→   399→      initial="hidden"
   400→   400→      animate="visible"
   401→   401→    >
   402→   402→      {/* ─── Course Header ────────────────────────────────── */}
   403→   403→      <motion.div variants={itemVariants}>
   404→   404→        <Card className="overflow-hidden">
   405→   405→          {/* Gradient Banner */}
   406→   406→          <div className={cn(
   407→   407→            'h-32 md:h-40 bg-gradient-to-r flex items-end p-6',
   408→   408→            getLevelAccent(course.level)
   409→   409→          )}>
