   180тЖТ   180тЖТfunction AnimatedCounter({ value, duration = 1200 }: { value: number; duration?: number }) {
   181тЖТ   181тЖТ  const [display, setDisplay] = useState(0);
   182тЖТ   182тЖТ  const startRef = useRef<number | null>(null);
   183тЖТ   183тЖТ
   184тЖТ   184тЖТ  useEffect(() => {
   185тЖТ   185тЖТ    startRef.current = null;
   186тЖТ   186тЖТ    const step = (timestamp: number) => {
   187тЖТ   187тЖТ      if (!startRef.current) startRef.current = timestamp;
   188тЖТ   188тЖТ      const progress = Math.min((timestamp - startRef.current) / duration, 1);
   189тЖТ   189тЖТ      setDisplay(Math.floor(progress * value));
   190тЖТ   190тЖТ      if (progress < 1) requestAnimationFrame(step);
   191тЖТ   191тЖТ    };
   192тЖТ   192тЖТ    requestAnimationFrame(step);
   193тЖТ   193тЖТ  }, [value, duration]);
   194тЖТ   194тЖТ
   195тЖТ   195тЖТ  return <>{display}</>;
   196тЖТ   196тЖТ}
