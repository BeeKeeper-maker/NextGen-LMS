   154→// Relative time helper
   155→function getRelativeTime(dateStr?: string) {
   156→  if (!dateStr) return '';
   157→  const now = new Date();
   158→  const date = new Date(dateStr);
   159→  const diffMs = now.getTime() - date.getTime();
   160→  const diffMins = Math.floor(diffMs / 60000);
   161→  const diffHours = Math.floor(diffMins / 60);
   162→  const diffDays = Math.floor(diffHours / 24);
   163→
   164→  if (diffMins < 1) return 'Just now';
   165→  if (diffMins < 60) return `${diffMins}m ago`;
   166→  if (diffHours < 24) return `${diffHours}h ago`;
   167→  if (diffDays < 7) return `${diffDays}d ago`;
   168→  return date.toLocaleDateString();
   169→}

   222→// ─── Animated Counter ─────────────────────────────
   223→function AnimatedCounter({ value, duration = 1.5 }: { value: string; duration?: number }) {
   224→  const displayRef = useRef<HTMLSpanElement>(null);
   225→  const hasAnimated = useRef(false);
   226→
   227→  useEffect(() => {
   228→    if (hasAnimated.current) return;
   229→    hasAnimated.current = true;
   230→
   231→    const el = displayRef.current;
   232→    if (!el) return;
   233→
   234→    const numericMatch = value.match(/[\d,]+/);
   235→    if (!numericMatch) {
   236→      el.textContent = value;
   237→      return;
   238→    }
   239→    const numericStr = numericMatch[0].replace(/,/g, '');
   240→    const target = parseInt(numericStr, 10);
   241→    if (isNaN(target)) {
   242→      el.textContent = value;
   243→      return;
   244→    }
   245→
   246→    const prefix = value.substring(0, value.indexOf(numericMatch[0]));
   247→    const suffix = value.substring(value.indexOf(numericMatch[0]) + numericMatch[0].length);
   248→    const startTime = performance.now();
   249→
   250→    const animate = (currentTime: number) => {
   251→      const elapsed = currentTime - startTime;
   252→      const progress = Math.min(elapsed / (duration * 1000), 1);
   253→      const eased = 1 - Math.pow(1 - progress, 3);
   254→      const current = Math.round(eased * target);
   255→      const formatted = current.toLocaleString();
   256→      if (displayRef.current) {
   257→        displayRef.current.textContent = `${prefix}${formatted}${suffix}`;
   258→      }
   259→      if (progress < 1) {
   260→        requestAnimationFrame(animate);
   261→      }
   262→    };
   263→    requestAnimationFrame(animate);
   264→  }, [value, duration]);
   265→
   266→  return <span ref={displayRef}>0</span>;
   267→}

   329→// ─── Floating Geometric Shape ─────────────────────
   330→function FloatingShape({ type, className, delay, duration }: { type: 'circle' | 'square' | 'triangle' | 'diamond'; className?: string; delay?: number; duration?: number }) {
   331→  const shapeClass = cn(
   332→    'absolute opacity-[0.07] dark:opacity-[0.05]',
   333→    type === 'circle' && 'rounded-full',
   334→    type === 'square' && 'rounded-lg rotate-12',
   335→    type === 'diamond' && 'rounded-md rotate-45',
   336→    className,
   337→  );
   338→
   339→  return (
   340→    <motion.div
   341→      className={shapeClass}
   342→      initial={{ y: 0, rotate: type === 'square' ? 12 : type === 'diamond' ? 45 : 0 }}
   343→      animate={{
   344→        y: [-10, 10, -10],
   345→        rotate: type === 'square' ? [12, 24, 12] : type === 'diamond' ? [45, 55, 45] : [0, 5, 0],
   346→      }}
   347→      transition={{
   348→        duration: duration || 6,
   349→        repeat: Infinity,
   350→        ease: 'easeInOut',
   351→        delay: delay || 0,
   352→      }}
   353→    />

   484→// ─── Parallax Tilt Card ──────────────────────────
   485→function TiltCard({ children, className }: { children: React.ReactNode; className?: string }) {
   486→  const cardRef = useRef<HTMLDivElement>(null);
   487→  const [rotateX, setRotateX] = useState(0);
   488→  const [rotateY, setRotateY] = useState(0);
   489→
   490→  const handleMouseMove = (e: React.MouseEvent<HTMLDivElement>) => {
   491→    if (!cardRef.current) return;
   492→    const rect = cardRef.current.getBoundingClientRect();
   493→    const x = e.clientX - rect.left;
   494→    const y = e.clientY - rect.top;
   495→    const centerX = rect.width / 2;
   496→    const centerY = rect.height / 2;
   497→    const maxTilt = 5;
   498→    const newRotateX = ((y - centerY) / centerY) * -maxTilt;
   499→    const newRotateY = ((x - centerX) / centerX) * maxTilt;
   500→    setRotateX(newRotateX);
   501→    setRotateY(newRotateY);
   502→  };
   503→
   504→  const handleMouseLeave = () => {
   505→    setRotateX(0);
   506→    setRotateY(0);
   507→  };
   508→
   509→  return (
   510→    <div
   511→      ref={cardRef}
   512→      onMouseMove={handleMouseMove}
   513→      onMouseLeave={handleMouseLeave}
   514→      className={className}
   515→      style={{
   516→        transform: `perspective(800px) rotateX(${rotateX}deg) rotateY(${rotateY}deg)`,
   517→        transition: 'transform 0.15s ease-out',
   518→      }}
   519→    >
   520→      {children}
   521→    </div>
   522→  );
   523→}

   602→function KPICircularProgress({ value, color, size = 48, strokeWidth = 4 }: { value: number; color: string; size?: number; strokeWidth?: number }) {
   603→  const [animatedValue, setAnimatedValue] = useState(0);
   604→
   605→  useEffect(() => {
   606→    const timer = setTimeout(() => setAnimatedValue(value), 150);
   607→    return () => clearTimeout(timer);
   608→  }, [value]);
   609→
   610→  const radius = (size - strokeWidth) / 2;
   611→  const circumference = 2 * Math.PI * radius;
   612→  const strokeDashoffset = circumference - (animatedValue / 100) * circumference;
   613→
   614→  return (
   615→    <svg width={size} height={size} className="-rotate-90">
   616→      <circle
   617→        cx={size / 2}
   618→        cy={size / 2}
   619→        r={radius}
   620→        fill="none"
   621→        stroke="currentColor"
   622→        strokeWidth={strokeWidth}
   623→        className="text-slate-200/50 dark:text-slate-700/50"
   624→      />
   625→      <motion.circle
   626→        cx={size / 2}
   627→        cy={size / 2}
   628→        r={radius}
   629→        fill="none"
   630→        stroke={color}
   631→        strokeWidth={strokeWidth}
   632→        strokeLinecap="round"
   633→        strokeDasharray={circumference}
   634→        initial={{ strokeDashoffset: circumference }}
   635→        animate={{ strokeDashoffset }}
   636→        transition={{ duration: 1.2, ease: 'easeOut', delay: 0.2 }}
   637→      />
   638→    </svg>
   639→  );
