    59→// ─── Format duration ───────────────────────────────────────
    60→function formatDuration(seconds?: number): string {
    61→  if (!seconds) return '';
    62→  const mins = Math.floor(seconds / 60);
    63→  if (mins < 60) return `${mins} min`;
    64→  const hours = Math.floor(mins / 60);
    65→  const remainMins = mins % 60;
    66→  return `${hours}h ${remainMins}m`;
    67→}

   310→export function LearnerCourse() {
   311→  const { currentUser } = useAppStore();
   312→  const course = demoCourses[0]; // React & Next.js Masterclass
   313→  const enrollment = demoEnrollments[0]; // Active enrollment for this course
   314→  const modules = course.modules || [];
   315→  const [activeTab, setActiveTab] = useState('curriculum');
   316→  const [activeLesson, setActiveLesson] = useState<Lesson | null>(null);
   317→  const [activeLessonModuleName, setActiveLessonModuleName] = useState('');
   318→  const [lessonProgressState, setLessonProgressState] = useState(lessonProgressMap);
   319→
   320→  // Calculate total lessons and completed
   321→  const allLessons = modules.flatMap(m => m.lessons || []);
   322→  const completedCount = allLessons.filter(l => lessonProgressState[l.id] === 'completed').length;
   323→  const totalLessons = allLessons.length;
   324→
   325→  // Handle clicking a lesson
   326→  const handleLessonClick = (lesson: Lesson) => {
   327→    setActiveLesson(lesson);
   328→    // Find the module name for this lesson
   329→    const parentModule = modules.find(m => m.lessons?.some(l => l.id === lesson.id));
   330→    setActiveLessonModuleName(parentModule?.title || '');
   331→  };
   332→
   333→  // Handle marking a lesson as complete
   334→  const handleMarkComplete = (lessonId: string) => {
   335→    setLessonProgressState(prev => ({ ...prev, [lessonId]: 'completed' }));
   336→  };
   337→
   338→  // Get next lessons for the sidebar
   339→  const getNextLessons = (): { id: string; title: string; moduleName: string; duration: number; contentType: Lesson['contentType'] }[] => {
   340→    if (!activeLesson) return [];
   341→    const currentIdx = allLessons.findIndex(l => l.id === activeLesson.id);
   342→    const next: { id: string; title: string; moduleName: string; duration: number; contentType: Lesson['contentType'] }[] = [];
   343→    for (let i = currentIdx + 1; i < Math.min(currentIdx + 4, allLessons.length); i++) {
   344→      const l = allLessons[i];
   345→      const mod = modules.find(m => m.lessons?.some(ml => ml.id === l.id));
   346→      next.push({ id: l.id, title: l.title, moduleName: mod?.title || '', duration: l.videoDuration || 0, contentType: l.contentType });
   347→    }
   348→    return next;
   349→  };
   350→
   351→  // If a lesson is active, show the Video Player
   352→  if (activeLesson) {
   353→    return (
   354→      <div className="p-4 md:p-6">
   355→        <VideoPlayer
   356→          lesson={activeLesson}
   357→          moduleName={activeLessonModuleName}
   358→          nextLessons={getNextLessons()}
   359→          onMarkComplete={handleMarkComplete}
   360→          onNextLesson={(lessonId) => {
   361→            const found = allLessons.find(l => l.id === lessonId);
   362→            if (found) handleLessonClick(found);
   363→          }}
   364→          onBack={() => setActiveLesson(null)}
   365→          isCompleted={lessonProgressState[activeLesson.id] === 'completed'}
   366→        />
   367→      </div>
   368→    );
   369→  }
   370→
   371→  return (
   372→    <motion.div
   373→      className="p-4 md:p-6 space-y-6"
   374→      variants={containerVariants}
   375→      initial="hidden"
   376→      animate="visible"
   377→    >
   378→      {/* ─── Course Header ────────────────────────────────── */}
   379→      <motion.div variants={itemVariants}>
   380→        <Card className="overflow-hidden">
   381→          {/* Gradient Banner */}
   382→          <div className={cn(
   383→            'h-32 md:h-40 bg-gradient-to-r flex items-end p-6',
   384→            getLevelAccent(course.level)
   385→          )}>
   386→            <div className="flex-1">
   387→              <div className="flex items-center gap-2 mb-2">
   388→                <Badge className="bg-white/20 text-white border-0 backdrop-blur-sm text-xs">
   389→                  {course.category}
   390→                </Badge>
   391→                <Badge className="bg-white/20 text-white border-0 backdrop-blur-sm text-xs capitalize">
   392→                  {course.level}
   393→                </Badge>
   394→              </div>
   395→              <h1 className="text-2xl md:text-3xl font-bold text-white leading-tight">{course.title}</h1>
   396→            </div>
   397→          </div>
   398→
   399→          <CardContent className="p-6">
   400→            {/* Course Meta */}
   401→            <div className="flex flex-wrap items-center gap-4 text-sm text-muted-foreground mb-4">
   402→              <div className="flex items-center gap-1.5">
   403→                <Avatar className="h-6 w-6">
   404→                  <AvatarFallback className="text-[10px] bg-slate-200 dark:bg-slate-700">SM</AvatarFallback>
