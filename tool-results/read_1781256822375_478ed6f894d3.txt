   123→// ─── Format duration ───────────────────────────────────────
   124→function formatDuration(seconds?: number): string {
   125→  if (!seconds) return '';
   126→  const mins = Math.floor(seconds / 60);
   127→  if (mins < 60) return `${mins} min`;
   128→  const hours = Math.floor(mins / 60);
   129→  const remainMins = mins % 60;
   130→  return `${hours}h ${remainMins}m`;
   131→}

   166→function useLessonNotes(courseId: string | undefined) {
   167→  const storageKey = `lms-notes-${courseId || 'default'}`;
   168→
   169→  const [notes, setNotes] = useState<Note[]>(() => {
   170→    if (typeof window === 'undefined') return [];
   171→    try {
   172→      const stored = localStorage.getItem(storageKey);
   173→      return stored ? JSON.parse(stored) : [];
   174→    } catch {
   175→      return [];
   176→    }
   177→  });
   178→
   179→  const persistNotes = useCallback((updated: Note[]) => {
   180→    setNotes(updated);
   181→    if (typeof window !== 'undefined') {
   182→      try {
   183→        localStorage.setItem(storageKey, JSON.stringify(updated));
   184→      } catch {
   185→        // ignore quota errors
   186→      }
   187→    }
   188→  }, [storageKey]);
   189→
   190→  const addNote = useCallback((note: Omit<Note, 'id' | 'createdAt' | 'updatedAt'>) => {
   191→    const newNote: Note = {
   192→      ...note,
   193→      id: `note-${Date.now()}-${Math.random().toString(36).slice(2, 7)}`,
   194→      createdAt: new Date().toISOString(),
   195→      updatedAt: new Date().toISOString(),
   196→    };
   197→    persistNotes([...notes, newNote]);
   198→    return newNote;
   199→  }, [notes, persistNotes]);
   200→
   201→  const updateNote = useCallback((id: string, updates: Partial<Note>) => {
   202→    const updated = notes.map(n =>
   203→      n.id === id ? { ...n, ...updates, updatedAt: new Date().toISOString() } : n
   204→    );
   205→    persistNotes(updated);
   206→  }, [notes, persistNotes]);
   207→
   208→  const deleteNote = useCallback((id: string) => {
   209→    persistNotes(notes.filter(n => n.id !== id));
   210→  }, [notes, persistNotes]);
   211→
   212→  return { notes, addNote, updateNote, deleteNote };
   213→}

   243→// ─── Time ago helper ──────────────────────────────────────
   244→function timeAgo(dateStr: string): string {
   245→  const seconds = Math.floor((Date.now() - new Date(dateStr).getTime()) / 1000);
   246→  if (seconds < 60) return 'just now';
   247→  const minutes = Math.floor(seconds / 60);
   248→  if (minutes < 60) return `${minutes}m ago`;
   249→  const hours = Math.floor(minutes / 60);
   250→  if (hours < 24) return `${hours}h ago`;
   251→  const days = Math.floor(hours / 24);
   252→  if (days < 30) return `${days}d ago`;
   253→  const months = Math.floor(days / 30);
   254→  return `${months}mo ago`;
   255→}

   468→// ─── Discussion Q&A Tab ────────────────────────────────────
   469→function DiscussionTab({ courseId, userId }: { courseId: string; userId: string }) {
   470→  const { data: discussionsData, isLoading } = useLessonDiscussions({ courseId });
   471→  const createDiscussion = useCreateDiscussion();
   472→
   473→  const [sortBy, setSortBy] = useState<'recent' | 'upvoted' | 'unanswered'>('recent');
   474→  const [searchQuery, setSearchQuery] = useState('');
   475→  const [expandedQuestion, setExpandedQuestion] = useState<string | null>(null);
   476→  const [showAskForm, setShowAskForm] = useState(false);
   477→  const [askTitle, setAskTitle] = useState('');
   478→  const [askContent, setAskContent] = useState('');
   479→  const [replyContent, setReplyContent] = useState('');
   480→  const [replyingTo, setReplyingTo] = useState<string | null>(null);
   481→
   482→  const discussions = discussionsData?.discussions || [];
   483→
   484→  // Separate top-level questions from replies & build reply map
   485→  const { topLevelDiscussions, repliesMap } = useMemo(() => {
   486→    const top: any[] = [];
   487→    const map: Record<string, any[]> = {};
   488→    discussions.forEach((d: any) => {
   489→      if (d.parentId) {
   490→        if (!map[d.parentId]) map[d.parentId] = [];
   491→        map[d.parentId].push(d);
   492→      } else {
   493→        top.push(d);
   494→      }
   495→    });
   496→    return { topLevelDiscussions: top, repliesMap: map };
   497→  }, [discussions]);
   498→
   499→  const unansweredCount = topLevelDiscussions.filter((q: any) => !(repliesMap[q.id]?.length > 0)).length;
   500→
   501→  const sortedQuestions = [...topLevelDiscussions].sort((a: any, b: any) => {
   502→    if (sortBy === 'recent') return new Date(b.createdAt).getTime() - new Date(a.createdAt).getTime();
   503→    if (sortBy === 'upvoted') return 0;
   504→    if (sortBy === 'unanswered') {
   505→      const aReplies = repliesMap[a.id]?.length || 0;
   506→      const bReplies = repliesMap[b.id]?.length || 0;
   507→      return aReplies === bReplies ? 0 : aReplies > bReplies ? 1 : -1;
   508→    }

   639→                  <div className="flex items-center gap-2 text-xs text-muted-foreground">

   646→                      {createDiscussion.isPending ? <Loader2 className="h-3 w-3 animate-spin" /> : <Send className="h-3 w-3" />} Post Question
   647→                    </Button>
   648→                  </div>
   649→                </div>

   655→

   657→      <div className="space-y-3">
   658→        <AnimatePresence mode="popLayout">
   659→          {filteredQuestions.map((question: any) => {

   676→                    <div className="flex items-start gap-3">

   679→                        <button className="p-1 rounded hover:bg-muted/50 transition-colors text-muted-foreground hover:text-foreground">
   680→                          <ChevronUp className="h-4 w-4" />
   681→                        </button>
   682→                        <span className="text-sm font-semibold text-foreground">{qReplies.length}</span>

   685→                        </button>
   686→                      </div>
   687→

   690→                        <div className="flex items-center gap-2 mb-1 flex-wrap">
   691→                          {isAnswered && (
   692→                            <Badge className="bg-emerald-100 text-emerald-700 dark:bg-emerald-900/30 dark:text-emerald-400 text-[10px] gap-0.5">
   693→                              <CheckCircle2 className="h-3 w-3" /> Answered
   694→                            </Badge>
   695→                          )}
   696→                          {question.lesson?.title && (
   697→                            <Badge variant="outline" className="text-[10px] gap-0.5">
   698→                              <BookOpen className="h-2.5 w-2.5" /> {question.lesson.title}
   699→                            </Badge>
   700→                          )}
   701→                          {question.isPinned && (
   702→                            <Badge className="bg-amber-100 text-amber-700 dark:bg-amber-900/30 dark:text-amber-400 text-[10px] gap-0.5">
   703→                              <Pin className="h-2.5 w-2.5" /> Pinned
   704→                            </Badge>
   705→                          )}
   706→                        </div>
   707→                        <p className="text-sm text-muted-foreground line-clamp-3">{question.content}</p>
   708→                        <div className="flex items-center gap-3 mt-2">
   709→                          <span className="text-xs text-muted-foreground">· {timeAgo(question.createdAt)}</span>
   710→                          <span className="flex items-center gap-1 text-xs text-muted-foreground">
   711→                            <MessageCircle className="h-3 w-3" /> {qReplies.length}
   712→                          </span>
   713→                        </div>
   714→                      </div>

   717→                    {/* Expanded Replies */}
   718→                    <AnimatePresence>
   719→                      {expandedQuestion === question.id && (
   720→                        <motion.div
   721→                          initial={{ height: 0, opacity: 0 }}
   722→                          animate={{ height: 'auto', opacity: 1 }}
   723→                          exit={{ height: 0, opacity: 0 }}
   724→                          transition={{ duration: 0.2 }}
   725→                          className="overflow-hidden"
   726→                        >
   727→                          <div className="mt-4 pt-4 border-t border-border/50 space-y-3">
   728→                            {qReplies.map((reply: any) => (
   729→                              <div key={reply.id} className={cn('ml-8 p-3 rounded-lg bg-muted/30')}>
   730→                                <p className="text-xs text-muted-foreground leading-relaxed">{reply.content}</p>
   731→                                <div className="flex items-center gap-3 mt-2">
   732→                                  <span className="text-[10px] text-muted-foreground">{timeAgo(reply.createdAt)}</span>
   733→                                  <button className="flex items-center gap-1 text-[10px] text-muted-foreground hover:text-foreground transition-colors">
   734→                                    <ThumbsUp className="h-3 w-3" />
   735→                                  </button>
   736→                                </div>
   737→                              </div>
   738→                            ))}
   739→
   740→                            {/* Reply input */}
   741→                            <div className="ml-8 flex gap-2">
   742→                              <Input
   743→                                placeholder="Write a reply..."
   744→                                value={replyingTo === question.id ? replyContent : ''}

   750→                                size="sm"

   756→                              </Button>
   757→                            </div>
   758→                          </div>
   759→                        </motion.div>
   760→                      )}
   761→                    </AnimatePresence>
   762→                  </CardContent>
   763→                </Card>
   764→              </motion.div>
   765→            );
   766→          })}
   767→        </AnimatePresence>
   768→      </div>
   769→
   770→      {filteredQuestions.length === 0 && (
   771→        <div className="text-center py-8">
   772→          <MessageCircleQuestion className="h-12 w-12 text-muted-foreground/30 mx-auto mb-3" />

   775→      )}
   776→
   777→      <Button variant="outline" className="w-full">View All Discussions</Button>

   782→// ─── Q&A Discussion Tab ────────────────────────────────────
   783→function QADiscussionTab({ courseId, userId, modules }: { courseId: string; userId: string; modules: Module[] }) {
   784→  const { data: discussionsData, isLoading } = useLessonDiscussions({ courseId });
   785→  const createDiscussion = useCreateDiscussion();
   786→
   787→  const [selectedLessonId, setSelectedLessonId] = useState<string>('all');
   788→  const [searchQuery, setSearchQuery] = useState('');
   789→  const [sortBy, setSortBy] = useState<'newest' | 'voted' | 'unresolved' | 'resolved'>('newest');
   790→  const [filterBy, setFilterBy] = useState<'all' | 'my' | 'unresolved' | 'resolved'>('all');
   791→  const [selectedQuestion, setSelectedQuestion] = useState<any | null>(null);
   792→  const [showAskDialog, setShowAskDialog] = useState(false);
   793→  const [askContent, setAskContent] = useState('');
   794→  const [askLessonId, setAskLessonId] = useState<string>(selectedLessonId === 'all' ? '' : selectedLessonId);
   795→  const [answerContent, setAnswerContent] = useState('');
   796→  const [expandedModules, setExpandedModules] = useState<Set<string>>(new Set(modules.map(m => m.id)));
   797→
   798→  const discussions = discussionsData?.discussions || [];
   799→
   800→  // Separate top-level from replies & build reply map
   801→  const { qaTopLevel, repliesMap } = useMemo(() => {
   802→    const top: any[] = [];
   803→    const map: Record<string, any[]> = {};
   804→    discussions.forEach((d: any) => {
   805→      if (d.parentId) {
   806→        if (!map[d.parentId]) map[d.parentId] = [];
   807→        map[d.parentId].push(d);
   808→      } else {
   809→        top.push(d);
   810→      }
   811→    });
   812→    return { qaTopLevel: top, repliesMap: map };
   813→  }, [discussions]);
   814→
   815→  // Build lesson map from modules
   816→  const lessonMap = new Map<string, { title: string; moduleId: string; moduleTitle: string }>();
   817→  modules.forEach(mod => {
   818→    (mod.lessons || []).forEach(lesson => {
   819→      lessonMap.set(lesson.id, { title: lesson.title, moduleId: mod.id, moduleTitle: mod.title });
   820→    });
   821→  });
   822→
   823→  // Filter questions
   824→  const filteredQuestions = qaTopLevel.filter((q: any) => {
   825→    if (selectedLessonId !== 'all' && q.lessonId !== selectedLessonId) return false;
   826→    if (filterBy === 'my' && q.userId !== userId) return false;
   827→    const qReplies = repliesMap[q.id] || [];
   828→    const isResolved = qReplies.length > 0;
   829→    if (filterBy === 'unresolved' && isResolved) return false;
   830→    if (filterBy === 'resolved' && !isResolved) return false;
   831→    if (searchQuery) {
   832→      const lq = searchQuery.toLowerCase();
   833→      if (!q.content.toLowerCase().includes(lq)) return false;
   834→    }
   835→    return true;
   836→  });
   837→
   838→  // Sort questions
   839→  const sortedQuestions = [...filteredQuestions].sort((a: any, b: any) => {
   840→    if (sortBy === 'newest') return new Date(b.createdAt).getTime() - new Date(a.createdAt).getTime();
   841→    if (sortBy === 'voted') return (repliesMap[b.id]?.length || 0) - (repliesMap[a.id]?.length || 0);
   842→    if (sortBy === 'unresolved') {
   843→      const aResolved = (repliesMap[a.id]?.length || 0) > 0;
   844→      const bResolved = (repliesMap[b.id]?.length || 0) > 0;
   845→      return aResolved === bResolved ? 0 : aResolved ? 1 : -1;
   846→    }
   847→    if (sortBy === 'resolved') {
   848→      const aResolved = (repliesMap[a.id]?.length || 0) > 0;
   849→      const bResolved = (repliesMap[b.id]?.length || 0) > 0;
   850→      return aResolved === bResolved ? 0 : aResolved ? -1 : 1;
   851→    }
   852→    return 0;
   853→  });
   854→
   855→  const toggleModule = (moduleId: string) => {
   856→    setExpandedModules(prev => {
   857→      const next = new Set(prev);
   858→      if (next.has(moduleId)) next.delete(moduleId);
   859→      else next.add(moduleId);
   860→      return next;
   861→    });
   862→  };
   863→
   864→  // Count questions per lesson
   865→  const questionCountsByLesson = new Map<string, number>();
   866→  const unresolvedCountsByLesson = new Map<string, number>();
   867→  qaTopLevel.forEach((q: any) => {
   868→    questionCountsByLesson.set(q.lessonId, (questionCountsByLesson.get(q.lessonId) || 0) + 1);
   869→    if (!(repliesMap[q.id]?.length > 0)) unresolvedCountsByLesson.set(q.lessonId, (unresolvedCountsByLesson.get(q.lessonId) || 0) + 1);
   870→  });
   871→
   872→  const unresolvedTotal = qaTopLevel.filter((q: any) => !(repliesMap[q.id]?.length > 0)).length;
   873→
   874→  // Render content with markdown
   875→  const renderContent = (content: string) => {
   876→    const parts = content.split(/(```[\s\S]*?```)/g);
   877→    return parts.map((part, i) => {
   878→      if (part.startsWith('```') && part.endsWith('```')) {
   879→        const codeContent = part.slice(3, -3).replace(/^\w+\n/, '');
   880→        return (
   881→          <pre key={i} className="bg-slate-900 dark:bg-slate-950 text-green-400 p-3 rounded-lg text-xs font-mono overflow-x-auto my-2 leading-relaxed">
   882→            <code>{codeContent}</code>
   883→          </pre>
   884→        );
   885→      }
   886→      const inlineParts = part.split(/(`[^`]+`)/g);
   887→      return (
   888→        <span key={i}>
   889→          {inlineParts.map((ip, j) => {
   890→            if (ip.startsWith('`') && ip.endsWith('`')) {
   891→              return <code key={j} className="bg-slate-100 dark:bg-slate-800 text-emerald-700 dark:text-emerald-400 px-1.5 py-0.5 rounded text-xs font-mono">{ip.slice(1, -1)}</code>;
   892→            }
   893→            const boldParts = ip.split(/(\*\*[^*]+\*\*)/g);
   894→            return boldParts.map((bp, k) => {
   895→              if (bp.startsWith('**') && bp.endsWith('**')) {
   896→                return <strong key={`${j}-${k}`} className="font-semibold text-foreground">{bp.slice(2, -2)}</strong>;
   897→              }
   898→              return <span key={`${j}-${k}`}>{bp}</span>;
   899→            });
   900→          })}
   901→        </span>
   902→      );
   903→    });
   904→  };
   905→
   906→  // Handle submit question
   907→  const handleSubmitQuestion = () => {
   908→    if (!askContent.trim() || !askLessonId) return;
   909→    createDiscussion.mutate({
   910→      lessonId: askLessonId,
   911→      userId,
   912→      content: askContent,
   913→    }, {
   914→      onSuccess: () => {
   915→        setAskContent('');
   916→        setShowAskDialog(false);
   917→      },
   918→    });
   919→  };
   920→
   921→  // Handle submit answer
   922→  const handleSubmitAnswer = () => {
   923→    if (!answerContent.trim() || !selectedQuestion) return;
   924→    createDiscussion.mutate({
   925→      lessonId: selectedQuestion.lessonId,
   926→      userId,
   927→      content: answerContent,
   928→      parentId: selectedQuestion.id,
   929→    }, {
   930→      onSuccess: () => {
   931→        setAnswerContent('');
   932→      },
   933→    });
   934→  };
   935→
   936→  if (isLoading) {
   937→    return (
   938→      <div className="space-y-3">
   939→        {Array.from({ length: 3 }).map((_, i) => (
   940→          <div key={i} className="rounded-xl p-4 bg-muted/30 animate-pulse space-y-3">
   941→            <div className="h-4 bg-muted rounded w-1/2" />
   942→            <div className="h-3 bg-muted rounded w-3/4" />
   943→          </div>
   944→        ))}
   945→      </div>
   946→    );
   947→  }
   948→
   949→  // ─── Thread Detail View ─────────────────────────────────
   950→  if (selectedQuestion) {
   951→    const q = selectedQuestion;
   952→    const qReplies = repliesMap[q.id] || [];
   953→    const isResolved = qReplies.length > 0;
   954→
   955→    return (
   956→      <div className="space-y-4">
   957→        <Button variant="ghost" size="sm" className="gap-1.5 text-muted-foreground hover:text-foreground" onClick={() => setSelectedQuestion(null)}>
   958→          <ChevronDown className="h-4 w-4 rotate-90" />
   959→          Back to Questions
   960→        </Button>
   961→
   962→        <motion.div initial={{ opacity: 0, y: 10 }} animate={{ opacity: 1, y: 0 }}>
   963→          <Card className={cn(glassCard, 'border-orange-200/50 dark:border-orange-800/30')}>
   964→            <CardContent className="p-6">
   965→              <div className="flex-1 min-w-0">
   966→                <div className="flex items-center gap-2 mb-2 flex-wrap">
   967→                  {isResolved ? (
   968→                    <Badge className="bg-emerald-100 text-emerald-700 dark:bg-emerald-900/30 dark:text-emerald-400 text-[10px] gap-0.5">
   969→                      <CheckCircle2 className="h-3 w-3" /> Resolved
   970→                    </Badge>
   971→                  ) : (
   972→                    <Badge className="bg-orange-100 text-orange-700 dark:bg-orange-900/30 dark:text-orange-400 text-[10px] gap-0.5">
   973→                      <Circle className="h-3 w-3" /> Unresolved
   974→                    </Badge>
   975→                  )}
   976→                  {q.lesson?.title && (
   977→                    <Badge variant="outline" className="text-[10px] gap-0.5">
   978→                      <BookOpen className="h-2.5 w-2.5" /> {q.lesson.title}
   979→                    </Badge>
   980→                  )}
   981→                </div>
   982→                <div className="text-sm text-muted-foreground leading-relaxed mb-3">
   983→                  {renderContent(q.content)}
   984→                </div>
   985→                <div className="flex items-center gap-2 text-xs text-muted-foreground">
   986→                  <span>{timeAgo(q.createdAt)}</span>
   987→                  <span>· {qReplies.length} {qReplies.length === 1 ? 'reply' : 'replies'}</span>
   988→                </div>
   989→              </div>
   990→            </CardContent>

  1651→function ProgressTab({ progressData, enrollmentProgress, course, streakDays }: {
  1652→  progressData: any[];
  1653→  enrollmentProgress: number;
  1654→  course: any;
  1655→  streakDays: number;

  1657→  // Derive timeline events from progress data
  1658→  const timelineEvents = useMemo(() => {
  1659→    if (!progressData || !course) return [];
  1660→    const events: { id: string; lessonTitle: string; moduleName: string; date: string; timeSpent: number; type: 'completed' | 'started' }[] = [];
  1661→
  1662→    // Build lesson lookup from course modules
  1663→    const lessonLookup = new Map<string, { title: string; moduleName: string }>();
  1664→    (course.modules || []).forEach((mod: any) => {
  1665→      (mod.lessons || []).forEach((lesson: any) => {
  1666→        lessonLookup.set(lesson.id, { title: lesson.title, moduleName: mod.title });
  1667→      });
  1668→    });
  1669→
  1670→    progressData.forEach((p: any) => {
  1671→      const lessonInfo = lessonLookup.get(p.lessonId);
  1672→      if (!lessonInfo) return;
  1673→
  1674→      const date = p.completedAt
  1675→        ? new Date(p.completedAt).toLocaleDateString('en-US', { month: 'short', day: 'numeric', year: 'numeric' })
  1676→        : new Date(p.updatedAt || p.createdAt).toLocaleDateString('en-US', { month: 'short', day: 'numeric', year: 'numeric' });
  1677→
  1678→      events.push({
  1679→        id: p.id,
  1680→        lessonTitle: lessonInfo.title,
  1681→        moduleName: lessonInfo.moduleName,
  1682→        date,
  1683→        timeSpent: Math.round((p.timeSpent || 0) / 60),
  1684→        type: p.status === 'completed' ? 'completed' : 'started',
  1685→      });
  1686→    });
  1687→
  1688→    // Sort by date (most recent first)

  1691→  }, [progressData, course]);
  1692→
  1693→  // Compute weekly activity from progress

  1695→    const days = ['Mon', 'Tue', 'Wed', 'Thu', 'Fri', 'Sat', 'Sun'];

  1713→      return { day, hours: Math.round(totalMinutes * 10) / 10 };
  1714→    });
  1715→  }, [progressData]);
  1716→
  1717→  // Compute milestones from progress data
  1718→  const completedCount = progressData?.filter((p: any) => p.status === 'completed').length || 0;
  1719→  const totalLessons = (course?.modules || []).flatMap((m: any) => m.lessons || []).length;
  1720→  const modulesComplete = (course?.modules || []).filter((mod: any) =>
  1721→    (mod.lessons || []).every((l: any) => progressData?.some((p: any) => p.lessonId === l.id && p.status === 'completed'))
  1722→  ).length;

  1725→    { id: 'ms-1', title: 'First Lesson Completed', description: 'Completed your first lesson', icon: '🎯', achieved: completedCount >= 1, date: completedCount >= 1 ? timelineEvents.find(e => e.type === 'completed')?.date || '' : '' },

  1727→    { id: 'ms-3', title: 'Module Master', description: 'Complete all lessons in a module', icon: '🏆', achieved: modulesComplete > 0, date: '' },
  1728→    { id: 'ms-4', title: 'Halfway There', description: 'Complete 50% of the course', icon: '🎓', achieved: enrollmentProgress >= 50, date: '' },
  1729→    { id: 'ms-5', title: 'Streak Champion', description: 'Maintain a 7-day learning streak', icon: '🔥', achieved: streakDays >= 7, date: '' },
  1730→    { id: 'ms-6', title: 'Course Graduate', description: 'Complete the entire course', icon: '🏅', achieved: enrollmentProgress >= 100, date: '' },
  1731→  ];
  1732→
  1733→  const totalTime = Math.round(progressData?.reduce((sum: number, p: any) => sum + (p.timeSpent || 0), 0) / 60) || 0;
  1734→  const maxHours = Math.max(...weeklyActivity.map(d => d.hours), 0.1);
