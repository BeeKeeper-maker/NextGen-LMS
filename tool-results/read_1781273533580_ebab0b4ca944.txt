    69→// ─── Helpers ────────────────────────────────────────────────
    70→function formatTime(seconds: number): string {
    71→  const h = Math.floor(seconds / 3600);
    72→  const m = Math.floor((seconds % 3600) / 60);
    73→  const s = Math.floor(seconds % 60);
    74→  if (h > 0) return `${h}:${m.toString().padStart(2, '0')}:${s.toString().padStart(2, '0')}`;
    75→  return `${m}:${s.toString().padStart(2, '0')}`;
    76→}
    77→
    78→function formatDuration(seconds?: number): string {
    79→  if (!seconds) return '';
    80→  const mins = Math.floor(seconds / 60);
    81→  if (mins < 60) return `${mins} min`;
    82→  const hours = Math.floor(mins / 60);
    83→  const remainMins = mins % 60;
    84→  return `${hours}h ${remainMins}m`;
    85→}

    96→// ─── Segment Tracker ────────────────────────────────────────
    97→class SegmentTracker {
    98→  private segments: [number, number][] = [];
    99→  private currentStart: number | null = null;
   100→
   101→  startSegment(time: number) {
   102→    if (this.currentStart === null) {
   103→      this.currentStart = time;
   104→    }
   105→  }
   106→
   107→  updateSegment(time: number) {
   108→    if (this.currentStart !== null) {
   109→      // Extend or create segment
   110→      const last = this.segments[this.segments.length - 1];
   111→      if (last && Math.abs(time - last[1]) < SEGMENT_MERGE_THRESHOLD) {
   112→        last[1] = Math.max(last[1], time);
   113→      } else {
   114→        if (this.currentStart !== null) {
   115→          this.segments.push([this.currentStart, time]);
   116→          this.mergeSegments();
   117→        }
   118→      }
   119→    }
   120→  }
   121→
   122→  stopSegment(time: number) {
   123→    if (this.currentStart !== null) {
   124→      this.segments.push([this.currentStart, time]);
   125→      this.mergeSegments();
   126→      this.currentStart = null;
   127→    }
   128→  }
   129→
   130→  private mergeSegments() {
   131→    if (this.segments.length <= 1) return;
   132→    const merged: [number, number][] = [this.segments[0]];
   133→    for (let i = 1; i < this.segments.length; i++) {
   134→      const last = merged[merged.length - 1];
   135→      const cur = this.segments[i];
   136→      if (cur[0] - last[1] < SEGMENT_MERGE_THRESHOLD) {
   137→        last[1] = Math.max(last[1], cur[1]);
   138→      } else {
   139→        merged.push(cur);
   140→      }
   141→    }
   142→    this.segments = merged;
   143→  }
   144→
   145→  getWatchedPercent(totalDuration: number): number {
   146→    if (totalDuration <= 0) return 0;
   147→    const watched = this.segments.reduce((sum, seg) => sum + (seg[1] - seg[0]), 0);
   148→    return Math.min(100, (watched / totalDuration) * 100);
   149→  }
   150→
   151→  getSegments(): [number, number][] {
   152→    return [...this.segments];
   153→  }
   154→
   155→  seekTo(time: number) {
   156→    // When seeking, end current segment and start new
   157→    if (this.currentStart !== null) {
   158→      this.segments.push([this.currentStart, this.currentStart + 0.5]);
   159→      this.currentStart = time;
   160→    }
   161→  }
   162→}

   171→  chapters = [],
   172→  compact = false,

   174→  moduleName,
   175→  nextLessons = [],
   176→  onMarkComplete,
   177→  onNextLesson,
   178→  onBack,
   179→  isCompleted: isCompletedProp = false,
   180→  totalDuration: totalDurationProp,
   181→}: VideoPlayerProps) {
   182→  // Resolve props from lesson or direct
   183→  const resolvedTitle = title || lesson?.title || 'Untitled Lesson';
   184→  const resolvedModuleName = moduleName || '';
   185→  const totalDuration = totalDurationProp || lesson?.videoDuration || 300;
   186→  const resolvedContentType = lesson?.contentType || 'video';
   187→
   188→  // ─── State ──────────────────────────────────────────────
   189→  const [isPlaying, setIsPlaying] = useState(false);
   190→  const [currentTime, setCurrentTime] = useState(initialPosition);
   191→  const [volume, setVolume] = useState(1);
   192→  const [isMuted, setIsMuted] = useState(false);
   193→  const [playbackSpeed, setPlaybackSpeed] = useState(1);
   194→  const [showControls, setShowControls] = useState(true);
   195→  const [showTitle, setShowTitle] = useState(true);
   196→  const [showSpeedMenu, setShowSpeedMenu] = useState(false);
   197→  const [isFullscreen, setIsFullscreen] = useState(false);
   198→  const [isPiP, setIsPiP] = useState(false);
   199→  const [isCompleted, setIsCompleted] = useState(isCompletedProp);
   200→  const [hoverTime, setHoverTime] = useState<number | null>(null);
   201→  const [hoverX, setHoverX] = useState(0);
   202→  const [progressBarWidth, setProgressBarWidth] = useState(0);
   203→  const [showResumePrompt, setShowResumePrompt] = useState(initialPosition > 0);
   204→  const [watchedPercent, setWatchedPercent] = useState(0);
   205→  const [bufferedPercent, setBufferedPercent] = useState(0);
   206→
   207→  // ─── Refs ───────────────────────────────────────────────
   208→  const containerRef = useRef<HTMLDivElement>(null);
   209→  const progressRef = useRef<HTMLDivElement>(null);
   210→  const controlsTimeoutRef = useRef<ReturnType<typeof setTimeout> | null>(null);
   211→  const intervalRef = useRef<ReturnType<typeof setInterval> | null>(null);
   212→  const segmentTrackerRef = useRef(new SegmentTracker());
   213→  const lastTimeRef = useRef(initialPosition);
   214→  const touchStartRef = useRef<{ x: number; y: number; time: number } | null>(null);
   215→
   216→  // ─── Computed ───────────────────────────────────────────
   217→  const progressPercent = (currentTime / totalDuration) * 100;
   218→  const displayVolume = isMuted ? 0 : volume;
   219→
   220→  const chapterAtTime = useMemo(() => {
   221→    if (chapters.length === 0) return null;
   222→    let current = chapters[0];
   223→    for (const ch of chapters) {
   224→      if (currentTime >= ch.time) current = ch;
   225→      else break;
   226→    }
   227→    return current;
   228→  }, [currentTime, chapters]);
   229→
   230→  // ─── Controls visibility ────────────────────────────────
   231→  const resetControlsTimer = useCallback(() => {
   232→    setShowControls(true);
   233→    if (controlsTimeoutRef.current) clearTimeout(controlsTimeoutRef.current);
   234→    if (isPlaying) {
   235→      controlsTimeoutRef.current = setTimeout(() => {
   236→        setShowControls(false);
   237→        setShowTitle(false);
   238→      }, CONTROLS_HIDE_DELAY);
   239→    }
   240→  }, [isPlaying]);
   241→
   242→  const handleMouseMove = useCallback(() => {
   243→    resetControlsTimer();
   244→  }, [resetControlsTimer]);
   245→
   246→  // ─── Playback simulation ────────────────────────────────
   247→  useEffect(() => {
   248→    if (isPlaying) {
   249→      segmentTrackerRef.current.startSegment(currentTime);
   250→      intervalRef.current = setInterval(() => {
   251→        setCurrentTime((prev) => {
   252→          const next = prev + playbackSpeed;
   253→          lastTimeRef.current = prev;
   254→
   255→          // Update segment tracker
   256→          segmentTrackerRef.current.updateSegment(next);
   257→
   258→          // Calculate watched percentage
   259→          const wp = segmentTrackerRef.current.getWatchedPercent(totalDuration);
   260→          setWatchedPercent(wp);
   261→
   262→          // Report progress
   263→          if (onProgress) {
   264→            onProgress(wp, next, segmentTrackerRef.current.getSegments());
   265→          }
   266→
   267→          // Auto-complete at threshold
   268→          if (wp >= AUTO_COMPLETE_THRESHOLD * 100 && !isCompleted) {
   269→            setIsCompleted(true);
   270→            if (onComplete) onComplete();
   271→            if (onMarkComplete && lesson?.id) onMarkComplete(lesson.id);
   272→          }
   273→
   274→          if (next >= totalDuration) {
   275→            setIsPlaying(false);
   276→            segmentTrackerRef.current.stopSegment(totalDuration);
   277→            if (!isCompleted) {
   278→              setIsCompleted(true);
   279→              if (onComplete) onComplete();
   280→              if (onMarkComplete && lesson?.id) onMarkComplete(lesson.id);
   281→            }
   282→            return totalDuration;
   283→          }
   284→          return next;
   285→        });
   286→      }, 1000);
   287→    } else {
   288→      segmentTrackerRef.current.stopSegment(lastTimeRef.current);
   289→      if (intervalRef.current) clearInterval(intervalRef.current);
   290→    }
   291→    return () => {
   292→      if (intervalRef.current) clearInterval(intervalRef.current);
   293→    };
   294→  }, [isPlaying, playbackSpeed, totalDuration]);
   295→
   296→  // Simulated buffering
   297→  useEffect(() => {
   298→    if (isPlaying) {
   299→      const bufferInterval = setInterval(() => {
   300→        setBufferedPercent(Math.min(100, progressPercent + 10 + Math.random() * 15));
   301→      }, 2000);
   302→      return () => clearInterval(bufferInterval);
   303→    }
   304→  }, [isPlaying, progressPercent]);
   305→
   306→  // ─── Toggle play ────────────────────────────────────────
   307→  const togglePlay = useCallback(() => {
   308→    const willPlay = !isPlaying;
   309→    setIsPlaying(willPlay);
   310→    setShowResumePrompt(false);
   311→    if (willPlay) {
   312→      setShowTitle(true);
   313→      resetControlsTimer();
   314→      setTimeout(() => setShowTitle(false), 3000);
   315→    } else {
   316→      if (controlsTimeoutRef.current) clearTimeout(controlsTimeoutRef.current);
   317→      setShowControls(true);
   318→      setShowTitle(true);
   319→    }
   320→  }, [isPlaying, resetControlsTimer]);
   321→
   322→  // ─── Seek ───────────────────────────────────────────────
   323→  const seekTo = useCallback((time: number) => {
   324→    const clamped = Math.max(0, Math.min(time, totalDuration));
   325→    segmentTrackerRef.current.seekTo(clamped);
   326→    setCurrentTime(clamped);
   327→    lastTimeRef.current = clamped;
   328→  }, [totalDuration]);
   329→
   330→  const skipForward = useCallback(() => seekTo(currentTime + 10), [currentTime, seekTo]);
   331→  const skipBackward = useCallback(() => seekTo(currentTime - 10), [currentTime, seekTo]);
   332→
   333→  // ─── Progress bar click ─────────────────────────────────
   334→  const handleProgressClick = useCallback(
   335→    (e: React.MouseEvent<HTMLDivElement>) => {
   336→      if (!progressRef.current) return;
   337→      const rect = progressRef.current.getBoundingClientRect();
   338→      const x = e.clientX - rect.left;
   339→      const percent = Math.max(0, Math.min(1, x / rect.width));
   340→      seekTo(percent * totalDuration);
   341→    },
   342→    [totalDuration, seekTo]
   343→  );
   344→
   345→  const handleProgressHover = useCallback(
   346→    (e: React.MouseEvent<HTMLDivElement>) => {
   347→      if (!progressRef.current) return;
   348→      const rect = progressRef.current.getBoundingClientRect();
   349→      const x = e.clientX - rect.left;
   350→      const percent = Math.max(0, Math.min(1, x / rect.width));
   351→      setHoverTime(percent * totalDuration);
   352→      setHoverX(e.clientX - rect.left);
   353→      setProgressBarWidth(rect.width);
   354→    },
   355→    [totalDuration]
   356→  );
   357→
   358→  const handleProgressLeave = useCallback(() => {
   359→    setHoverTime(null);
   360→  }, []);
   361→
   362→  // ─── Volume ─────────────────────────────────────────────
   363→  const toggleMute = useCallback(() => setIsMuted(!isMuted), [isMuted]);
   364→  const handleVolumeChange = useCallback((e: React.ChangeEvent<HTMLInputElement>) => {
   365→    const val = parseFloat(e.target.value);
   366→    setVolume(val);
   367→    setIsMuted(val === 0);
   368→  }, []);
   369→
   370→  // ─── Fullscreen ─────────────────────────────────────────
   371→  const toggleFullscreen = useCallback(() => {
   372→    if (!containerRef.current) return;
   373→    if (!document.fullscreenElement) {
   374→      containerRef.current.requestFullscreen().then(() => setIsFullscreen(true)).catch(() => {});
   375→    } else {
   376→      document.exitFullscreen().then(() => setIsFullscreen(false)).catch(() => {});
   377→    }
   378→  }, []);
   379→
   380→  useEffect(() => {
   381→    const handler = () => setIsFullscreen(!!document.fullscreenElement);
   382→    document.addEventListener('fullscreenchange', handler);
   383→    return () => document.removeEventListener('fullscreenchange', handler);
   384→  }, []);
   385→
   386→  // ─── PiP (simulated) ───────────────────────────────────
   387→  const togglePiP = useCallback(() => {
   388→    setIsPiP(!isPiP);
   389→  }, [isPiP]);
   390→
   391→  // ─── Resume ─────────────────────────────────────────────

   393→    setShowResumePrompt(false);
   394→    setIsPlaying(true);
   395→    resetControlsTimer();
   396→  }, [resetControlsTimer]);
   397→
   398→  const handleStartFromBeginning = useCallback(() => {
   399→    setCurrentTime(0);
   400→    lastTimeRef.current = 0;

   402→    setIsPlaying(true);

   405→
   406→  // ─── Keyboard shortcuts ─────────────────────────────────
   407→  useEffect(() => {
   408→    const handler = (e: KeyboardEvent) => {
   409→      // Don't capture if focused on input
   410→      if (e.target instanceof HTMLInputElement || e.target instanceof HTMLTextAreaElement) return;
   411→
   412→      switch (e.key) {
   413→        case ' ':
   414→        case 'k':
   415→          e.preventDefault();
   416→          togglePlay();
   417→          break;
   418→        case 'ArrowLeft':
   419→          e.preventDefault();
   420→          skipBackward();
   421→          break;
   422→        case 'ArrowRight':
   423→          e.preventDefault();
   424→          skipForward();
   425→          break;
   426→        case 'ArrowUp':
   427→          e.preventDefault();
   428→          setVolume((v) => {
   429→            const nv = Math.min(1, v + 0.1);

   432→          });
   433→          break;
   434→        case 'ArrowDown':
   435→          e.preventDefault();
   436→          setVolume((v) => {
   437→            const nv = Math.max(0, v - 0.1);
   438→            setIsMuted(nv === 0);
   439→            return nv;
   440→          });
   441→          break;
   442→        case 'f':
   443→        case 'F':
   444→          e.preventDefault();
   445→          toggleFullscreen();
   446→          break;
   447→        case 'm':
   448→        case 'M':
   449→          e.preventDefault();

   451→          break;
   452→        case '<':
   453→        case ',':
   454→          e.preventDefault();
   455→          setPlaybackSpeed((s) => {
   456→            const idx = PLAYBACK_SPEEDS.indexOf(s);
   457→            return idx > 0 ? PLAYBACK_SPEEDS[idx - 1] : s;
   458→          });
   459→          break;
   460→        case '>':
   461→        case '.':
   462→          e.preventDefault();
   463→          setPlaybackSpeed((s) => {
   464→            const idx = PLAYBACK_SPEEDS.indexOf(s);
   465→            return idx < PLAYBACK_SPEEDS.length - 1 ? PLAYBACK_SPEEDS[idx + 1] : s;
   466→          });
   467→          break;
   468→      }
   469→    };
   470→    window.addEventListener('keydown', handler);
   471→    return () => window.removeEventListener('keydown', handler);
   472→  }, [togglePlay, skipForward, skipBackward, toggleFullscreen, toggleMute]);
   473→
   474→  // ─── Touch / Swipe handling ─────────────────────────────

   477→      x: e.touches[0].clientX,
   478→      y: e.touches[0].clientY,
   479→      time: Date.now(),
   480→    };
   481→  }, []);
   482→
   483→  const handleTouchEnd = useCallback((e: React.TouchEvent) => {
   484→    if (!touchStartRef.current) return;
   485→    const dx = e.changedTouches[0].clientX - touchStartRef.current.x;
   486→    const dy = e.changedTouches[0].clientY - touchStartRef.current.y;
   487→    const dt = Date.now() - touchStartRef.current.time;
   488→
   489→    // Only horizontal swipes
   490→    if (Math.abs(dx) > 50 && Math.abs(dx) > Math.abs(dy) * 2 && dt < 500) {
   491→      const seekAmount = (dx / 300) * 30; // up to 30s based on swipe distance
   492→      seekTo(currentTime + seekAmount);
   493→    }
   494→    touchStartRef.current = null;
   495→  }, [currentTime, seekTo]);
   496→
   497→  // ─── Mark complete manually ─────────────────────────────
   498→  const handleMarkComplete = () => {
   499→    setIsCompleted(true);
   500→    if (onComplete) onComplete();
   501→    if (onMarkComplete && lesson?.id) onMarkComplete(lesson.id);
   502→  };
   503→
   504→  // ─── Chapter markers on progress bar ────────────────────
   505→  const chapterMarkers = useMemo(() => {
   506→    if (chapters.length === 0) return null;
   507→    return chapters
   508→      .filter((ch) => ch.time > 0)
   509→      .map((ch, i) => (
   510→        <div
   511→          key={i}
   512→          className="absolute top-0 bottom-0 w-0.5 bg-white/40 z-10"
   513→          style={{ left: `${(ch.time / totalDuration) * 100}%` }}
   514→          title={ch.title}
   515→        />
   516→      ));
   517→  }, [chapters, totalDuration]);
   518→
   519→  // ─── Volume icon ────────────────────────────────────────
   520→  const VolumeIcon = displayVolume === 0 ? VolumeX : displayVolume < 0.5 ? Volume1 : Volume2;
   521→
   522→  return (
   523→    <div className="space-y-4">
   524→      {/* Back Button */}
   525→      {onBack && !compact && (
   526→        <motion.div initial={{ opacity: 0, x: -10 }} animate={{ opacity: 1, x: 0 }}>
   527→          <Button variant="ghost" size="sm" onClick={onBack} className="gap-1.5 text-muted-foreground hover:text-foreground">
   528→            <ChevronRight className="h-4 w-4 rotate-180" />
   529→            Back to Course
   530→          </Button>
   531→        </motion.div>
   532→      )}
   533→
   534→      <div className={cn(compact ? 'flex flex-col' : 'flex flex-col lg:flex-row gap-4')}>
   535→        {/* ─── Video Player Container ──────────────────────── */}
   536→        <div className="flex-1 min-w-0">
   537→          <motion.div
   538→            ref={containerRef}
   539→            initial={{ opacity: 0, y: 10 }}
   540→            animate={{ opacity: 1, y: 0 }}
   541→            transition={{ duration: 0.4 }}
   542→            className={cn(
   543→              'relative rounded-xl overflow-hidden bg-black shadow-2xl group/container',
   544→              isPiP && 'fixed bottom-4 right-4 w-80 h-45 z-50 rounded-lg shadow-2xl'
   545→            )}
   546→            onMouseMove={handleMouseMove}
   547→            onMouseLeave={() => isPlaying && setShowControls(false)}
   548→            onTouchStart={handleTouchStart}
   549→            onTouchEnd={handleTouchEnd}
   550→          >
   551→            {/* 16:9 Aspect Ratio Container */}
   552→            <div className="relative w-full" style={{ aspectRatio: isPiP ? '16/9' : '16/9' }}>
   553→              {/* Simulated Video Content - Animated Gradient */}
   554→              <div className="absolute inset-0">
   555→                <div
   556→                  className="absolute inset-0"
   557→                  style={{
   558→                    background: isPlaying
   559→                      ? `linear-gradient(${135 + currentTime * 0.5}deg, 
   560→                          #0f172a 0%, 
   561→                          #1e293b 25%, 
   562→                          #334155 50%, 
   563→                          #1e293b 75%, 
   564→                          #0f172a 100%)`
   565→                      : 'linear-gradient(135deg, #0f172a 0%, #1e293b 50%, #334155 100%)',
   566→                    transition: 'background 2s ease',
   567→                  }}
   568→                />
   569→                {/* Animated wave pattern */}
   570→                {isPlaying && (
   571→                  <div className="absolute inset-0 opacity-20">
   572→                    <div
