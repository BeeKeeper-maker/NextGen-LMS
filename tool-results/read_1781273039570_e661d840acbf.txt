   154→   154→// Relative time helper
   155→   155→function getRelativeTime(dateStr?: string) {
   156→   156→  if (!dateStr) return '';
   157→   157→  const now = new Date();
   158→   158→  const date = new Date(dateStr);
   159→   159→  const diffMs = now.getTime() - date.getTime();
   160→   160→  const diffMins = Math.floor(diffMs / 60000);
   161→   161→  const diffHours = Math.floor(diffMins / 60);
   162→   162→  const diffDays = Math.floor(diffHours / 24);
   163→   163→
   164→   164→  if (diffMins < 1) return 'Just now';
   165→   165→  if (diffMins < 60) return `${diffMins}m ago`;
   166→   166→  if (diffHours < 24) return `${diffHours}h ago`;
   167→   167→  if (diffDays < 7) return `${diffDays}d ago`;
   168→   168→  return date.toLocaleDateString();
   169→   169→}

   222→   222→// ─── Animated Counter ─────────────────────────────
   223→   223→function AnimatedCounter({ value, duration = 1.5 }: { value: string; duration?: number }) {
   224→   224→  const displayRef = useRef<HTMLSpanElement>(null);
   225→   225→  const hasAnimated = useRef(false);
   226→   226→
   227→   227→  useEffect(() => {
   228→   228→    if (hasAnimated.current) return;
   229→   229→    hasAnimated.current = true;
   230→   230→
   231→   231→    const el = displayRef.current;
   232→   232→    if (!el) return;
   233→   233→
   234→   234→    const numericMatch = value.match(/[\d,]+/);
   235→   235→    if (!numericMatch) {
   236→   236→      el.textContent = value;
   237→   237→      return;
   238→   238→    }
   239→   239→    const numericStr = numericMatch[0].replace(/,/g, '');
   240→   240→    const target = parseInt(numericStr, 10);
   241→   241→    if (isNaN(target)) {
   242→   242→      el.textContent = value;
   243→   243→      return;
   244→   244→    }
   245→   245→
   246→   246→    const prefix = value.substring(0, value.indexOf(numericMatch[0]));
   247→   247→    const suffix = value.substring(value.indexOf(numericMatch[0]) + numericMatch[0].length);
   248→   248→    const startTime = performance.now();
   249→   249→
   250→   250→    const animate = (currentTime: number) => {
   251→   251→      const elapsed = currentTime - startTime;
   252→   252→      const progress = Math.min(elapsed / (duration * 1000), 1);
   253→   253→      const eased = 1 - Math.pow(1 - progress, 3);
   254→   254→      const current = Math.round(eased * target);
   255→   255→      const formatted = current.toLocaleString();
   256→   256→      if (displayRef.current) {
   257→   257→        displayRef.current.textContent = `${prefix}${formatted}${suffix}`;
   258→   258→      }
   259→   259→      if (progress < 1) {
   260→   260→        requestAnimationFrame(animate);
   261→   261→      }
   262→   262→    };
   263→   263→    requestAnimationFrame(animate);
   264→   264→  }, [value, duration]);
   265→   265→
   266→   266→  return <span ref={displayRef}>0</span>;
   267→   267→}

   270→   270→function MiniProgressRing({ percentage, size = 36, strokeWidth = 3, color = '#10b981' }: { percentage: number; size?: number; strokeWidth?: number; color?: string }) {
   271→   271→  const [animatedPercent, setAnimatedPercent] = useState(0);
   272→   272→
   273→   273→  useEffect(() => {
   274→   274→    const timer = setTimeout(() => setAnimatedPercent(percentage), 200);
   275→   275→    return () => clearTimeout(timer);
   276→   276→  }, [percentage]);
   277→   277→
   278→   278→  const radius = (size - strokeWidth) / 2;
   279→   279→  const circumference = 2 * Math.PI * radius;
   280→   280→  const strokeDashoffset = circumference - (animatedPercent / 100) * circumference;
   281→   281→
   282→   282→  return (
   283→   283→    <svg width={size} height={size} className="-rotate-90">
   284→   284→      <circle
   285→   285→        cx={size / 2}
   286→   286→        cy={size / 2}
   287→   287→        r={radius}
   288→   288→        fill="none"
   289→   289→        stroke="currentColor"
   290→   290→        strokeWidth={strokeWidth}
   291→   291→        className="text-slate-200 dark:text-slate-700"
   292→   292→      />
   293→   293→      <motion.circle
   294→   294→        cx={size / 2}
   295→   295→        cy={size / 2}
   296→   296→        r={radius}
   297→   297→        fill="none"
   298→   298→        stroke={color}
   299→   299→        strokeWidth={strokeWidth}
   300→   300→        strokeLinecap="round"
   301→   301→        strokeDasharray={circumference}
   302→   302→        initial={{ strokeDashoffset: circumference }}
   303→   303→        animate={{ strokeDashoffset }}
   304→   304→        transition={{ duration: 1.2, ease: 'easeOut', delay: 0.3 }}
   305→   305→      />
   306→   306→    </svg>
   307→   307→  );
   308→   308→}

   330→   330→function FloatingShape({ type, className, delay, duration }: { type: 'circle' | 'square' | 'triangle' | 'diamond'; className?: string; delay?: number; duration?: number }) {
   331→   331→  const shapeClass = cn(
   332→   332→    'absolute opacity-[0.07] dark:opacity-[0.05]',
   333→   333→    type === 'circle' && 'rounded-full',
   334→   334→    type === 'square' && 'rounded-lg rotate-12',
   335→   335→    type === 'diamond' && 'rounded-md rotate-45',
   336→   336→    className,
   337→   337→  );
   338→   338→
   339→   339→  return (
   340→   340→    <motion.div
   341→   341→      className={shapeClass}
   342→   342→      initial={{ y: 0, rotate: type === 'square' ? 12 : type === 'diamond' ? 45 : 0 }}
   343→   343→      animate={{
   344→   344→        y: [-10, 10, -10],
   345→   345→        rotate: type === 'square' ? [12, 24, 12] : type === 'diamond' ? [45, 55, 45] : [0, 5, 0],
   346→   346→      }}
   347→   347→      transition={{
   348→   348→        duration: duration || 6,
   349→   349→        repeat: Infinity,
   350→   350→        ease: 'easeInOut',
   351→   351→        delay: delay || 0,
   352→   352→      }}
   353→   353→    />
   354→   354→  );
