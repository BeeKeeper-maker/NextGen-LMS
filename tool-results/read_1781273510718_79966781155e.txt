   421→// ─── Trend Line Data (Linear Regression) ─────────────────────
   422→function computeLinearRegression(data: { duration: number; completion: number }[]): { x1: number; y1: number; x2: number; y2: number } {
   423→  const n = data.length;
   424→  let sumX = 0, sumY = 0, sumXY = 0, sumXX = 0;
   425→  data.forEach(d => {
   426→    sumX += d.duration;
   427→    sumY += d.completion;
   428→    sumXY += d.duration * d.completion;
   429→    sumXX += d.duration * d.duration;
   430→  });
   431→  const slope = (n * sumXY - sumX * sumY) / (n * sumXX - sumX * sumX);
   432→  const intercept = (sumY - slope * sumX) / n;
   433→  const minX = Math.min(...data.map(d => d.duration)) - 2;
   434→  const maxX = Math.max(...data.map(d => d.duration)) + 2;
   435→  return {
   436→    x1: minX,
   437→    y1: Math.max(0, Math.min(100, slope * minX + intercept)),
   438→    x2: maxX,
   439→    y2: Math.max(0, Math.min(100, slope * maxX + intercept)),
   440→  };
   441→}

   497→// ─── Animated Number Counter ────────────────────────────────
   498→function AnimatedCounter({ value, prefix = '', suffix = '', duration = 1500 }: {
   499→  value: number;
   500→  prefix?: string;
   501→  suffix?: string;
   502→  duration?: number;
   503→}) {
   504→  const [displayed, setDisplayed] = useState(0);
   505→  const startTime = useRef<number | null>(null);
   506→  const frameRef = useRef<number>(0);
   507→  const animateRef = useRef<((timestamp: number) => void) | null>(null);
   508→
   509→  useEffect(() => {
   510→    animateRef.current = (timestamp: number) => {
   511→      if (!startTime.current) startTime.current = timestamp;
   512→      const progress = Math.min((timestamp - startTime.current) / duration, 1);
   513→      const eased = 1 - Math.pow(1 - progress, 3);
   514→      setDisplayed(Math.floor(eased * value));
   515→      if (progress < 1) {
   516→        frameRef.current = requestAnimationFrame((ts) => animateRef.current?.(ts));
   517→      }
   518→    };
   519→    startTime.current = null;
   520→    frameRef.current = requestAnimationFrame((ts) => animateRef.current?.(ts));
   521→    return () => cancelAnimationFrame(frameRef.current);
   522→  }, [value, duration]);
   523→
   524→  return (
   525→    <span>
   526→      {prefix}{displayed.toLocaleString()}{suffix}
   527→    </span>
   528→  );
   529→}
