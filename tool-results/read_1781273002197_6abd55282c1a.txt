   213→// ---- Data Mapping Helpers ----
   214→function apiCertToTemplate(cert: Record<string, unknown>): CertTemplate {
   215→  let parsed: Record<string, unknown> = {};
   216→  try {
   217→    const templateField = cert.template;
   218→    parsed = typeof templateField === 'string' ? JSON.parse(templateField) : (templateField as Record<string, unknown>) || {};
   219→  } catch {
   220→    parsed = {};
   221→  }
   222→  return {
   223→    id: cert.id as string,
   224→    tenantId: cert.tenantId as string,
   225→    name: cert.name as string,
   226→    template: cert.template as string,
   227→    backgroundUrl: (cert.backgroundUrl as string) || '',
   228→    isActive: cert.isActive as boolean,
   229→    elements: (parsed.elements as CertElement[]) || defaultElements,
   230→    bg_color: (parsed.bg_color as string) || '#fefefe',
   231→    font_family: (parsed.font_family as string) || 'Georgia, serif',
   232→    width: (parsed.width as number) || 800,
   233→    height: (parsed.height as number) || 600,
   234→    borderStyle: parsed.borderStyle as string | undefined,
   235→    sealType: parsed.sealType as string | undefined,
   236→    hasWatermark: parsed.hasWatermark as boolean | undefined,
   237→    backgroundTemplate: parsed.backgroundTemplate as string | undefined,
   238→  };
   239→}

   274→// ---- Helper: Animated Counter ----
   275→function AnimatedCounter({ target, duration = 1500 }: { target: number; duration?: number }) {
   276→  const [count, setCount] = useState(0);
   277→  const prevTarget = useRef(0);
   278→
   279→  useEffect(() => {
   280→    if (target === prevTarget.current) return;
   281→    prevTarget.current = target;
   282→    let start = 0;
   283→    const step = target / (duration / 16);
   284→    const timer = setInterval(() => {
   285→      start += step;
   286→      if (start >= target) {
   287→        setCount(target);
   288→        clearInterval(timer);
   289→      } else {
   290→        setCount(Math.floor(start));
   291→      }
   292→    }, 16);
   293→    return () => clearInterval(timer);
   294→  }, [target, duration]);
   295→
   296→  return <span>{count}</span>;
   297→}

   432→function StatsDashboard({ awards, templates }: { awards: IssuedCert[]; templates: CertTemplate[] }) {
   433→  // Compute monthly issuance from real data
   434→  const monthlyIssuance = useMemo(() => {
   435→    const monthMap: Record<string, number> = {};
   436→    const monthNames = ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun', 'Jul', 'Aug', 'Sep', 'Oct', 'Nov', 'Dec'];
   437→    const now = new Date();
   438→    // Show last 7 months
   439→    for (let i = 6; i >= 0; i--) {
   440→      const d = new Date(now.getFullYear(), now.getMonth() - i, 1);
   441→      const key = `${d.getFullYear()}-${String(d.getMonth() + 1).padStart(2, '0')}`;
   442→      monthMap[key] = 0;
   443→    }
   444→    awards.forEach(a => {
   445→      const d = new Date(a.issuedAt);
   446→      const key = `${d.getFullYear()}-${String(d.getMonth() + 1).padStart(2, '0')}`;
   447→      if (key in monthMap) {
   448→        monthMap[key]++;
   449→      }
   450→    });
   451→    return Object.entries(monthMap).map(([key, count]) => {
   452→      const [y, m] = key.split('-');
   453→      return { month: monthNames[parseInt(m) - 1], count };
   454→    });
   455→  }, [awards]);
   456→
   457→  const maxMonthlyCount = Math.max(...monthlyIssuance.map(m => m.count), 1);
   458→
   459→  // Compute stats from real data
   460→  const thisMonth = useMemo(() => {
   461→    const now = new Date();
   462→    return awards.filter(a => {
   463→      const d = new Date(a.issuedAt);
   464→      return d.getMonth() === now.getMonth() && d.getFullYear() === now.getFullYear();
   465→    }).length;
   466→  }, [awards]);
   467→
   468→  const topCourse = useMemo(() => {
   469→    const courseCount: Record<string, number> = {};
   470→    awards.forEach(a => {
   471→      courseCount[a.courseName] = (courseCount[a.courseName] || 0) + 1;
   472→    });
   473→    const sorted = Object.entries(courseCount).sort(([, a], [, b]) => b - a);
   474→    return sorted[0]?.[0] || 'N/A';
   475→  }, [awards]);
   476→
   477→  const verifiedCount = awards.filter(a => a.status === 'verified').length;
   478→  const verificationRate = awards.length > 0 ? Math.round((verifiedCount / awards.length) * 100) : 0;
   479→
   480→  const stats = [
   481→    {
   482→      label: 'Total Issued',
   483→      value: awards.length,
   484→      icon: Award,
