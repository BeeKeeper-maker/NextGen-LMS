   276вҶ’   276вҶ’function computeLinearRegression(data: { duration: number; completion: number }[]): { x1: number; y1: number; x2: number; y2: number } {
   277вҶ’   277вҶ’  const n = data.length;
   278вҶ’   278вҶ’  let sumX = 0, sumY = 0, sumXY = 0, sumXX = 0;
   279вҶ’   279вҶ’  data.forEach(d => {
   280вҶ’   280вҶ’    sumX += d.duration;
   281вҶ’   281вҶ’    sumY += d.completion;
   282вҶ’   282вҶ’    sumXY += d.duration * d.completion;
   283вҶ’   283вҶ’    sumXX += d.duration * d.duration;
   284вҶ’   284вҶ’  });
   285вҶ’   285вҶ’  const slope = (n * sumXY - sumX * sumY) / (n * sumXX - sumX * sumX);
   286вҶ’   286вҶ’  const intercept = (sumY - slope * sumX) / n;
   287вҶ’   287вҶ’  const minX = Math.min(...data.map(d => d.duration)) - 2;
   288вҶ’   288вҶ’  const maxX = Math.max(...data.map(d => d.duration)) + 2;
   289вҶ’   289вҶ’  return {
   290вҶ’   290вҶ’    x1: minX,
   291вҶ’   291вҶ’    y1: Math.max(0, Math.min(100, slope * minX + intercept)),
   292вҶ’   292вҶ’    x2: maxX,
   293вҶ’   293вҶ’    y2: Math.max(0, Math.min(100, slope * maxX + intercept)),
   294вҶ’   294вҶ’  };
