   117→function timeAgo(dateStr: string) {
   118→  const now = new Date();
   119→  const date = new Date(dateStr);
   120→  const diffMs = now.getTime() - date.getTime();
   121→  const diffMins = Math.floor(diffMs / 60000);
   122→  if (diffMins < 60) return `${diffMins}m ago`;
   123→  const diffHours = Math.floor(diffMins / 60);
   124→  if (diffHours < 24) return `${diffHours}h ago`;
   125→  const diffDays = Math.floor(diffHours / 24);
   126→  if (diffDays < 30) return `${diffDays}d ago`;
   127→  return `${Math.floor(diffDays / 30)}mo ago`;
   128→}

   142→function AnimatedCounter({ target, duration = 1.5 }: { target: number; duration?: number }) {
   143→  const [count, setCount] = useState(0);
   144→  useEffect(() => {
   145→    let start = 0;
   146→    const increment = target / (duration * 60);
   147→    const timer = setInterval(() => {
   148→      start += increment;
   149→      if (start >= target) {
   150→        setCount(target);
   151→        clearInterval(timer);
   152→      } else {
   153→        setCount(Math.floor(start));
   154→      }
   155→    }, 1000 / 60);
   156→    return () => clearInterval(timer);
   157→  }, [target, duration]);
   158→
   159→  return <span>{count.toLocaleString()}</span>;
   160→}

   164→function SparklineChart({ data, color = '#10B981', width = 48, height = 20 }: { data: number[]; color?: string; width?: number; height?: number }) {
   165→  if (data.length < 2) return null;
   166→  const max = Math.max(...data);
   167→  const min = Math.min(...data);
   168→  const range = max - min || 1;
   169→  const padding = 2;
   170→
   171→  const points = data.map((val, i) => {
   172→    const x = padding + (i / (data.length - 1)) * (width - padding * 2);
   173→    const y = height - padding - ((val - min) / range) * (height - padding * 2);
   174→    return `${x},${y}`;
   175→  }).join(' ');
   176→
   177→  const areaPoints = `${padding},${height - padding} ${points} ${width - padding},${height - padding}`;
   178→
   179→  return (
   180→    <svg width={width} height={height} className="shrink-0">
   181→      <defs>
   182→        <linearGradient id={`grad-${color.replace('#', '')}`} x1="0" y1="0" x2="0" y2="1">
   183→          <stop offset="0%" stopColor={color} stopOpacity="0.3" />
   184→          <stop offset="100%" stopColor={color} stopOpacity="0.05" />
   185→        </linearGradient>
   186→      </defs>
   187→      <polygon points={areaPoints} fill={`url(#grad-${color.replace('#', '')})`} />
   188→      <polyline points={points} fill="none" stroke={color} strokeWidth="1.5" strokeLinecap="round" strokeLinejoin="round" />
   189→    </svg>
   190→  );
   191→}

   230→          >

   889→function PostFeedCard({
   890→  post,
   891→  index,
   892→  isExpanded,
   893→  isLiked,
   894→  isBookmarked,
   895→  reactions,
   896→  userReactions,
   897→  floatingEmojis,
   898→  onToggleComments,
   899→  onToggleLike,
   900→  onToggleBookmark,
   901→  onEmojiReaction,
   902→  onHeartBurst,
   903→}: {
   904→  post: CommunityPost;
   905→  index: number;
   906→  isExpanded: boolean;
   907→  isLiked: boolean;
   908→  isBookmarked: boolean;
   909→  reactions: Record<string, number>;
   910→  userReactions: Set<string>;
   911→  floatingEmojis: { id: string; postId: string; emoji: string }[];
   912→  onToggleComments: () => void;
   913→  onToggleLike: () => void;
   914→  onToggleBookmark: () => void;
   915→  onEmojiReaction: (emoji: EmojiType) => void;
   916→  onHeartBurst: (e: React.MouseEvent) => void;
   917→}) {
   918→  const roleBadge = getRoleBadge(post.author?.role || 'learner');
   919→  const readingTime = estimateReadingTime(post.content);
   920→  const engagementScore = getEngagementScore(post);
   921→
   922→  // Find most popular reaction
   923→  const topReaction = Object.entries(reactions).sort((a, b) => b[1] - a[1])[0];
   924→
   925→  // Gradient border state
   926→  const [isHovered, setIsHovered] = useState(false);
   927→
   928→  return (
