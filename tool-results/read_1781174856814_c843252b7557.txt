   216→function TypingText({ text, speed = 35, delay = 0 }: { text: string; speed?: number; delay?: number }) {
   217→  const [displayed, setDisplayed] = useState('');
   218→  const [started, setStarted] = useState(false);
   219→
   220→  useEffect(() => {
   221→    const timer = setTimeout(() => setStarted(true), delay);
   222→    return () => clearTimeout(timer);
   223→  }, [delay]);
   224→
   225→  useEffect(() => {
   226→    if (!started) return;
   227→    if (displayed.length < text.length) {
   228→      const timer = setTimeout(() => {
   229→        setDisplayed(text.slice(0, displayed.length + 1));
   230→      }, speed);
   231→      return () => clearTimeout(timer);
   232→    }
   233→  }, [displayed, text, speed, started]);
   234→
   235→  return (
   236→    <span>
   237→      {displayed}
   238→      {displayed.length < text.length && started && (
   239→        <span className="inline-block w-0.5 h-5 bg-emerald-500 ml-0.5 animate-pulse align-middle" />
   240→      )}
   241→    </span>
   242→  );
   243→}

   532→
   533→    const start = count;
   534→    const diff = target - start;
   535→    const startTime = performance.now();
   536→
   537→    const animate = (now: number) => {
   538→      const elapsed = now - startTime;
   539→      const progress = Math.min(elapsed / duration, 1);
   540→      // easeOutCubic
   541→      const eased = 1 - Math.pow(1 - progress, 3);
   542→      setCount(Math.round(start + diff * eased));
   543→      if (progress < 1) {
   544→        requestAnimationFrame(animate);
   545→      }
   546→    };
   547→
   548→    requestAnimationFrame(animate);
   549→  }, [target, duration]);
   550→
   551→  return count;
   552→}
