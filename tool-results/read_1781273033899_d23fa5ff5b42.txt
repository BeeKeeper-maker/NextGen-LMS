   124→// ─── Format time helper ─────────────────────────────────────
   125→function formatTime(seconds: number): string {
   126→  const h = Math.floor(seconds / 3600);
   127→  const m = Math.floor((seconds % 3600) / 60);
   128→  const s = seconds % 60;
   129→  if (h > 0) return `${h}:${String(m).padStart(2, '0')}:${String(s).padStart(2, '0')}`;
   130→  return `${m}:${String(s).padStart(2, '0')}`;
   131→}
   132→
   133→// ─── Parse options from JSON ────────────────────────────────
   134→function parseOptions(optionsStr: string | null | undefined): string[] {
   135→  if (!optionsStr) return [];
   136→  try {
   137→    const parsed = JSON.parse(optionsStr);
   138→    return Array.isArray(parsed) ? parsed : [];
   139→  } catch {
   140→    return [];
   141→  }
   142→}
   143→
   144→// ─── Parse correct answer from JSON ─────────────────────────
   145→function parseCorrectAnswer(answerStr: string | null | undefined): string | string[] {
   146→  if (!answerStr) return '';
   147→  try {
   148→    const parsed = JSON.parse(answerStr);
   149→    if (Array.isArray(parsed)) return parsed;
   150→    return String(parsed);
   151→  } catch {
   152→    return answerStr;
   153→  }
   154→}

   156→// ─── Difficulty badge color ─────────────────────────────────
   157→function getDifficultyStyle(difficulty?: string) {
   158→  switch (difficulty) {
   159→    case 'easy':
   160→      return 'bg-emerald-100 text-emerald-700 dark:bg-emerald-900/30 dark:text-emerald-400';
   161→    case 'medium':
   162→      return 'bg-amber-100 text-amber-700 dark:bg-amber-900/30 dark:text-amber-400';
   163→    case 'hard':
   164→      return 'bg-red-100 text-red-700 dark:bg-red-900/30 dark:text-red-400';
   165→    default:
   166→      return 'bg-slate-100 text-slate-700 dark:bg-slate-800/30 dark:text-slate-400';
   167→  }
   168→}

   178→
   179→  const { data: assessment, isLoading: assessmentLoading, error: assessmentError } = useAssessment(assessmentId);
   180→  const submitQuiz = useSubmitQuiz();
   181→
   182→  const [phase, setPhase] = useState<PlayerPhase>('start');
   183→  const [currentQuestionIdx, setCurrentQuestionIdx] = useState(0);
   184→  const [answers, setAnswers] = useState<Record<string, string>>({});
   185→  const [markedForReview, setMarkedForReview] = useState<Set<string>>(new Set());
   186→  const [timeRemaining, setTimeRemaining] = useState<number>(0);
   187→  const [startTime, setStartTime] = useState<number>(0);
   188→  const [direction, setDirection] = useState(1);
   189→  const [showSubmitDialog, setShowSubmitDialog] = useState(false);
   190→  const [results, setResults] = useState<QuizResults | null>(null);
   191→  const [isSubmitting, setIsSubmitting] = useState(false);
   192→  const timerRef = useRef<ReturnType<typeof setInterval> | null>(null);
   193→
   194→  const typedAssessment = assessment as AssessmentData | null;
   195→  const questions = typedAssessment?.questions || [];
   196→  const currentQuestion = questions[currentQuestionIdx];
   197→
   198→  // ─── Timer logic ────────────────────────────────────────
   199→  useEffect(() => {
   200→    if (phase !== 'taking' || !typedAssessment?.timeLimit) return;
   201→
   202→    timerRef.current = setInterval(() => {
   203→      setTimeRemaining((prev) => {
   204→        if (prev <= 1) {
   205→          if (timerRef.current) clearInterval(timerRef.current);
   206→          // Auto-submit when timer expires
   207→          handleAutoSubmit();
   208→          return 0;
   209→        }
   210→        return prev - 1;
   211→      });
   212→    }, 1000);
   213→
   214→    return () => {
   215→      if (timerRef.current) clearInterval(timerRef.current);
   216→    };
   217→  }, [phase]);
   218→
   219→  const handleAutoSubmit = useCallback(() => {
   220→    toast.warning('Time is up! Your quiz has been auto-submitted.');
   221→    handleSubmit(true);
   222→  }, [answers, userId, tenantId, assessmentId]);
   223→
   224→  // ─── Start quiz ─────────────────────────────────────────
   225→  const handleStartQuiz = () => {
   226→    if (!typedAssessment) return;
   227→    setPhase('taking');
   228→    setStartTime(Date.now());
   229→    setCurrentQuestionIdx(0);
   230→    setAnswers({});
   231→    setMarkedForReview(new Set());
   232→
   233→    if (typedAssessment.timeLimit) {
   234→      setTimeRemaining(typedAssessment.timeLimit * 60);
   235→    }
   236→  };
   237→
   238→  // ─── Answer handling ────────────────────────────────────
   239→  const setAnswer = (questionId: string, answer: string) => {
   240→    setAnswers((prev) => ({ ...prev, [questionId]: answer }));
   241→  };
   242→
   243→  const toggleReviewMark = (questionId: string) => {
   244→    setMarkedForReview((prev) => {
   245→      const next = new Set(prev);
   246→      if (next.has(questionId)) next.delete(questionId);
   247→      else next.add(questionId);
   248→      return next;
   249→    });
   250→  };
   251→
   252→  // ─── Navigation ─────────────────────────────────────────
   253→  const goToQuestion = (idx: number) => {
   254→    setDirection(idx > currentQuestionIdx ? 1 : -1);
   255→    setCurrentQuestionIdx(idx);
   256→  };
   257→
   258→  const goNext = () => {
   259→    if (currentQuestionIdx < questions.length - 1) {
   260→      setDirection(1);
   261→      setCurrentQuestionIdx((i) => i + 1);
   262→    }
   263→  };
   264→
   265→  const goPrev = () => {
   266→    if (currentQuestionIdx > 0) {
   267→      setDirection(-1);
   268→      setCurrentQuestionIdx((i) => i - 1);
   269→    }
   270→  };
   271→
   272→  // ─── Submit quiz ────────────────────────────────────────
   273→  const handleSubmit = async (isAutoSubmit = false) => {
   274→    if (!isAutoSubmit && !showSubmitDialog) {
   275→      setShowSubmitDialog(true);
   276→      return;
   277→    }
   278→
   279→    setIsSubmitting(true);
   280→    if (timerRef.current) clearInterval(timerRef.current);
   281→
   282→    const timeTaken = Math.round((Date.now() - startTime) / 1000);
   283→
   284→    try {
   285→      const result = await submitQuiz.mutateAsync({
   286→        assessmentId,
   287→        userId,
   288→        tenantId,
   289→        answers,
   290→        timeTaken,
   291→      });
   292→      setResults(result as QuizResults);
   293→      setPhase('results');
   294→      setShowSubmitDialog(false);
   295→    } catch (err: any) {
   296→      toast.error(err?.message || 'Failed to submit quiz');
   297→    } finally {
   298→      setIsSubmitting(false);
   299→    }
   300→  };
   301→
   302→  // ─── Retake quiz ────────────────────────────────────────
   303→  const handleRetake = () => {
   304→    setPhase('start');
   305→    setCurrentQuestionIdx(0);
   306→    setAnswers({});
   307→    setMarkedForReview(new Set());
   308→    setResults(null);
   309→  };
   310→
   311→  // ─── Computed values ────────────────────────────────────
   312→  const answeredCount = Object.keys(answers).filter((k) => answers[k]?.trim()).length;
   313→  const unansweredCount = questions.length - answeredCount;
   314→  const reviewCount = markedForReview.size;
   315→  const totalPoints = questions.reduce((sum, q) => sum + q.points, 0);
   316→  const previousAttempts = previousSubmissions?.length || 0;
   317→  const attemptsRemaining = (typedAssessment?.maxAttempts || 3) - previousAttempts;
   318→
   319→  // Timer color
   320→  const timerUrgent = typedAssessment?.timeLimit ? timeRemaining <= 300 && timeRemaining > 0 : false;
   321→  const timerCritical = typedAssessment?.timeLimit ? timeRemaining <= 60 && timeRemaining > 0 : false;
   322→
   323→  // ─── Loading state ──────────────────────────────────────
   324→  if (assessmentLoading) {
   325→    return (
   326→      <div className="p-4 md:p-6 space-y-6">
   327→        <div className="rounded-xl overflow-hidden">
   328→          <div className="h-32 md:h-40 bg-gradient-to-r from-violet-600 to-purple-800 animate-pulse" />
   329→          <div className="p-6 space-y-4">
   330→            <div className="h-5 bg-muted rounded animate-pulse w-1/3" />
   331→            <div className="h-3 bg-muted rounded animate-pulse w-2/3" />
   332→            <div className="h-2.5 bg-muted rounded animate-pulse w-full" />
   333→            <div className="flex gap-3">
   334→              <div className="h-10 w-40 bg-muted rounded animate-pulse" />
   335→              <div className="h-10 w-32 bg-muted rounded animate-pulse" />
   336→            </div>
   337→          </div>
   338→        </div>
   339→        <div className="grid grid-cols-3 gap-4">
   340→          {[1, 2, 3].map((i) => (
   341→            <div key={i} className="rounded-xl p-4 bg-muted/30 animate-pulse space-y-3">
   342→              <div className="h-4 bg-muted rounded w-1/2" />
   343→              <div className="h-3 bg-muted rounded w-3/4" />
   344→            </div>
   345→          ))}
   346→        </div>
   347→      </div>
   348→    );
   349→  }
   350→
   351→  if (assessmentError || !typedAssessment) {
   352→    return (
   353→      <div className="p-6 text-center">
   354→        <Card className={cn(glassCard, 'max-w-md mx-auto')}>
   355→          <CardContent className="p-8">
   356→            <AlertTriangle className="h-12 w-12 text-orange-500 mx-auto mb-4" />
   357→            <h3 className="text-lg font-semibold mb-2">Failed to Load Assessment</h3>
   358→            <p className="text-sm text-muted-foreground mb-4">There was an error loading this assessment. Please try again later.</p>
   359→            <Button variant="outline" onClick={onClose}>
   360→              Go Back
   361→            </Button>
   362→          </CardContent>
   363→        </Card>
   364→      </div>
   365→    );
   366→  }
   367→
   368→  // Check if max attempts reached
   369→  if (attemptsRemaining <= 0 && phase === 'start') {
   370→    return (
   371→      <div className="p-6 text-center">
   372→        <Card className={cn(glassCard, 'max-w-md mx-auto')}>
   373→          <CardContent className="p-8">
   374→            <XCircle className="h-12 w-12 text-red-500 mx-auto mb-4" />
   375→            <h3 className="text-lg font-semibold mb-2">No Attempts Remaining</h3>
   376→            <p className="text-sm text-muted-foreground mb-4">
   377→              You&apos;ve used all {typedAssessment.maxAttempts} allowed attempt{typedAssessment.maxAttempts > 1 ? 's' : ''} for this assessment.
   378→            </p>
   379→            <Button variant="outline" onClick={onClose}>
   380→              Back to Course
   381→            </Button>
   382→          </CardContent>
   383→        </Card>
   384→      </div>
   385→    );
   386→  }
   387→
   388→  // ═══════════════════════════════════════════════════════════
   389→  // PHASE: START
   390→  // ═══════════════════════════════════════════════════════════
   391→  if (phase === 'start') {
   392→    return (
   393→      <motion.div variants={fadeVariants} initial="hidden" animate="visible" exit="exit" className="p-4 md:p-6">
   394→        {/* Header Banner */}
   395→        <div className="relative rounded-xl overflow-hidden mb-6">
   396→          <div className="bg-gradient-to-r from-violet-600 to-purple-800 p-6 md:p-8">
   397→            <div className="flex items-start justify-between">
   398→              <div className="flex-1">
   399→                <Badge className="bg-white/20 text-white hover:bg-white/30 mb-3">
   400→                  <FileCheck className="h-3 w-3 mr-1" />
   401→                  {typedAssessment.type === 'quiz' ? 'Quiz' : typedAssessment.type.charAt(0).toUpperCase() + typedAssessment.type.slice(1)}
   402→                </Badge>
   403→                <h2 className="text-2xl md:text-3xl font-bold text-white mb-2">{typedAssessment.title}</h2>
   404→                {typedAssessment.description && (
   405→                  <p className="text-white/80 text-sm md:text-base max-w-2xl">{typedAssessment.description}</p>
   406→                )}
   407→              </div>
   408→            </div>
   409→          </div>
   410→        </div>
   411→
   412→        {/* Stats Cards */}
   413→        <div className="grid grid-cols-2 md:grid-cols-4 gap-3 md:gap-4 mb-6">
   414→          <Card className={cn(glassCard)}>
   415→            <CardContent className="p-4 text-center">
   416→              <BookOpen className="h-5 w-5 text-violet-600 mx-auto mb-2" />
   417→              <div className="text-2xl font-bold">{questions.length}</div>
   418→              <div className="text-xs text-muted-foreground">Questions</div>
   419→            </CardContent>
   420→          </Card>
   421→          <Card className={cn(glassCard)}>
   422→            <CardContent className="p-4 text-center">
   423→              <Target className="h-5 w-5 text-emerald-600 mx-auto mb-2" />
   424→              <div className="text-2xl font-bold">{totalPoints}</div>
   425→              <div className="text-xs text-muted-foreground">Total Points</div>
   426→            </CardContent>
   427→          </Card>
   428→          <Card className={cn(glassCard)}>
   429→            <CardContent className="p-4 text-center">
   430→              <Timer className="h-5 w-5 text-amber-600 mx-auto mb-2" />
   431→              <div className="text-2xl font-bold">
   432→                {typedAssessment.timeLimit ? `${typedAssessment.timeLimit}m` : '∞'}
   433→              </div>
   434→              <div className="text-xs text-muted-foreground">Time Limit</div>
   435→            </CardContent>
   436→          </Card>
   437→          <Card className={cn(glassCard)}>
   438→            <CardContent className="p-4 text-center">
   439→              <Award className="h-5 w-5 text-orange-600 mx-auto mb-2" />
   440→              <div className="text-2xl font-bold">{typedAssessment.passingScore}%</div>
   441→              <div className="text-xs text-muted-foreground">Pass Score</div>

   448→          <CardContent className="p-4 md:p-6 space-y-4">
   449→            <div className="flex items-start gap-3 p-3 rounded-lg bg-amber-50 dark:bg-amber-900/20 border border-amber-200 dark:border-amber-800/40">
   450→              <AlertTriangle className="h-5 w-5 text-amber-600 shrink-0 mt-0.5" />

   464→            {previousAttempts > 0 && (

   467→                <div>

   469→                  <p className="text-xs text-blue-700 dark:text-blue-400 mt-1">

   475→
   476→            <div className="flex flex-wrap gap-2">

   498→            className="gap-2 bg-gradient-to-r from-violet-600 to-purple-700 hover:from-violet-700 hover:to-purple-800 text-white"
   499→          >
   500→            <Play className="h-4 w-4" />
   501→            Start Quiz

   514→  if (phase === 'taking') {
   515→    return (
   516→      <div className="p-4 md:p-6">
   517→        {/* Top Bar: Timer + Progress */}
   518→        <div className="flex flex-col sm:flex-row items-start sm:items-center justify-between gap-3 mb-4">
   519→          <div className="flex items-center gap-3">

   521→          </div>
   522→          <div className="flex items-center gap-3">

  1179→function AnswerInput({
  1180→  question,
  1181→  answer,
  1182→  onAnswerChange,
  1183→}: {
  1184→  question: QuestionData;
  1185→  answer: string;
  1186→  onAnswerChange: (val: string) => void;
  1187→}) {
  1188→  const options = parseOptions(question.options);
  1189→
  1190→  switch (question.type) {
  1191→    case 'multiple_choice':
  1192→      return (
  1193→        <RadioGroup value={answer} onValueChange={onAnswerChange} className="space-y-2">
  1194→          {options.map((opt, idx) => {
  1195→            const optionLabel = String.fromCharCode(65 + idx);

  1198→                key={idx}
  1199→                htmlFor={`q-${question.id}-${idx}`}
  1200→                className={cn(
  1201→                  'flex items-center gap-3 p-3 rounded-lg border cursor-pointer transition-all',
  1202→                  answer === opt

  1212→              </Label>

  1215→        </RadioGroup>

  1217→
  1218→    case 'true_false':

  1220→        <RadioGroup value={answer} onValueChange={onAnswerChange} className="flex gap-4">

  1222→            htmlFor={`q-${question.id}-true`}

  1224→              'flex-1 flex items-center justify-center gap-2 p-4 rounded-lg border cursor-pointer transition-all',
  1225→              answer === 'True'

  1228→            )}
  1229→          >
  1230→            <RadioGroupItem value="True" id={`q-${question.id}-true`} />
  1231→            <CheckCircle2 className="h-4 w-4 text-emerald-600" />
  1232→            <span className="text-sm font-medium">True</span>
  1233→          </Label>
  1234→          <Label
  1235→            htmlFor={`q-${question.id}-false`}
  1236→            className={cn(
  1237→              'flex-1 flex items-center justify-center gap-2 p-4 rounded-lg border cursor-pointer transition-all',
  1238→              answer === 'False'

  1242→          >

  1252→        <div className="space-y-2">
  1253→          <Input
  1254→            value={answer}
  1255→            onChange={(e) => onAnswerChange(e.target.value)}
  1256→            placeholder="Type your answer here..."

  1304→    default:

  1310→          className="text-sm"
  1311→        />
