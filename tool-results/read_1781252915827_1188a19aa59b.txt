    58→   557→   557→function AnimatedCounter({ value, prefix = '', suffix = '', duration = 1500 }: {
    59→   558→   558→  value: number;
    60→   559→   559→  prefix?: string;
    61→   560→   560→  suffix?: string;
    62→   561→   561→  duration?: number;
    63→   562→   562→}) {
    64→   563→   563→  const [displayed, setDisplayed] = useState(0);
    65→   564→   564→  const startTime = useRef<number | null>(null);
    66→   565→   565→  const frameRef = useRef<number>(0);
    67→   566→   566→  const animateRef = useRef<((timestamp: number) => void) | null>(null);
    68→   567→   567→
    69→   568→   568→  useEffect(() => {
    70→   569→   569→    animateRef.current = (timestamp: number) => {
    71→   570→   570→      if (!startTime.current) startTime.current = timestamp;
    72→   571→   571→      const progress = Math.min((timestamp - startTime.current) / duration, 1);
    73→   572→   572→      const eased = 1 - Math.pow(1 - progress, 3);
    74→   573→   573→      setDisplayed(Math.floor(eased * value));
    75→   574→   574→      if (progress < 1) {
    76→   575→   575→        frameRef.current = requestAnimationFrame((ts) => animateRef.current?.(ts));
    77→   576→   576→      }
    78→   577→   577→    };
    79→   578→   578→    startTime.current = null;
    80→   579→   579→    frameRef.current = requestAnimationFrame((ts) => animateRef.current?.(ts));
    81→   580→   580→    return () => cancelAnimationFrame(frameRef.current);
    82→   581→   581→  }, [value, duration]);
    83→   582→   582→
    84→   583→   583→  return (
    85→   584→   584→    <span>
    86→   585→   585→      {prefix}{displayed.toLocaleString()}{suffix}
    87→   586→   586→    </span>
    88→   587→   587→  );
    89→   588→   588→}
