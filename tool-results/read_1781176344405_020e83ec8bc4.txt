    94→function formatDuration(seconds: number): string {
    95→  const mins = Math.floor(seconds / 60);
    96→  if (mins < 60) return `${mins}m`;
    97→  const hrs = Math.floor(mins / 60);
    98→  const rem = mins % 60;
    99→  return rem > 0 ? `${hrs}h ${rem}m` : `${hrs}h`;
   100→}

   401→function CourseBuilderTab() {
   402→  const [selectedCourseId, setSelectedCourseId] = useState<string>(demoCourses[0].id);
   403→  const [selectedLessonId, setSelectedLessonId] = useState<string | null>(null);
   404→  const [expandedModules, setExpandedModules] = useState<Set<string>>(
   405→    new Set(demoCourses[0].modules?.map((m) => m.id) || [])
   406→  );
   407→
   408→  const selectedCourse = useMemo(
   409→    () => demoCourses.find((c) => c.id === selectedCourseId) || demoCourses[0],
   410→    [selectedCourseId]
   411→  );
   412→
   413→  const selectedLesson = useMemo(() => {
   414→    if (!selectedLessonId) return null;
   415→    for (const mod of selectedCourse.modules || []) {
   416→      const found = mod.lessons?.find((l) => l.id === selectedLessonId);
   417→      if (found) return found;
   418→    }
   419→    return null;
   420→  }, [selectedLessonId, selectedCourse]);
   421→
   422→  const toggleModule = (moduleId: string) => {
   423→    setExpandedModules((prev) => {
   424→      const next = new Set(prev);
   425→      if (next.has(moduleId)) next.delete(moduleId);
   426→      else next.add(moduleId);
   427→      return next;
   428→    });
   429→  };
   430→
   431→  const totalLessons = useMemo(
   432→    () =>
   433→      (selectedCourse.modules || []).reduce(
   434→        (acc, m) => acc + (m.lessons?.length || 0),
   435→        0
   436→      ),
   437→    [selectedCourse]
   438→  );
   439→
   440→  const totalDurationSec = useMemo(
   441→    () =>
   442→      (selectedCourse.modules || []).reduce(
   443→        (acc, m) =>
   444→          acc +
   445→          (m.lessons || []).reduce(
   446→            (a, l) => a + (l.videoDuration || 0),
   447→            0
   448→          ),
   449→        0
   450→      ),
   451→    [selectedCourse]
   452→  );
   453→
   454→  return (
   455→    <div className="space-y-4">
   456→      {/* Course selector */}
   457→      <div className="flex items-center gap-3">
   458→        <Label className="text-sm font-medium shrink-0">Course:</Label>
   459→        <Select value={selectedCourseId} onValueChange={(v) => { setSelectedCourseId(v); setSelectedLessonId(null); }}>
   460→          <SelectTrigger className="w-full sm:w-[320px]">
   461→            <SelectValue />
   462→          </SelectTrigger>
   463→          <SelectContent>
   464→            {demoCourses.map((c) => (
   465→              <SelectItem key={c.id} value={c.id}>
   466→                {c.title}
   467→              </SelectItem>
   468→            ))}
   469→          </SelectContent>
   470→        </Select>
   471→      </div>
   472→
   473→      {/* Two-panel layout */}
   474→      <div className="grid grid-cols-1 lg:grid-cols-[340px_1fr] gap-4 min-h-[600px]">
   475→        {/* Left panel: Module & Lesson Tree */}
   476→        <Card className="overflow-hidden">
   477→          <div className="p-4 border-b bg-muted/30">
   478→            <div className="flex items-center justify-between">
   479→              <h3 className="text-sm font-semibold text-foreground">Curriculum</h3>
   480→              <div className="flex items-center gap-1">
   481→                <Button variant="ghost" size="sm" className="h-7 text-xs gap-1 text-emerald-600 dark:text-emerald-400 hover:text-emerald-700 dark:hover:text-emerald-300">
   482→                  <Plus className="h-3 w-3" /> Module
   483→                </Button>
   484→              </div>
   485→            </div>
   486→            <div className="flex items-center gap-3 mt-2 text-xs text-muted-foreground">
   487→              <span>{(selectedCourse.modules || []).length} modules</span>
   488→              <span>{totalLessons} lessons</span>
   489→              <span>{formatDuration(totalDurationSec)}</span>
   490→            </div>
   491→          </div>
   492→
   493→          <ScrollArea className="h-[540px]">
   494→            <div className="p-2">
   495→              {(selectedCourse.modules || []).length === 0 ? (
   496→                <div className="flex flex-col items-center justify-center py-12 text-muted-foreground">
   497→                  <Layers className="h-8 w-8 mb-2 opacity-40" />
   498→                  <p className="text-sm">No modules yet</p>
   499→                  <p className="text-xs">Add a module to start building</p>
   500→                </div>
   501→              ) : (
   502→                (selectedCourse.modules || [])
   503→                  .sort((a, b) => a.orderIndex - b.orderIndex)
   504→                  .map((mod) => (
   505→                    <ModuleTreeItem
   506→                      key={mod.id}
   507→                      module={mod}
   508→                      isExpanded={expandedModules.has(mod.id)}
   509→                      onToggle={() => toggleModule(mod.id)}
   510→                      selectedLessonId={selectedLessonId}
   511→                      onSelectLesson={setSelectedLessonId}
   512→                    />
   513→                  ))
   514→              )}
   515→            </div>
   516→          </ScrollArea>
   517→        </Card>
   518→
   519→        {/* Right panel: Content Editor */}
   520→        <Card className="overflow-hidden">
   521→          {selectedLesson ? (
   522→            <LessonEditor lesson={selectedLesson} />
   523→          ) : (
   524→            <div className="flex flex-col items-center justify-center h-full min-h-[400px] text-muted-foreground">
   525→              <BookOpen className="h-12 w-12 mb-4 opacity-30" />
   526→              <p className="text-lg font-medium">Select a lesson</p>
   527→              <p className="text-sm">Choose a lesson from the curriculum tree to edit its content</p>
   528→            </div>
   529→          )}
   530→        </Card>
   531→      </div>
   532→    </div>
   533→  );
   534→}

   536→function ModuleTreeItem({
   537→  module,
   538→  isExpanded,
   539→  onToggle,
   540→  selectedLessonId,
   541→  onSelectLesson,
   542→}: {
