   180→   180→function TypingText({ text, speed = 35, delay = 0 }: { text: string; speed?: number; delay?: number }) {
   181→   181→  const [displayed, setDisplayed] = useState('');
   182→   182→  const [started, setStarted] = useState(false);
   183→   183→
   184→   184→  useEffect(() => {
   185→   185→    const timer = setTimeout(() => setStarted(true), delay);
   186→   186→    return () => clearTimeout(timer);
   187→   187→  }, [delay]);
   188→   188→
   189→   189→  useEffect(() => {
   190→   190→    if (!started) return;
   191→   191→    if (displayed.length < text.length) {
   192→   192→      const timer = setTimeout(() => {
   193→   193→        setDisplayed(text.slice(0, displayed.length + 1));
   194→   194→      }, speed);
   195→   195→      return () => clearTimeout(timer);
   196→   196→    }
   197→   197→  }, [displayed, text, speed, started]);
   198→   198→
   199→   199→  return (
   200→   200→    <span>
   201→   201→      {displayed}
   202→   202→      {displayed.length < text.length && started && (
   203→   203→        <span className="inline-block w-0.5 h-5 bg-emerald-500 ml-0.5 animate-pulse align-middle" />
   204→   204→      )}
   205→   205→    </span>
   206→   206→  );
   207→   207→}
