   129→// Derive session status from event
   130→function getSessionStatus(event: CalendarEvent): SessionStatus {
   131→  if ((event as CalendarEvent & { status?: SessionStatus }).status === 'cancelled') return 'cancelled';
   132→  const now = new Date();
   133→  const start = parseISO(event.startDate);
   134→  const end = parseISO(event.endDate);
   135→  if (isBefore(end, now)) return 'completed';
   136→  if (isAfter(start, now)) return 'upcoming';
   137→  return 'live';
   138→}

   164→// Countdown Timer Component
   165→function CountdownTimer({ targetDate, color }: { targetDate: string; color: string }) {
   166→  const [timeLeft, setTimeLeft] = useState('');
   167→  const [isUrgent, setIsUrgent] = useState(false);
   168→  const intervalRef = useRef<ReturnType<typeof setInterval> | null>(null);
   169→
   170→  useEffect(() => {
   171→    const updateTimer = () => {
   172→      const now = new Date();
   173→      const target = parseISO(targetDate);
   174→      const diffSeconds = differenceInSeconds(target, now);
   175→
   176→      if (diffSeconds <= 0) {
   177→        setTimeLeft('Starting now');
   178→        setIsUrgent(true);
   179→        if (intervalRef.current) clearInterval(intervalRef.current);
   180→        return;
   181→      }
   182→
   183→      setIsUrgent(diffSeconds < 900); // 15 minutes
   184→
   185→      const days = Math.floor(diffSeconds / 86400);
   186→      const hours = Math.floor((diffSeconds % 86400) / 3600);
   187→      const minutes = Math.floor((diffSeconds % 3600) / 60);
   188→      const seconds = diffSeconds % 60;
   189→
   190→      if (days > 0) {
   191→        setTimeLeft(`${days}d ${hours}h ${minutes}m`);
   192→      } else if (hours > 0) {
   193→        setTimeLeft(`${hours}h ${minutes}m ${seconds}s`);
   194→      } else if (minutes > 0) {
   195→        setTimeLeft(`${minutes}m ${seconds}s`);
   196→      } else {
   197→        setTimeLeft(`${seconds}s`);
   198→      }
   199→    };
   200→
   201→    updateTimer();
   202→    intervalRef.current = setInterval(updateTimer, 1000);
   203→
   204→    return () => {
   205→      if (intervalRef.current) clearInterval(intervalRef.current);
   206→    };
   207→  }, [targetDate]);
   208→
   209→  return (
   210→    <motion.div
   211→      className="flex items-center gap-1.5 text-xs font-mono"
   212→      animate={isUrgent ? { scale: [1, 1.02, 1] } : {}}
   213→      transition={isUrgent ? { duration: 0.8, repeat: Infinity } : {}}
   214→    >
   215→      <Timer className="h-3.5 w-3.5" style={{ color }} />
   216→      <span className={isUrgent ? 'text-red-500 font-bold' : 'text-muted-foreground'}>
   217→        {timeLeft}
   218→      </span>
   219→    </motion.div>
   220→  );
   221→}

   223→// Attendee Avatars Component
   224→function AttendeeAvatars({ attendees, maxAttendees }: { attendees: number; maxAttendees: number }) {
   225→  const displayCount = Math.min(attendees, 5);
   226→  const overflow = attendees - displayCount;
   227→  const percent = Math.round((attendees / maxAttendees) * 100);
   228→
   229→  return (
   230→    <div className="flex items-center gap-2">
   231→      <div className="flex -space-x-2">
   232→        {Array.from({ length: displayCount }).map((_, i) => (
   233→          <Avatar key={i} className="h-6 w-6 border-2 border-background">
   234→            <AvatarFallback className="text-[8px] font-medium" style={{ backgroundColor: presetColors[i % presetColors.length] + '30', color: presetColors[i % presetColors.length] }}>
   235→              {attendeeNames[i]}
   236→            </AvatarFallback>
   237→          </Avatar>
   238→        ))}
   239→        {overflow > 0 && (
   240→          <Avatar className="h-6 w-6 border-2 border-background">
   241→            <AvatarFallback className="text-[8px] font-medium bg-muted text-muted-foreground">
   242→              +{overflow}
   243→            </AvatarFallback>
   244→          </Avatar>
   245→        )}
   246→      </div>
   247→      <div className="flex items-center gap-2 min-w-[100px]">
   248→        <Progress value={percent} className="h-1.5 flex-1" />
   249→        <span className="text-[10px] text-muted-foreground whitespace-nowrap font-medium">
   250→          {attendees}/{maxAttendees}
   251→        </span>
   252→      </div>
   253→    </div>
   254→  );
   255→}
