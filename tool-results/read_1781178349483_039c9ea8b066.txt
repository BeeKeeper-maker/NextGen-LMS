    60→// ─── Format duration ───────────────────────────────────────
    61→function formatDuration(seconds?: number): string {
    62→  if (!seconds) return '';
    63→  const mins = Math.floor(seconds / 60);
    64→  if (mins < 60) return `${mins} min`;
    65→  const hours = Math.floor(mins / 60);
    66→  const remainMins = mins % 60;
    67→  return `${hours}h ${remainMins}m`;
    68→}

   326→// ─── Main Component ────────────────────────────────────────
   327→export function LearnerCourse() {
   328→  const { currentUser } = useAppStore();
   329→  const course = demoCourses[0]; // React & Next.js Masterclass
   330→  const enrollment = demoEnrollments[0]; // Active enrollment for this course
   331→  const modules = course.modules || [];
   332→  const [activeTab, setActiveTab] = useState('curriculum');
   333→  const [activeLesson, setActiveLesson] = useState<Lesson | null>(null);
   334→  const [activeLessonModuleName, setActiveLessonModuleName] = useState('');
   335→  const [lessonProgressState, setLessonProgressState] = useState(lessonProgressMap);
   336→
   337→  // Calculate total lessons and completed
   338→  const allLessons = modules.flatMap(m => m.lessons || []);
   339→  const completedCount = allLessons.filter(l => lessonProgressState[l.id] === 'completed').length;
   340→  const totalLessons = allLessons.length;
   341→
   342→  // Handle clicking a lesson
   343→  const handleLessonClick = (lesson: Lesson) => {
   344→    setActiveLesson(lesson);
   345→    // Find the module name for this lesson
   346→    const parentModule = modules.find(m => m.lessons?.some(l => l.id === lesson.id));
   347→    setActiveLessonModuleName(parentModule?.title || '');
   348→  };
   349→
   350→  // Handle marking a lesson as complete
   351→  const handleMarkComplete = (lessonId: string) => {
   352→    setLessonProgressState(prev => ({ ...prev, [lessonId]: 'completed' }));
   353→  };
   354→
   355→  // Get next lessons for the sidebar
   356→  const getNextLessons = (): { id: string; title: string; moduleName: string; duration: number; contentType: Lesson['contentType'] }[] => {
   357→    if (!activeLesson) return [];
   358→    const currentIdx = allLessons.findIndex(l => l.id === activeLesson.id);
   359→    const next: { id: string; title: string; moduleName: string; duration: number; contentType: Lesson['contentType'] }[] = [];
   360→    for (let i = currentIdx + 1; i < Math.min(currentIdx + 4, allLessons.length); i++) {
   361→      const l = allLessons[i];
   362→      const mod = modules.find(m => m.lessons?.some(ml => ml.id === l.id));
   363→      next.push({ id: l.id, title: l.title, moduleName: mod?.title || '', duration: l.videoDuration || 0, contentType: l.contentType });
   364→    }
   365→    return next;
   366→  };
   367→
   368→  // If a lesson is active, show the enhanced Video Player
   369→  if (activeLesson) {
   370→    return (
   371→      <div className="p-4 md:p-6">
   372→        <VideoPlayer
   373→          title={activeLesson.title}
   374→          lesson={activeLesson}
   375→          moduleName={activeLessonModuleName}
   376→          nextLessons={getNextLessons()}
   377→          onMarkComplete={handleMarkComplete}
   378→          onNextLesson={(lessonId) => {
   379→            const found = allLessons.find(l => l.id === lessonId);
   380→            if (found) handleLessonClick(found);
   381→          }}
   382→          onBack={() => setActiveLesson(null)}
   383→          isCompleted={lessonProgressState[activeLesson.id] === 'completed'}
   384→          initialPosition={resumePositions[activeLesson.id] || 0}
   385→          chapters={demoChapters}
   386→          onProgress={(progress, currentTime) => {
   387→            // Store resume position
   388→            resumePositions[activeLesson.id] = currentTime;
   389→          }}
   390→        />
   391→      </div>
   392→    );
   393→  }
   394→
   395→  return (
   396→    <motion.div
   397→      className="p-4 md:p-6 space-y-6"
   398→      variants={containerVariants}
   399→      initial="hidden"
   400→      animate="visible"
   401→    >
   402→      {/* ─── Course Header ────────────────────────────────── */}
   403→      <motion.div variants={itemVariants}>
   404→        <Card className="overflow-hidden">
   405→          {/* Gradient Banner */}
   406→          <div className={cn(
   407→            'h-32 md:h-40 bg-gradient-to-r flex items-end p-6',
   408→            getLevelAccent(course.level)
   409→          )}>
   410→            <div className="flex-1">
   411→              <div className="flex items-center gap-2 mb-2">
   412→                <Badge className="bg-white/20 text-white border-0 backdrop-blur-sm text-xs">
   413→                  {course.category}
   414→                </Badge>
   415→                <Badge className="bg-white/20 text-white border-0 backdrop-blur-sm text-xs capitalize">
   416→                  {course.level}
   417→                </Badge>
   418→              </div>
   419→              <h1 className="text-2xl md:text-3xl font-bold text-white leading-tight">{course.title}</h1>
   420→            </div>
   421→          </div>
   422→
   423→          <CardContent className="p-6">
   424→            {/* Course Meta */}
   425→            <div className="flex flex-wrap items-center gap-4 text-sm text-muted-foreground mb-4">
   426→              <div className="flex items-center gap-1.5">
   427→                <Avatar className="h-6 w-6">
   428→                  <AvatarFallback className="text-[10px] bg-slate-200 dark:bg-slate-700">SM</AvatarFallback>
   429→                </Avatar>
   430→                <span>Sarah Mitchell</span>
   431→              </div>
   432→              <div className="flex items-center gap-1">
   433→                <Star className="h-4 w-4 fill-amber-400 text-amber-400" />
   434→                <span className="font-medium text-foreground">{course.avgRating}</span>
   435→                <span>({course.totalRatings} ratings)</span>
   436→              </div>
   437→              <div className="flex items-center gap-1">
   438→                <Users className="h-4 w-4" />
   439→                <span>{course.enrollmentCount.toLocaleString()} enrolled</span>
   440→              </div>
   441→              {course.durationHours && (
   442→                <div className="flex items-center gap-1">
   443→                  <Clock className="h-4 w-4" />
