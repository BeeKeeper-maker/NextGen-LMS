   238→// Derive session status from event
   239→function getSessionStatus(event: CalendarEvent): SessionStatus {
   240→  if ((event as CalendarEvent & { status?: SessionStatus }).status === 'cancelled') return 'cancelled';
   241→  const now = new Date();
   242→  const start = parseISO(event.startDate);
   243→  const end = parseISO(event.endDate);
   244→  if (isBefore(end, now)) return 'completed';
   245→  if (isAfter(start, now)) return 'upcoming';
   246→  return 'live';
   247→}

   273→// Countdown Timer Component
   274→function CountdownTimer({ targetDate, color }: { targetDate: string; color: string }) {
   275→  const [timeLeft, setTimeLeft] = useState('');
   276→  const [isUrgent, setIsUrgent] = useState(false);
   277→  const intervalRef = useRef<ReturnType<typeof setInterval> | null>(null);
   278→
   279→  useEffect(() => {
   280→    const updateTimer = () => {
   281→      const now = new Date();
   282→      const target = parseISO(targetDate);
   283→      const diffSeconds = differenceInSeconds(target, now);
   284→
   285→      if (diffSeconds <= 0) {
   286→        setTimeLeft('Starting now');
   287→        setIsUrgent(true);
   288→        if (intervalRef.current) clearInterval(intervalRef.current);
   289→        return;
   290→      }
   291→
   292→      setIsUrgent(diffSeconds < 900); // 15 minutes
   293→
   294→      const days = Math.floor(diffSeconds / 86400);
   295→      const hours = Math.floor((diffSeconds % 86400) / 3600);
   296→      const minutes = Math.floor((diffSeconds % 3600) / 60);
   297→      const seconds = diffSeconds % 60;
   298→
   299→      if (days > 0) {
   300→        setTimeLeft(`${days}d ${hours}h ${minutes}m`);
   301→      } else if (hours > 0) {
   302→        setTimeLeft(`${hours}h ${minutes}m ${seconds}s`);
   303→      } else if (minutes > 0) {
   304→        setTimeLeft(`${minutes}m ${seconds}s`);
   305→      } else {
   306→        setTimeLeft(`${seconds}s`);
   307→      }
   308→    };
   309→
   310→    updateTimer();
   311→    intervalRef.current = setInterval(updateTimer, 1000);
   312→
   313→    return () => {
   314→      if (intervalRef.current) clearInterval(intervalRef.current);
   315→    };
   316→  }, [targetDate]);
   317→
   318→  return (
   319→    <motion.div
   320→      className="flex items-center gap-1.5 text-xs font-mono"
   321→      animate={isUrgent ? { scale: [1, 1.02, 1] } : {}}
   322→      transition={isUrgent ? { duration: 0.8, repeat: Infinity } : {}}
   323→    >
   324→      <Timer className="h-3.5 w-3.5" style={{ color }} />
   325→      <span className={isUrgent ? 'text-red-500 font-bold' : 'text-muted-foreground'}>
   326→        {timeLeft}
   327→      </span>
   328→    </motion.div>
   329→  );
   330→}

   333→function AttendeeAvatars({ attendees, maxAttendees }: { attendees: number; maxAttendees: number }) {
   334→  const displayCount = Math.min(attendees, 5);
   335→  const overflow = attendees - displayCount;
   336→  const percent = Math.round((attendees / maxAttendees) * 100);
   337→
   338→  return (
   339→    <div className="flex items-center gap-2">
   340→      <div className="flex -space-x-2">
   341→        {Array.from({ length: displayCount }).map((_, i) => (
   342→          <Avatar key={i} className="h-6 w-6 border-2 border-background">
   343→            <AvatarFallback className="text-[8px] font-medium" style={{ backgroundColor: presetColors[i % presetColors.length] + '30', color: presetColors[i % presetColors.length] }}>
   344→              {attendeeNames[i]}
   345→            </AvatarFallback>
   346→          </Avatar>
   347→        ))}
   348→        {overflow > 0 && (
   349→          <Avatar className="h-6 w-6 border-2 border-background">
   350→            <AvatarFallback className="text-[8px] font-medium bg-muted text-muted-foreground">
   351→              +{overflow}
   352→            </AvatarFallback>
   353→          </Avatar>
   354→        )}
   355→      </div>
   356→      <div className="flex items-center gap-2 min-w-[100px]">

   358→        <span className="text-[10px] text-muted-foreground whitespace-nowrap font-medium">
   359→          {attendees}/{maxAttendees}
