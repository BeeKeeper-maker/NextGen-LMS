   122→// ============================================================
   123→// Typing Animation Component
   124→// ============================================================
   125→function TypingText({ text, speed = 35, delay = 0 }: { text: string; speed?: number; delay?: number }) {
   126→  const [displayed, setDisplayed] = useState('');
   127→  const [started, setStarted] = useState(false);
   128→
   129→  useEffect(() => {
   130→    const timer = setTimeout(() => setStarted(true), delay);
   131→    return () => clearTimeout(timer);
   132→  }, [delay]);
   133→
   134→  useEffect(() => {
   135→    if (!started) return;
   136→    if (displayed.length < text.length) {
   137→      const timer = setTimeout(() => {
   138→        setDisplayed(text.slice(0, displayed.length + 1));
   139→      }, speed);
   140→      return () => clearTimeout(timer);
   141→    }
   142→  }, [displayed, text, speed, started]);
   143→
   144→  return (
   145→    <span>
   146→      {displayed}
   147→      {displayed.length < text.length && started && (
   148→        <span className="inline-block w-0.5 h-5 bg-emerald-500 ml-0.5 animate-pulse align-middle" />
   149→      )}
   150→    </span>
   151→  );
   152→}
