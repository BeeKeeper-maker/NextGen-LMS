    71→// Relative time helper
    72→function getRelativeTime(dateStr?: string) {
    73→  if (!dateStr) return '';
    74→  const now = new Date();
    75→  const date = new Date(dateStr);
    76→  const diffMs = now.getTime() - date.getTime();
    77→  const diffMins = Math.floor(diffMs / 60000);
    78→  const diffHours = Math.floor(diffMins / 60);
    79→  const diffDays = Math.floor(diffHours / 24);
    80→
    81→  if (diffMins < 60) return `${diffMins}m ago`;
    82→  if (diffHours < 24) return `${diffHours}h ago`;
    83→  if (diffDays < 7) return `${diffDays}d ago`;
    84→  return date.toLocaleDateString();
    85→}

   220→// ─── Parallax Tilt Card ─────────────────────────────────────
   221→function TiltCard({ children, className }: { children: React.ReactNode; className?: string }) {
   222→  const cardRef = useRef<HTMLDivElement>(null);
   223→  const [rotateX, setRotateX] = useState(0);
   224→  const [rotateY, setRotateY] = useState(0);
   225→
   226→  const handleMouseMove = (e: React.MouseEvent<HTMLDivElement>) => {
   227→    if (!cardRef.current) return;
   228→    const rect = cardRef.current.getBoundingClientRect();
   229→    const x = e.clientX - rect.left;
   230→    const y = e.clientY - rect.top;
   231→    const centerX = rect.width / 2;
   232→    const centerY = rect.height / 2;
   233→    const maxTilt = 5;
   234→    const newRotateX = ((y - centerY) / centerY) * -maxTilt;
   235→    const newRotateY = ((x - centerX) / centerX) * maxTilt;
   236→    setRotateX(newRotateX);
   237→    setRotateY(newRotateY);
   238→  };
   239→
   240→  const handleMouseLeave = () => {
   241→    setRotateX(0);
   242→    setRotateY(0);
   243→  };
   244→
   245→  return (
   246→    <div
   247→      ref={cardRef}
   248→      onMouseMove={handleMouseMove}
   249→      onMouseLeave={handleMouseLeave}
   250→      className={className}
   251→      style={{
   252→        transform: `perspective(800px) rotateX(${rotateX}deg) rotateY(${rotateY}deg)`,
   253→        transition: 'transform 0.15s ease-out',
   254→      }}
   255→    >
   256→      {children}
   257→    </div>
   258→  );
   259→}
