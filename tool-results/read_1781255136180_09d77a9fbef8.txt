   117→// ─── Format duration ───────────────────────────────────────
   118→function formatDuration(seconds?: number): string {
   119→  if (!seconds) return '';
   120→  const mins = Math.floor(seconds / 60);
   121→  if (mins < 60) return `${mins} min`;
   122→  const hours = Math.floor(mins / 60);
   123→  const remainMins = mins % 60;
   124→  return `${hours}h ${remainMins}m`;
   125→}

   717→// ─── Time ago helper ──────────────────────────────────────
   718→function timeAgo(dateStr: string): string {
   719→  const seconds = Math.floor((Date.now() - new Date(dateStr).getTime()) / 1000);
   720→  if (seconds < 60) return 'just now';
   721→  const minutes = Math.floor(seconds / 60);
   722→  if (minutes < 60) return `${minutes}m ago`;
   723→  const hours = Math.floor(minutes / 60);
   724→  if (hours < 24) return `${hours}h ago`;
   725→  const days = Math.floor(hours / 24);
   726→  if (days < 30) return `${days}d ago`;
   727→  const months = Math.floor(days / 30);
   728→  return `${months}mo ago`;
   729→}
