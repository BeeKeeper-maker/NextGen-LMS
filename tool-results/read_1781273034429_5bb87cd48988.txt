   132→// ─── Format duration ───────────────────────────────────────
   133→function formatDuration(seconds?: number): string {
   134→  if (!seconds) return '';
   135→  const mins = Math.floor(seconds / 60);
   136→  if (mins < 60) return `${mins} min`;
   137→  const hours = Math.floor(mins / 60);
   138→  const remainMins = mins % 60;
   139→  return `${hours}h ${remainMins}m`;
   140→}

   175→function useLessonNotes(courseId: string | undefined) {
   176→  const storageKey = `lms-notes-${courseId || 'default'}`;
   177→
   178→  const [notes, setNotes] = useState<Note[]>(() => {
   179→    if (typeof window === 'undefined') return [];
   180→    try {
   181→      const stored = localStorage.getItem(storageKey);
   182→      return stored ? JSON.parse(stored) : [];
   183→    } catch {
   184→      return [];
   185→    }
   186→  });
   187→
   188→  const persistNotes = useCallback((updated: Note[]) => {
   189→    setNotes(updated);
   190→    if (typeof window !== 'undefined') {
   191→      try {
   192→        localStorage.setItem(storageKey, JSON.stringify(updated));
   193→      } catch {
   194→        // ignore quota errors
   195→      }
   196→    }
   197→  }, [storageKey]);
   198→
   199→  const addNote = useCallback((note: Omit<Note, 'id' | 'createdAt' | 'updatedAt'>) => {
   200→    const newNote: Note = {
   201→      ...note,
   202→      id: `note-${Date.now()}-${Math.random().toString(36).slice(2, 7)}`,
   203→      createdAt: new Date().toISOString(),
   204→      updatedAt: new Date().toISOString(),
   205→    };
   206→    persistNotes([...notes, newNote]);
   207→    return newNote;
   208→  }, [notes, persistNotes]);
   209→
   210→  const updateNote = useCallback((id: string, updates: Partial<Note>) => {
   211→    const updated = notes.map(n =>
   212→      n.id === id ? { ...n, ...updates, updatedAt: new Date().toISOString() } : n
   213→    );
   214→    persistNotes(updated);
   215→  }, [notes, persistNotes]);
   216→
   217→  const deleteNote = useCallback((id: string) => {
   218→    persistNotes(notes.filter(n => n.id !== id));
   219→  }, [notes, persistNotes]);
   220→
   221→  return { notes, addNote, updateNote, deleteNote };
   222→}

   252→// ─── Time ago helper ──────────────────────────────────────
   253→function timeAgo(dateStr: string): string {
   254→  const seconds = Math.floor((Date.now() - new Date(dateStr).getTime()) / 1000);
   255→  if (seconds < 60) return 'just now';
   256→  const minutes = Math.floor(seconds / 60);
   257→  if (minutes < 60) return `${minutes}m ago`;
   258→  const hours = Math.floor(minutes / 60);
   259→  if (hours < 24) return `${hours}h ago`;
   260→  const days = Math.floor(hours / 24);
   261→  if (days < 30) return `${days}d ago`;
   262→  const months = Math.floor(days / 30);
   263→  return `${months}mo ago`;
   264→}

   510→

   534→      lessonId: discussions[0]?.lessonId || '',
   535→      userId,

   537→    }, {

   608→            placeholder="Search discussions..."
   609→            value={searchQuery}

   667→                      {createDiscussion.isPending ? <Loader2 className="h-3 w-3 animate-spin" /> : <Send className="h-3 w-3" />} Post Question
   668→                    </Button>
   669→                  </div>
   670→                </div>
   671→              </CardContent>
   672→            </Card>
   673→          </motion.div>
   674→        )}
   675→      </AnimatePresence>
   676→
   677→      {/* Questions List */}
   678→      <div className="space-y-3">
   679→        <AnimatePresence mode="popLayout">
   680→          {filteredQuestions.map((question: any) => {

   682→            const isAnswered = qReplies.length > 0;
   683→            return (

   685→                key={question.id}
   686→                layout
   687→                initial={{ opacity: 0, y: 10 }}
   688→                animate={{ opacity: 1, y: 0 }}
   689→                exit={{ opacity: 0, y: -10 }}
   690→              >
   691→                <Card className={cn(
   692→                  'hover:shadow-md transition-all duration-200 cursor-pointer',
   693→                  glassCard,
   694→                  isAnswered && 'border-emerald-200 dark:border-emerald-800/40'
   695→                )}>
   696→                  <CardContent className="p-4">
   697→                    <div className="flex items-start gap-3">
   698→                      {/* Vote column */}
   699→                      <div className="flex flex-col items-center gap-1 pt-1">
   700→                        <button className="p-1 rounded hover:bg-muted/50 transition-colors text-muted-foreground hover:text-foreground">
   701→                          <ChevronUp className="h-4 w-4" />
   702→                        </button>
   703→                        <span className="text-sm font-semibold text-foreground">{qReplies.length}</span>
   704→                        <button className="p-1 rounded hover:bg-muted/50 transition-colors text-muted-foreground hover:text-foreground">
   705→                          <ChevronDown className="h-4 w-4" />
   706→                        </button>
   707→                      </div>
   708→

   713→                            <Badge className="bg-emerald-100 text-emerald-700 dark:bg-emerald-900/30 dark:text-emerald-400 text-[10px] gap-0.5">
   714→                              <CheckCircle2 className="h-3 w-3" /> Answered
   715→                            </Badge>
   716→                          )}
   717→                          {question.lesson?.title && (
   718→                            <Badge variant="outline" className="text-[10px] gap-0.5">
   719→                              <BookOpen className="h-2.5 w-2.5" /> {question.lesson.title}
   720→                            </Badge>
   721→                          )}
   722→                          {question.isPinned && (
   723→                            <Badge className="bg-amber-100 text-amber-700 dark:bg-amber-900/30 dark:text-amber-400 text-[10px] gap-0.5">
   724→                              <Pin className="h-2.5 w-2.5" /> Pinned
   725→                            </Badge>
   726→                          )}
   727→                        </div>
   728→                        <p className="text-sm text-muted-foreground line-clamp-3">{question.content}</p>
   729→                        <div className="flex items-center gap-3 mt-2">
   730→                          <span className="text-xs text-muted-foreground">· {timeAgo(question.createdAt)}</span>
   731→                          <span className="flex items-center gap-1 text-xs text-muted-foreground">
   732→                            <MessageCircle className="h-3 w-3" /> {qReplies.length}
   733→                          </span>
   734→                        </div>
   735→                      </div>
   736→                    </div>
   737→
   738→                    {/* Expanded Replies */}
   739→                    <AnimatePresence>
   740→                      {expandedQuestion === question.id && (
   741→                        <motion.div
   742→                          initial={{ height: 0, opacity: 0 }}
   743→                          animate={{ height: 'auto', opacity: 1 }}
   744→                          exit={{ height: 0, opacity: 0 }}
   745→                          transition={{ duration: 0.2 }}
   746→                          className="overflow-hidden"
   747→                        >
   748→                          <div className="mt-4 pt-4 border-t border-border/50 space-y-3">
   749→                            {qReplies.map((reply: any) => (
   750→                              <div key={reply.id} className={cn('ml-8 p-3 rounded-lg bg-muted/30')}>
   751→                                <p className="text-xs text-muted-foreground leading-relaxed">{reply.content}</p>
   752→                                <div className="flex items-center gap-3 mt-2">
   753→                                  <span className="text-[10px] text-muted-foreground">{timeAgo(reply.createdAt)}</span>
   754→                                  <button className="flex items-center gap-1 text-[10px] text-muted-foreground hover:text-foreground transition-colors">
   755→                                    <ThumbsUp className="h-3 w-3" />
   756→                                  </button>
   757→                                </div>
   758→                              </div>
   759→                            ))}
   760→
   761→                            {/* Reply input */}
   762→                            <div className="ml-8 flex gap-2">
   763→                              <div className="flex-1">
   764→                                <Input
   765→                                  placeholder="Write a reply..."
   766→                                  value={replyingTo === question.id ? replyContent : ''}
   767→                                  onChange={(e) => { setReplyingTo(question.id); setReplyContent(e.target.value); if (discussionErrors.replyContent) setDiscussionErrors({}); }}
   768→                                  className={`h-8 text-xs ${discussionErrors.replyContent && replyingTo === question.id ? 'border-destructive' : ''}`}

   774→                                size="sm"
   775→                                className="h-8 gap-1 bg-emerald-600 hover:bg-emerald-700 shrink-0"

   778→                              >
   779→                                {createDiscussion.isPending ? <Loader2 className="h-3 w-3 animate-spin" /> : <Send className="h-3 w-3" />}
   780→                              </Button>
   781→                            </div>
   782→                          </div>
   783→                        </motion.div>
   784→                      )}
   785→                    </AnimatePresence>
   786→                  </CardContent>
   787→                </Card>
   788→              </motion.div>
   789→            );
   790→          })}
   791→        </AnimatePresence>
   792→      </div>
   793→
   794→      {filteredQuestions.length === 0 && (
   795→        <div className="text-center py-8">
   796→          <MessageCircleQuestion className="h-12 w-12 text-muted-foreground/30 mx-auto mb-3" />
   797→          <p className="text-sm text-muted-foreground">No discussions yet. Be the first to ask a question!</p>
   798→        </div>
   799→      )}
   800→
   801→      <Button variant="outline" className="w-full">View All Discussions</Button>
   802→    </div>
   803→  );

   806→// ─── Q&A Discussion Tab ────────────────────────────────────
   807→function QADiscussionTab({ courseId, userId, modules }: { courseId: string; userId: string; modules: Module[] }) {
   808→  const { data: discussionsData, isLoading } = useLessonDiscussions({ courseId });
   809→  const createDiscussion = useCreateDiscussion();
   810→
   811→  const [selectedLessonId, setSelectedLessonId] = useState<string>('all');
   812→  const [searchQuery, setSearchQuery] = useState('');
   813→  const [sortBy, setSortBy] = useState<'newest' | 'voted' | 'unresolved' | 'resolved'>('newest');
   814→  const [filterBy, setFilterBy] = useState<'all' | 'my' | 'unresolved' | 'resolved'>('all');
   815→  const [selectedQuestion, setSelectedQuestion] = useState<any | null>(null);
   816→  const [showAskDialog, setShowAskDialog] = useState(false);
   817→  const [askContent, setAskContent] = useState('');
   818→  const [askLessonId, setAskLessonId] = useState<string>(selectedLessonId === 'all' ? '' : selectedLessonId);
   819→  const [answerContent, setAnswerContent] = useState('');
   820→  const [qaErrors, setQaErrors] = useState<Record<string, string>>({});
   821→  const [expandedModules, setExpandedModules] = useState<Set<string>>(new Set(modules.map(m => m.id)));
   822→
   823→  const discussions = discussionsData?.discussions || [];
   824→
   825→  // Separate top-level from replies & build reply map
   826→  const { qaTopLevel, repliesMap } = useMemo(() => {
   827→    const top: any[] = [];
   828→    const map: Record<string, any[]> = {};
   829→    discussions.forEach((d: any) => {
   830→      if (d.parentId) {
   831→        if (!map[d.parentId]) map[d.parentId] = [];
   832→        map[d.parentId].push(d);
   833→      } else {
   834→        top.push(d);
   835→      }
   836→    });
   837→    return { qaTopLevel: top, repliesMap: map };
   838→  }, [discussions]);
   839→
   840→  // Build lesson map from modules
   841→  const lessonMap = new Map<string, { title: string; moduleId: string; moduleTitle: string }>();
   842→  modules.forEach(mod => {
   843→    (mod.lessons || []).forEach(lesson => {
   844→      lessonMap.set(lesson.id, { title: lesson.title, moduleId: mod.id, moduleTitle: mod.title });
   845→    });
   846→  });
   847→
   848→  // Filter questions
   849→  const filteredQuestions = qaTopLevel.filter((q: any) => {
   850→    if (selectedLessonId !== 'all' && q.lessonId !== selectedLessonId) return false;
   851→    if (filterBy === 'my' && q.userId !== userId) return false;
   852→    const qReplies = repliesMap[q.id] || [];
   853→    const isResolved = qReplies.length > 0;
   854→    if (filterBy === 'unresolved' && isResolved) return false;
   855→    if (filterBy === 'resolved' && !isResolved) return false;
   856→    if (searchQuery) {
   857→      const lq = searchQuery.toLowerCase();
   858→      if (!q.content.toLowerCase().includes(lq)) return false;
   859→    }
   860→    return true;
   861→  });
   862→
   863→  // Sort questions
   864→  const sortedQuestions = [...filteredQuestions].sort((a: any, b: any) => {
   865→    if (sortBy === 'newest') return new Date(b.createdAt).getTime() - new Date(a.createdAt).getTime();
   866→    if (sortBy === 'voted') return (repliesMap[b.id]?.length || 0) - (repliesMap[a.id]?.length || 0);
   867→    if (sortBy === 'unresolved') {
   868→      const aResolved = (repliesMap[a.id]?.length || 0) > 0;
   869→      const bResolved = (repliesMap[b.id]?.length || 0) > 0;
   870→      return aResolved === bResolved ? 0 : aResolved ? 1 : -1;
   871→    }
   872→    if (sortBy === 'resolved') {
   873→      const aResolved = (repliesMap[a.id]?.length || 0) > 0;
   874→      const bResolved = (repliesMap[b.id]?.length || 0) > 0;
   875→      return aResolved === bResolved ? 0 : aResolved ? -1 : 1;
   876→    }
   877→    return 0;
   878→  });
   879→
   880→  const toggleModule = (moduleId: string) => {
   881→    setExpandedModules(prev => {
   882→      const next = new Set(prev);
   883→      if (next.has(moduleId)) next.delete(moduleId);
   884→      else next.add(moduleId);
   885→      return next;
   886→    });
   887→  };
   888→
   889→  // Count questions per lesson
   890→  const questionCountsByLesson = new Map<string, number>();
   891→  const unresolvedCountsByLesson = new Map<string, number>();
   892→  qaTopLevel.forEach((q: any) => {
   893→    questionCountsByLesson.set(q.lessonId, (questionCountsByLesson.get(q.lessonId) || 0) + 1);
   894→    if (!(repliesMap[q.id]?.length > 0)) unresolvedCountsByLesson.set(q.lessonId, (unresolvedCountsByLesson.get(q.lessonId) || 0) + 1);
   895→  });
   896→
   897→  const unresolvedTotal = qaTopLevel.filter((q: any) => !(repliesMap[q.id]?.length > 0)).length;
   898→
   899→  // Render content with markdown
   900→  const renderContent = (content: string) => {
   901→    const parts = content.split(/(```[\s\S]*?```)/g);
   902→    return parts.map((part, i) => {
   903→      if (part.startsWith('```') && part.endsWith('```')) {
   904→        const codeContent = part.slice(3, -3).replace(/^\w+\n/, '');
   905→        return (
   906→          <pre key={i} className="bg-slate-900 dark:bg-slate-950 text-green-400 p-3 rounded-lg text-xs font-mono overflow-x-auto my-2 leading-relaxed">
   907→            <code>{codeContent}</code>
   908→          </pre>
   909→        );
   910→      }
   911→      const inlineParts = part.split(/(`[^`]+`)/g);
   912→      return (
   913→        <span key={i}>
   914→          {inlineParts.map((ip, j) => {
   915→            if (ip.startsWith('`') && ip.endsWith('`')) {
   916→              return <code key={j} className="bg-slate-100 dark:bg-slate-800 text-emerald-700 dark:text-emerald-400 px-1.5 py-0.5 rounded text-xs font-mono">{ip.slice(1, -1)}</code>;
   917→            }
   918→            const boldParts = ip.split(/(\*\*[^*]+\*\*)/g);
   919→            return boldParts.map((bp, k) => {
   920→              if (bp.startsWith('**') && bp.endsWith('**')) {
   921→                return <strong key={`${j}-${k}`} className="font-semibold text-foreground">{bp.slice(2, -2)}</strong>;
   922→              }
   923→              return <span key={`${j}-${k}`}>{bp}</span>;
   924→            });
   925→          })}
   926→        </span>
   927→      );
   928→    });
   929→  };
   930→
   931→  // Handle submit question
   932→  const handleSubmitQuestion = () => {
   933→    const errs = validateFields({
   934→      askContent: [required(askContent, 'Question'), minLength(askContent, 5, 'Question')],
   935→    });
   936→    if (!askLessonId) errs.askLessonId = 'Please select a lesson';
   937→    setQaErrors(errs);
   938→    if (Object.keys(errs).length > 0) return;
   939→    createDiscussion.mutate({
   940→      lessonId: askLessonId,
   941→      userId,
   942→      content: askContent,
   943→    }, {
   944→      onSuccess: () => {
   945→        setAskContent('');
   946→        setShowAskDialog(false);
   947→        setQaErrors({});
   948→      },
   949→    });
   950→  };
   951→
   952→  // Handle submit answer
   953→  const handleSubmitAnswer = () => {
   954→    const errs = validateFields({
   955→      answerContent: [required(answerContent, 'Answer'), minLength(answerContent, 5, 'Answer')],
   956→    });
   957→    setQaErrors(errs);
   958→    if (Object.keys(errs).length > 0) return;
   959→    if (!selectedQuestion) return;
   960→    createDiscussion.mutate({
   961→      lessonId: selectedQuestion.lessonId,
   962→      userId,
   963→      content: answerContent,
   964→      parentId: selectedQuestion.id,
   965→    }, {
   966→      onSuccess: () => {
   967→        setAnswerContent('');
   968→        setQaErrors({});
   969→      },
   970→    });
   971→  };
   972→
   973→  if (isLoading) {
   974→    return (
   975→      <div className="space-y-3">
   976→        {Array.from({ length: 3 }).map((_, i) => (
   977→          <div key={i} className="rounded-xl p-4 bg-muted/30 animate-pulse space-y-3">
   978→            <div className="h-4 bg-muted rounded w-1/2" />
   979→            <div className="h-3 bg-muted rounded w-3/4" />
   980→          </div>
   981→        ))}
   982→      </div>
   983→    );
   984→  }
   985→
   986→  // ─── Thread Detail View ─────────────────────────────────

  1691→function ProgressTab({ progressData, enrollmentProgress, course, streakDays }: {
  1692→  progressData: any[];
  1693→  enrollmentProgress: number;
  1694→  course: any;
  1695→  streakDays: number;
  1696→}) {
  1697→  // Derive timeline events from progress data
  1698→  const timelineEvents = useMemo(() => {
  1699→    if (!progressData || !course) return [];
  1700→    const events: { id: string; lessonTitle: string; moduleName: string; date: string; timeSpent: number; type: 'completed' | 'started' }[] = [];
  1701→
  1702→    // Build lesson lookup from course modules
  1703→    const lessonLookup = new Map<string, { title: string; moduleName: string }>();
  1704→    (course.modules || []).forEach((mod: any) => {
  1705→      (mod.lessons || []).forEach((lesson: any) => {
  1706→        lessonLookup.set(lesson.id, { title: lesson.title, moduleName: mod.title });
  1707→      });
  1708→    });
  1709→
  1710→    progressData.forEach((p: any) => {
  1711→      const lessonInfo = lessonLookup.get(p.lessonId);
  1712→      if (!lessonInfo) return;
  1713→
  1714→      const date = p.completedAt
  1715→        ? new Date(p.completedAt).toLocaleDateString('en-US', { month: 'short', day: 'numeric', year: 'numeric' })
  1716→        : new Date(p.updatedAt || p.createdAt).toLocaleDateString('en-US', { month: 'short', day: 'numeric', year: 'numeric' });
  1717→
  1718→      events.push({
  1719→        id: p.id,
  1720→        lessonTitle: lessonInfo.title,
  1721→        moduleName: lessonInfo.moduleName,
  1722→        date,
  1723→        timeSpent: Math.round((p.timeSpent || 0) / 60),
  1724→        type: p.status === 'completed' ? 'completed' : 'started',
  1725→      });
  1726→    });
  1727→
  1728→    // Sort by date (most recent first)
  1729→    events.sort((a, b) => new Date(b.date).getTime() - new Date(a.date).getTime());

  1732→
  1733→  // Compute weekly activity from progress

  1741→      const date = new Date(now);
  1742→      date.setDate(now.getDate() + mondayOffset + i);
  1743→      const dateStr = date.toDateString();

  1745→      let totalMinutes = 0;

  1748→        if (updatedDate === dateStr) {
  1749→          totalMinutes += (p.timeSpent || 0) / 60;

  1753→      return { day, hours: Math.round(totalMinutes * 10) / 10 };
  1754→    });

  1761→    (mod.lessons || []).every((l: any) => progressData?.some((p: any) => p.lessonId === l.id && p.status === 'completed'))

  1768→    { id: 'ms-4', title: 'Halfway There', description: 'Complete 50% of the course', icon: '🎓', achieved: enrollmentProgress >= 50, date: '' },
  1769→    { id: 'ms-5', title: 'Streak Champion', description: 'Maintain a 7-day learning streak', icon: '🔥', achieved: streakDays >= 7, date: '' },

  1772→
  1773→  const totalTime = Math.round(progressData?.reduce((sum: number, p: any) => sum + (p.timeSpent || 0), 0) / 60) || 0;
  1774→  const maxHours = Math.max(...weeklyActivity.map(d => d.hours), 0.1);
