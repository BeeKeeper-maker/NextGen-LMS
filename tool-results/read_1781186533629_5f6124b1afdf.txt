    79Рєњ    79Рєњ  const now = new Date();
    80Рєњ    80Рєњ  const date = new Date(dateStr);
    81Рєњ    81Рєњ  const diffMs = now.getTime() - date.getTime();
    82Рєњ    82Рєњ  const diffMins = Math.floor(diffMs / 60000);
    83Рєњ    83Рєњ  if (diffMins < 60) return `${diffMins}m ago`;
    84Рєњ    84Рєњ  const diffHours = Math.floor(diffMins / 60);
    85Рєњ    85Рєњ  if (diffHours < 24) return `${diffHours}h ago`;
    86Рєњ    86Рєњ  const diffDays = Math.floor(diffHours / 24);
    87Рєњ    87Рєњ  if (diffDays < 30) return `${diffDays}d ago`;
    88Рєњ    88Рєњ  return `${Math.floor(diffDays / 30)}mo ago`;
    89Рєњ    89Рєњ}
