   390→// ─── Animated Counter ────────────────────────────────────────
   391→function AnimatedCounter({ value, duration = 1500 }: { value: string; duration?: number }) {
   392→  const [displayed, setDisplayed] = useState('0');
   393→  const hasAnimated = useRef(false);
   394→
   395→  useEffect(() => {
   396→    if (hasAnimated.current) return;
   397→    hasAnimated.current = true;
   398→
   399→    const numericMatch = value.match(/[\d,.]+/);
   400→    if (!numericMatch) {
   401→      return;
   402→    }
   403→
   404→    const numericStr = numericMatch[0];
   405→    const prefix = value.slice(0, value.indexOf(numericStr));
   406→    const suffix = value.slice(value.indexOf(numericStr) + numericStr.length);
   407→    const cleanNumber = parseFloat(numericStr.replace(/,/g, ''));
   408→    const hasDecimals = numericStr.includes('.');
   409→    const decimalPlaces = hasDecimals ? numericStr.split('.')[1].length : 0;
   410→
   411→    const startTime = Date.now();
   412→    const animate = () => {
   413→      const elapsed = Date.now() - startTime;
   414→      const progress = Math.min(elapsed / duration, 1);
   415→      const eased = 1 - Math.pow(1 - progress, 3);
   416→      const current = cleanNumber * eased;
   417→
   418→      let formatted: string;
   419→      if (hasDecimals) {
   420→        formatted = current.toFixed(decimalPlaces);
   421→      } else {
   422→        formatted = Math.round(current).toLocaleString('en-US');
   423→      }
   424→
   425→      setDisplayed(`${prefix}${formatted}${suffix}`);
   426→
   427→      if (progress < 1) {
   428→        requestAnimationFrame(animate);
   429→      }
   430→    };
   431→
   432→    requestAnimationFrame(animate);
   433→  }, [value, duration]);
   434→
   435→  const numericMatch = value.match(/[\d,.]+/);
   436→  if (!numericMatch) {
   437→    return <span>{value}</span>;
   438→  }
   439→
   440→  return <span>{displayed}</span>;
   441→}

   443→// ─── Percentage Animated Counter ─────────────────────────────
   444→function PercentageCounter({ value, duration = 1200 }: { value: number; duration?: number }) {
   445→  const [displayed, setDisplayed] = useState(0);
   446→  const hasAnimated = useRef(false);
   447→
   448→  useEffect(() => {
   449→    if (hasAnimated.current) return;
   450→    hasAnimated.current = true;
   451→
   452→    const startTime = Date.now();
   453→    const animate = () => {
   454→      const elapsed = Date.now() - startTime;
   455→      const progress = Math.min(elapsed / duration, 1);
   456→      const eased = 1 - Math.pow(1 - progress, 3);
   457→      const current = value * eased;
   458→      setDisplayed(current);
   459→
   460→      if (progress < 1) {
   461→        requestAnimationFrame(animate);
   462→      }
   463→    };
   464→
   465→    requestAnimationFrame(animate);
   466→  }, [value, duration]);
   467→
   468→  return <span>{displayed.toFixed(1)}%</span>;
   469→}

   471→// ─── Circular Progress Ring ──────────────────────────────────
   472→function ProgressRing({ percent, size = 32, strokeWidth = 3, color = '#10b981', animate = true }: { percent: number; size?: number; strokeWidth?: number; color?: string; animate?: boolean }) {
   473→  const radius = (size - strokeWidth) / 2;
   474→  const circumference = 2 * Math.PI * radius;
   475→  const offset = circumference - (percent / 100) * circumference;
   476→
   477→  return (
   478→    <svg width={size} height={size} className="shrink-0">
   479→      <circle
   480→        cx={size / 2}
   481→        cy={size / 2}
   482→        r={radius}
   483→        fill="none"
   484→        stroke="currentColor"
   485→        strokeWidth={strokeWidth}
   486→        className="text-muted/30"
   487→      />
   488→      <motion.circle
   489→        cx={size / 2}
   490→        cy={size / 2}
   491→        r={radius}
   492→        fill="none"
   493→        stroke={color}
   494→        strokeWidth={strokeWidth}
   495→        strokeLinecap="round"
   496→        strokeDasharray={circumference}
   497→        initial={animate ? { strokeDashoffset: circumference } : { strokeDashoffset: offset }}
   498→        animate={{ strokeDashoffset: offset }}
   499→        transition={{ duration: 1, ease: 'easeOut' }}
   500→        transform={`rotate(-90 ${size / 2} ${size / 2})`}
   501→      />
   502→    </svg>
   503→  );
   504→}

  1024→  const funnelColors = [
