   116тЖТ   116тЖТ/** Map API review data to the CourseReview display type */
   117тЖТ   117тЖТfunction mapApiReviewToDisplay(apiReview: any): CourseReview {
   118тЖТ   118тЖТ  const authorName = apiReview.author?.name || 'Unknown User';
   119тЖТ   119тЖТ  const initials = authorName.split(' ').map((n: string) => n[0]).join('').toUpperCase().slice(0, 2);
   120тЖТ   120тЖТ  // Extract title from content: first sentence or first 60 chars
   121тЖТ   121тЖТ  const firstSentence = apiReview.content?.split(/[.!\n]/)[0] || '';
   122тЖТ   122тЖТ  const title = firstSentence.length > 60 ? firstSentence.slice(0, 57) + '...' : firstSentence || 'Course Review';
   123тЖТ   123тЖТ  // Map API status to display status
   124тЖТ   124тЖТ  let displayStatus: CourseReview['status'] = 'published';
   125тЖТ   125тЖТ  if (apiReview.status === 'flagged' || apiReview.flagged) displayStatus = 'flagged';
   126тЖТ   126тЖТ  else if (apiReview.status === 'rejected') displayStatus = 'hidden';
   127тЖТ   127тЖТ  else if (apiReview.status === 'pending' || apiReview.status === 'approved') displayStatus = 'published';
   128тЖТ   128тЖТ  return {
   129тЖТ   129тЖТ    id: apiReview.id,
   130тЖТ   130тЖТ    courseId: apiReview.courseId,
   131тЖТ   131тЖТ    userId: apiReview.authorId,
   132тЖТ   132тЖТ    userName: authorName,
   133тЖТ   133тЖТ    userAvatar: initials,
   134тЖТ   134тЖТ    rating: apiReview.rating,
   135тЖТ   135тЖТ    title,
   136тЖТ   136тЖТ    content: apiReview.content || '',
   137тЖТ   137тЖТ    date: apiReview.createdAt,
   138тЖТ   138тЖТ    helpfulCount: 0,
   139тЖТ   139тЖТ    isHelpful: false,
   140тЖТ   140тЖТ    courseProgress: 0,
   141тЖТ   141тЖТ    isVerifiedPurchase: (apiReview.author?._count?.enrollments ?? 0) > 0,
   142тЖТ   142тЖТ    tags: [],
   143тЖТ   143тЖТ    status: displayStatus,
   144тЖТ   144тЖТ    instructorReply: apiReview.adminResponse
   145тЖТ   145тЖТ      ? { instructorName: 'Admin', content: apiReview.adminResponse, date: apiReview.updatedAt }
   146тЖТ   146тЖТ      : undefined,
   147тЖТ   147тЖТ    isOwnReview: false,
   148тЖТ   148тЖТ  };
