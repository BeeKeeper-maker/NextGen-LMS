   397→// ─── Profile Completion Ring Component ───────────────────
   398→function ProfileCompletionRing({ percentage, size = 64 }: { percentage: number; size?: number }) {
   399→  const strokeWidth = 4;
   400→  const radius = (size - strokeWidth) / 2;
   401→  const circumference = 2 * Math.PI * radius;
   402→  const offset = circumference - (percentage / 100) * circumference;
   403→
   404→  return (
   405→    <div className="relative" style={{ width: size, height: size }}>
   406→      <svg width={size} height={size} className="-rotate-90">
   407→        <circle
   408→          cx={size / 2}
   409→          cy={size / 2}
   410→          r={radius}
   411→          fill="none"
   412→          stroke="hsl(var(--muted))"
   413→          strokeWidth={strokeWidth}
   414→        />
   415→        <motion.circle
   416→          cx={size / 2}
   417→          cy={size / 2}
   418→          r={radius}
   419→          fill="none"
   420→          stroke="url(#completionGradient)"
   421→          strokeWidth={strokeWidth}
   422→          strokeLinecap="round"
   423→          strokeDasharray={circumference}
   424→          initial={{ strokeDashoffset: circumference }}
   425→          animate={{ strokeDashoffset: offset }}
   426→          transition={{ duration: 1.5, ease: 'easeOut', delay: 0.5 }}
   427→        />
   428→        <defs>
   429→          <linearGradient id="completionGradient" x1="0%" y1="0%" x2="100%" y2="0%">
   430→            <stop offset="0%" stopColor="#10b981" />
   431→            <stop offset="100%" stopColor="#06b6d4" />
   432→          </linearGradient>
   433→        </defs>
   434→      </svg>
   435→      <div className="absolute inset-0 flex items-center justify-center">
   436→        <span className="text-xs font-bold text-foreground">{percentage}%</span>
   437→      </div>
   438→    </div>
   439→  );
