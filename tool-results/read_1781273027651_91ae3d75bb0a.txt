   259→// ============================================================
   260→// Typing Animation Component
   261→// ============================================================
   262→function TypingText({ text, speed = 35, delay = 0 }: { text: string; speed?: number; delay?: number }) {
   263→  const [displayed, setDisplayed] = useState('');
   264→  const [started, setStarted] = useState(false);
   265→
   266→  useEffect(() => {
   267→    const timer = setTimeout(() => setStarted(true), delay);
   268→    return () => clearTimeout(timer);
   269→  }, [delay]);
   270→
   271→  useEffect(() => {
   272→    if (!started) return;
   273→    if (displayed.length < text.length) {
   274→      const timer = setTimeout(() => {
   275→        setDisplayed(text.slice(0, displayed.length + 1));
   276→      }, speed);
   277→      return () => clearTimeout(timer);
   278→    }
   279→  }, [displayed, text, speed, started]);
   280→
   281→  return (
   282→    <span>
   283→      {displayed}
   284→      {displayed.length < text.length && started && (
   285→        <span className="inline-block w-0.5 h-5 bg-emerald-500 ml-0.5 animate-pulse align-middle" />
   286→      )}
   287→    </span>
   288→  );
   289→}

   294→function RotatingText({ phrases, interval = 3000 }: { phrases: string[]; interval?: number }) {
   295→  const [index, setIndex] = useState(0);
   296→  const [direction, setDirection] = useState(1);
   297→
   298→  useEffect(() => {
   299→    const timer = setInterval(() => {
   300→      setDirection(1);
   301→      setIndex((prev) => (prev + 1) % phrases.length);
   302→    }, interval);
   303→    return () => clearInterval(timer);
   304→  }, [phrases.length, interval]);
   305→
   306→  return (
   307→    <AnimatePresence mode="wait" custom={direction}>
   308→      <motion.span
   309→        key={phrases[index]}
   310→        custom={direction}
   311→        initial={{ y: 20, opacity: 0 }}
   312→        animate={{ y: 0, opacity: 1 }}
   313→        exit={{ y: -20, opacity: 0 }}
   314→        transition={{ duration: 0.4, ease: 'easeInOut' }}
   315→        className="inline-block bg-gradient-to-r from-emerald-600 via-emerald-500 to-teal-500 bg-clip-text text-transparent font-semibold"
   316→      >
   317→        {phrases[index]}
   318→      </motion.span>
   319→    </AnimatePresence>
   320→  );
   321→}

   361→  id: i,

   363→  y: ((i * 23 + 11) % 90) + 5,

   371→// Pre-compute constellation lines (deterministic, no state needed)
   372→const constellationLines = constellationData.flatMap((p1, i) =>
   373→  constellationData.slice(i + 1).flatMap((p2) => {
   374→    const dx = p1.x - p2.x;
   375→    const dy = p1.y - p2.y;
   376→    const dist = Math.sqrt(dx * dx + dy * dy);
   377→    if (dist < 25) {
   378→      return [{ x1: p1.x, y1: p1.y, x2: p2.x, y2: p2.y, opacity: Math.max(0, 1 - dist / 25) * 0.15 }];
   379→    }
   380→    return [];
   381→  })
   382→);

   707→// ============================================================
   708→// Animated Counter Hook
   709→// ============================================================
   710→function useAnimatedCounter(target: number, duration: number = 800) {
   711→  const [count, setCount] = useState(0);
   712→  const prevTarget = useRef(0);
   713→
   714→  useEffect(() => {
   715→    if (target === prevTarget.current) return;
   716→    prevTarget.current = target;
   717→
   718→    const start = count;
   719→    const diff = target - start;
   720→    const startTime = performance.now();
   721→
   722→    const animate = (now: number) => {
   723→      const elapsed = now - startTime;
   724→      const progress = Math.min(elapsed / duration, 1);
   725→      // easeOutCubic
   726→      const eased = 1 - Math.pow(1 - progress, 3);
   727→      setCount(Math.round(start + diff * eased));
   728→      if (progress < 1) {
   729→        requestAnimationFrame(animate);
   730→      }
   731→    };
   732→
   733→    requestAnimationFrame(animate);
   734→  }, [target, duration]);
   735→
   736→  return count;
   737→}

  1007→  function StatCounter({ target, suffix, label, gradient, isDecimal = false }: {
  1008→    target: number; suffix: string; label: string; gradient: string; isDecimal?: boolean;
  1009→  }) {
  1010→    const [count, setCount] = useState(0);

  1022→        const elapsed = now - startTime;
  1023→        const progress = Math.min(elapsed / duration, 1);
  1024→        // easeOutExpo
  1025→        const eased = progress === 1 ? 1 : 1 - Math.pow(2, -10 * progress);

  1830→function AnimatedPrice({ target, isInView }: { target: number; isInView: boolean }) {
  1831→  const [count, setCount] = useState(0);
  1832→  const hasAnimated = useRef(false);
  1833→
  1834→  useEffect(() => {
  1835→    if (!isInView || hasAnimated.current) return;
  1836→    hasAnimated.current = true;
  1837→
  1838→    const duration = 1200;
  1839→    const startTime = performance.now();
  1840→
  1841→    const animate = (now: number) => {
  1842→      const elapsed = now - startTime;
  1843→      const progress = Math.min(elapsed / duration, 1);
  1844→      const eased = 1 - Math.pow(1 - progress, 3);
  1845→      setCount(Math.round(target * eased));
  1846→      if (progress < 1) {
  1847→        requestAnimationFrame(animate);
  1848→      }
  1849→    };
  1850→
  1851→    requestAnimationFrame(animate);
  1852→  }, [target, isInView]);
  1853→
  1854→  return <span>{count}</span>;
  1855→}
