   225→// ---- Animated Counter ----
   226→function AnimatedCounter({ value, duration = 1200 }: { value: number; duration?: number }) {
   227→  const [display, setDisplay] = useState(0);
   228→  const startRef = useRef<number | null>(null);
   229→
   230→  useEffect(() => {
   231→    startRef.current = null;
   232→    const step = (timestamp: number) => {
   233→      if (!startRef.current) startRef.current = timestamp;
   234→      const progress = Math.min((timestamp - startRef.current) / duration, 1);
   235→      setDisplay(Math.floor(progress * value));
   236→      if (progress < 1) requestAnimationFrame(step);
   237→    };
   238→    requestAnimationFrame(step);
   239→  }, [value, duration]);
   240→
   241→  return <>{display}</>;
   242→}

   364→// ---- Per-Question Statistics ----
   365→function QuestionStatsPanel({ questions }: { questions: Question[] }) {
   366→  // Generate simulated stats for each question
   367→  const stats = useMemo(() => {
   368→    return questions.map((q, idx) => {
   369→      const seed = q.id.charCodeAt(q.id.length - 1);
   370→      const timesAnswered = 45 + (seed * 13) % 200;
   371→      const correctPercent = q.difficulty === 'easy' ? 70 + (seed % 25) : q.difficulty === 'medium' ? 45 + (seed % 30) : 20 + (seed % 35);
   372→      const avgTime = q.difficulty === 'easy' ? 15 + (seed % 20) : q.difficulty === 'medium' ? 30 + (seed % 40) : 50 + (seed % 60);
   373→      const discriminationIndex = +(0.3 + ((seed * 7) % 60) / 100).toFixed(2);
   374→      const distractorCounts: Record<string, number> = {};
   375→      if (q.type === 'multiple_choice' && q.options) {
   376→        q.options.forEach((_, oi) => {
   377→          if (String(oi) !== q.correctAnswer) {
   378→            distractorCounts[String(oi)] = Math.floor((100 - correctPercent) * (0.2 + ((oi * 13 + seed) % 50) / 100));
   379→          }
   380→        });
   381→      }
   382→      return { ...q, timesAnswered, correctPercent, avgTime, discriminationIndex, distractorCounts, idx };
   383→    });
   384→  }, [questions]);
   385→
   386→  const getColorClass = (pct: number) => {
   387→    if (pct > 80) return 'text-emerald-600 dark:text-emerald-400';
   388→    if (pct >= 50) return 'text-amber-600 dark:text-amber-400';
   389→    return 'text-red-600 dark:text-red-400';
   390→  };
   391→
   392→  const getBgClass = (pct: number) => {
   393→    if (pct > 80) return 'bg-emerald-50 dark:bg-emerald-950/20 border-emerald-200 dark:border-emerald-800';
   394→    if (pct >= 50) return 'bg-amber-50 dark:bg-amber-950/20 border-amber-200 dark:border-amber-800';
   395→    return 'bg-red-50 dark:bg-red-950/20 border-red-200 dark:border-red-800';
   396→  };
   397→
   398→  const getDiscriminationLabel = (di: number) => {
   399→    if (di >= 0.7) return { text: 'Excellent', cls: 'text-emerald-600 dark:text-emerald-400' };
   400→    if (di >= 0.4) return { text: 'Good', cls: 'text-amber-600 dark:text-amber-400' };
   401→    return { text: 'Poor', cls: 'text-red-600 dark:text-red-400' };
   402→  };
   403→
   404→  return (
   405→    <div className="space-y-3 max-h-96 overflow-y-auto pr-1 custom-scrollbar">
   406→      {stats.map((s, i) => {
   407→        const disc = getDiscriminationLabel(s.discriminationIndex);
   408→        return (
   409→          <motion.div
   410→            key={s.id}
   411→            initial={{ opacity: 0, y: 8 }}
   412→            animate={{ opacity: 1, y: 0 }}
   413→            transition={{ delay: i * 0.05 }}
   414→            className={`p-3 rounded-lg border ${getBgClass(s.correctPercent)} transition-all hover:shadow-md`}
   415→          >
   416→            <div className="flex items-start justify-between gap-2">
   417→              <div className="flex-1 min-w-0">
   418→                <div className="flex items-center gap-2 mb-1">
   419→                  <span className="text-xs font-mono text-muted-foreground">Q{s.idx + 1}</span>
   420→                  <QuestionTypeBadge type={s.type} />
   421→                  <DifficultyBadge difficulty={s.difficulty} />
   422→                </div>
   423→                <p className="text-sm font-medium truncate">{s.question}</p>
   424→              </div>
   425→              <div className="text-right shrink-0">
   426→                <div className={`text-lg font-bold ${getColorClass(s.correctPercent)}`}>{s.correctPercent}%</div>
   427→                <div className="text-xs text-muted-foreground">correct</div>
   428→              </div>
   429→            </div>
   430→
   431→            <div className="grid grid-cols-3 gap-3 mt-2 pt-2 border-t border-black/5 dark:border-white/5">
   432→              <div className="text-center">
   433→                <div className="text-sm font-semibold">{s.timesAnswered}</div>
   434→                <div className="text-xs text-muted-foreground flex items-center justify-center gap-1">
   435→                  <Users className="h-3 w-3" /> answered
   436→                </div>
   437→              </div>
   438→              <div className="text-center">
   439→                <div className="text-sm font-semibold">{s.avgTime}s</div>
   440→                <div className="text-xs text-muted-foreground flex items-center justify-center gap-1">
   441→                  <Clock className="h-3 w-3" /> avg time
   442→                </div>
   443→              </div>
   444→              <div className="text-center">
   445→                <div className={`text-sm font-semibold ${disc.cls}`}>{s.discriminationIndex}</div>
   446→                <div className="text-xs text-muted-foreground">{disc.text}</div>
   447→              </div>
   448→            </div>
   449→
   450→            {/* Distractor analysis */}
   451→            {s.type === 'multiple_choice' && s.options && Object.keys(s.distractorCounts).length > 0 && (
   452→              <div className="mt-2 pt-2 border-t border-black/5 dark:border-white/5">

   638→function AssessmentList({

   649→  onEdit: (a: Assessment) => void;
   650→  onPreview: (a: Assessment) => void;

   654→}) {
   655→  const [search, setSearch] = useState('');
   656→  const [typeFilter, setTypeFilter] = useState<string>('all');
   657→  const [statusFilter, setStatusFilter] = useState<string>('all');
   658→  const [sortField, setSortField] = useState<SortField>('name');

   662→
   663→  const filtered = assessments.filter(a => {
   664→    const matchesSearch = a.title.toLowerCase().includes(search.toLowerCase()) ||
   665→      a.description?.toLowerCase().includes(search.toLowerCase());
   666→    const matchesType = typeFilter === 'all' || a.type === typeFilter;
   667→    const matchesStatus = statusFilter === 'all' ||
   668→      (statusFilter === 'published' && a.isPublished) ||
   669→      (statusFilter === 'draft' && !a.isPublished);
   670→    return matchesSearch && matchesType && matchesStatus;

   675→    const seed = a.id.charCodeAt(a.id.length - 1);

   677→      submissions: 12 + (seed * 7) % 156,
   678→      avgScore: 55 + (seed * 3) % 35,
   679→      passRate: 50 + (seed * 5) % 40,
   680→    };
   681→  };

   685→    const sa = getSimStats(a);

   688→      case 'name': cmp = a.title.localeCompare(b.title); break;

   690→      case 'submissions': cmp = sa.submissions - sb.submissions; break;
   691→      case 'avgScore': cmp = sa.avgScore - sb.avgScore; break;

   714→          </h2>
   715→          <p className="text-muted-foreground text-sm mt-1">Create and manage quizzes, assignments, and coding challenges</p>

   718→          <Plus className="h-4 w-4" /> Create Assessment

   720→      </div>

   722→      {/* Animated Stats Row */}
   723→      <div className="grid grid-cols-2 sm:grid-cols-4 gap-4">
   724→        {[
   725→          { label: 'Total Assessments', value: assessments.length, icon: ListChecks, gradient: 'from-violet-500 to-purple-500' },
   726→          { label: 'Total Submissions', value: totalSubmissions, icon: Send, gradient: 'from-emerald-500 to-teal-500' },
   727→          { label: 'Avg Score', value: avgScore, icon: Target, gradient: 'from-amber-500 to-orange-500', suffix: '%' },
   728→          { label: 'Pass Rate', value: passRate, icon: Award, gradient: 'from-rose-500 to-pink-500', suffix: '%' },
   729→        ].map((stat, i) => (
   730→          <motion.div
   731→            key={stat.label}
   732→            initial={{ opacity: 0, y: 12, scale: 0.95 }}
   733→            animate={{ opacity: 1, y: 0, scale: 1 }}
   734→            transition={{ delay: 0.1 + i * 0.1, type: 'spring', stiffness: 200 }}

   737→              <CardContent className="p-4 flex items-center gap-3">
   738→                <div className={`p-2.5 rounded-xl bg-gradient-to-br ${stat.gradient} shadow-lg group-hover:scale-110 transition-transform duration-300`}>
   739→                  <stat.icon className="h-5 w-5 text-white" />
