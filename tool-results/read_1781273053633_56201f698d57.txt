  1007→  function StatCounter({ target, suffix, label, gradient, isDecimal = false }: {
  1008→    target: number; suffix: string; label: string; gradient: string; isDecimal?: boolean;
  1009→  }) {
  1010→    const [count, setCount] = useState(0);
  1011→    const prevTarget = useRef(0);
  1012→
  1013→    useEffect(() => {
  1014→      if (!isInView) return;
  1015→      if (target === prevTarget.current) return;
  1016→      prevTarget.current = target;
  1017→
  1018→      const duration = 2000;
  1019→      const startTime = performance.now();
  1020→
  1021→      const animate = (now: number) => {
  1022→        const elapsed = now - startTime;
  1023→        const progress = Math.min(elapsed / duration, 1);
  1024→        // easeOutExpo
  1025→        const eased = progress === 1 ? 1 : 1 - Math.pow(2, -10 * progress);
  1026→        const current = target * eased;
  1027→        setCount(isDecimal ? parseFloat(current.toFixed(1)) : Math.round(current));
  1028→        if (progress < 1) {
  1029→          requestAnimationFrame(animate);
  1030→        }
  1031→      };
  1032→
  1033→      requestAnimationFrame(animate);
  1034→    }, [target, isInView, isDecimal]);
  1035→
  1036→    const displayValue = isDecimal
  1037→      ? count.toFixed(1)
  1038→      : count.toLocaleString('en-US');
  1039→
  1040→    return (
  1041→      <motion.div
  1042→        initial={{ opacity: 0, y: 30 }}
  1043→        animate={isInView ? { opacity: 1, y: 0 } : {}}

  1056→          </svg>

  1058→
  1059→        <div className="relative rounded-2xl border border-border/30 bg-white/50 dark:bg-white/5 backdrop-blur-md p-6 sm:p-8 text-center transition-all duration-300 group-hover:border-border/60 group-hover:shadow-lg group-hover:scale-[1.02]">
