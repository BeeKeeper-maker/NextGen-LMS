    78→function timeAgo(dateStr: string) {
    79→  const now = new Date();
    80→  const date = new Date(dateStr);
    81→  const diffMs = now.getTime() - date.getTime();
    82→  const diffMins = Math.floor(diffMs / 60000);
    83→  if (diffMins < 60) return `${diffMins}m ago`;
    84→  const diffHours = Math.floor(diffMins / 60);
    85→  if (diffHours < 24) return `${diffHours}h ago`;
    86→  const diffDays = Math.floor(diffHours / 24);
    87→  if (diffDays < 30) return `${diffDays}d ago`;
    88→  return `${Math.floor(diffDays / 30)}mo ago`;
    89→}
