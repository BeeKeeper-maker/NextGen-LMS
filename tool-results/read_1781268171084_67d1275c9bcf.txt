   124→// ─── Format duration ───────────────────────────────────────
   125→function formatDuration(seconds?: number): string {
   126→  if (!seconds) return '';
   127→  const mins = Math.floor(seconds / 60);
   128→  if (mins < 60) return `${mins} min`;
   129→  const hours = Math.floor(mins / 60);
   130→  const remainMins = mins % 60;
   131→  return `${hours}h ${remainMins}m`;
   132→}

   167→function useLessonNotes(courseId: string | undefined) {
   168→  const storageKey = `lms-notes-${courseId || 'default'}`;
   169→
   170→  const [notes, setNotes] = useState<Note[]>(() => {
   171→    if (typeof window === 'undefined') return [];
   172→    try {
   173→      const stored = localStorage.getItem(storageKey);
   174→      return stored ? JSON.parse(stored) : [];
   175→    } catch {
   176→      return [];
   177→    }
   178→  });
   179→
   180→  const persistNotes = useCallback((updated: Note[]) => {
   181→    setNotes(updated);
   182→    if (typeof window !== 'undefined') {
   183→      try {
   184→        localStorage.setItem(storageKey, JSON.stringify(updated));
   185→      } catch {
   186→        // ignore quota errors
   187→      }
   188→    }
   189→  }, [storageKey]);
   190→
   191→  const addNote = useCallback((note: Omit<Note, 'id' | 'createdAt' | 'updatedAt'>) => {
   192→    const newNote: Note = {
   193→      ...note,
   194→      id: `note-${Date.now()}-${Math.random().toString(36).slice(2, 7)}`,
   195→      createdAt: new Date().toISOString(),
   196→      updatedAt: new Date().toISOString(),
   197→    };
   198→    persistNotes([...notes, newNote]);
   199→    return newNote;
   200→  }, [notes, persistNotes]);
   201→
   202→  const updateNote = useCallback((id: string, updates: Partial<Note>) => {
   203→    const updated = notes.map(n =>
   204→      n.id === id ? { ...n, ...updates, updatedAt: new Date().toISOString() } : n
   205→    );
   206→    persistNotes(updated);
   207→  }, [notes, persistNotes]);
   208→
   209→  const deleteNote = useCallback((id: string) => {
   210→    persistNotes(notes.filter(n => n.id !== id));
   211→  }, [notes, persistNotes]);
   212→
   213→  return { notes, addNote, updateNote, deleteNote };
   214→}

   244→// ─── Time ago helper ──────────────────────────────────────
   245→function timeAgo(dateStr: string): string {
   246→  const seconds = Math.floor((Date.now() - new Date(dateStr).getTime()) / 1000);
   247→  if (seconds < 60) return 'just now';
   248→  const minutes = Math.floor(seconds / 60);
   249→  if (minutes < 60) return `${minutes}m ago`;
   250→  const hours = Math.floor(minutes / 60);
   251→  if (hours < 24) return `${hours}h ago`;
   252→  const days = Math.floor(hours / 24);
   253→  if (days < 30) return `${days}d ago`;
   254→  const months = Math.floor(days / 30);
   255→  return `${months}mo ago`;
   256→}

   469→// ─── Discussion Q&A Tab ────────────────────────────────────

   488→    const top: any[] = [];
   489→    const map: Record<string, any[]> = {};
   490→    discussions.forEach((d: any) => {
   491→      if (d.parentId) {

   502→
   503→  const sortedQuestions = [...topLevelDiscussions].sort((a: any, b: any) => {
   504→    if (sortBy === 'recent') return new Date(b.createdAt).getTime() - new Date(a.createdAt).getTime();
   505→    if (sortBy === 'upvoted') return 0;
   506→    if (sortBy === 'unanswered') {

   719→                        </div>
   720→                        <p className="text-sm text-muted-foreground line-clamp-3">{question.content}</p>
   721→                        <div className="flex items-center gap-3 mt-2">
   722→                          <span className="text-xs text-muted-foreground">· {timeAgo(question.createdAt)}</span>

   734→                          initial={{ height: 0, opacity: 0 }}

   743→                                <p className="text-xs text-muted-foreground leading-relaxed">{reply.content}</p>
   744→                                <div className="flex items-center gap-3 mt-2">
   745→                                  <span className="text-[10px] text-muted-foreground">{timeAgo(reply.createdAt)}</span>
   746→                                  <button className="flex items-center gap-1 text-[10px] text-muted-foreground hover:text-foreground transition-colors">
   747→                                    <ThumbsUp className="h-3 w-3" />
   748→                                  </button>

   757→                                  placeholder="Write a reply..."
   758→                                  value={replyingTo === question.id ? replyContent : ''}
   759→                                  onChange={(e) => { setReplyingTo(question.id); setReplyContent(e.target.value); if (discussionErrors.replyContent) setDiscussionErrors({}); }}

   778→                  </CardContent>

   782→          })}
   783→        </AnimatePresence>
   784→      </div>

   786→      {filteredQuestions.length === 0 && (

   791→      )}
   792→
   793→      <Button variant="outline" className="w-full">View All Discussions</Button>

   798→// ─── Q&A Discussion Tab ────────────────────────────────────
   799→function QADiscussionTab({ courseId, userId, modules }: { courseId: string; userId: string; modules: Module[] }) {
   800→  const { data: discussionsData, isLoading } = useLessonDiscussions({ courseId });
   801→  const createDiscussion = useCreateDiscussion();
   802→
   803→  const [selectedLessonId, setSelectedLessonId] = useState<string>('all');
   804→  const [searchQuery, setSearchQuery] = useState('');
   805→  const [sortBy, setSortBy] = useState<'newest' | 'voted' | 'unresolved' | 'resolved'>('newest');
   806→  const [filterBy, setFilterBy] = useState<'all' | 'my' | 'unresolved' | 'resolved'>('all');
   807→  const [selectedQuestion, setSelectedQuestion] = useState<any | null>(null);
   808→  const [showAskDialog, setShowAskDialog] = useState(false);
   809→  const [askContent, setAskContent] = useState('');
   810→  const [askLessonId, setAskLessonId] = useState<string>(selectedLessonId === 'all' ? '' : selectedLessonId);
   811→  const [answerContent, setAnswerContent] = useState('');
   812→  const [qaErrors, setQaErrors] = useState<Record<string, string>>({});
   813→  const [expandedModules, setExpandedModules] = useState<Set<string>>(new Set(modules.map(m => m.id)));
   814→
   815→  const discussions = discussionsData?.discussions || [];
   816→
   817→  // Separate top-level from replies & build reply map
   818→  const { qaTopLevel, repliesMap } = useMemo(() => {
   819→    const top: any[] = [];
   820→    const map: Record<string, any[]> = {};
   821→    discussions.forEach((d: any) => {
   822→      if (d.parentId) {
   823→        if (!map[d.parentId]) map[d.parentId] = [];
   824→        map[d.parentId].push(d);
   825→      } else {
   826→        top.push(d);
   827→      }
   828→    });
   829→    return { qaTopLevel: top, repliesMap: map };
   830→  }, [discussions]);
   831→
   832→  // Build lesson map from modules
   833→  const lessonMap = new Map<string, { title: string; moduleId: string; moduleTitle: string }>();
   834→  modules.forEach(mod => {
   835→    (mod.lessons || []).forEach(lesson => {
   836→      lessonMap.set(lesson.id, { title: lesson.title, moduleId: mod.id, moduleTitle: mod.title });
   837→    });
   838→  });
   839→
   840→  // Filter questions
   841→  const filteredQuestions = qaTopLevel.filter((q: any) => {
   842→    if (selectedLessonId !== 'all' && q.lessonId !== selectedLessonId) return false;
   843→    if (filterBy === 'my' && q.userId !== userId) return false;
   844→    const qReplies = repliesMap[q.id] || [];
   845→    const isResolved = qReplies.length > 0;
   846→    if (filterBy === 'unresolved' && isResolved) return false;
   847→    if (filterBy === 'resolved' && !isResolved) return false;
   848→    if (searchQuery) {
   849→      const lq = searchQuery.toLowerCase();
   850→      if (!q.content.toLowerCase().includes(lq)) return false;
   851→    }
   852→    return true;
   853→  });
   854→
   855→  // Sort questions
   856→  const sortedQuestions = [...filteredQuestions].sort((a: any, b: any) => {
   857→    if (sortBy === 'newest') return new Date(b.createdAt).getTime() - new Date(a.createdAt).getTime();
   858→    if (sortBy === 'voted') return (repliesMap[b.id]?.length || 0) - (repliesMap[a.id]?.length || 0);
   859→    if (sortBy === 'unresolved') {
   860→      const aResolved = (repliesMap[a.id]?.length || 0) > 0;
   861→      const bResolved = (repliesMap[b.id]?.length || 0) > 0;
   862→      return aResolved === bResolved ? 0 : aResolved ? 1 : -1;
   863→    }
   864→    if (sortBy === 'resolved') {
   865→      const aResolved = (repliesMap[a.id]?.length || 0) > 0;
   866→      const bResolved = (repliesMap[b.id]?.length || 0) > 0;
   867→      return aResolved === bResolved ? 0 : aResolved ? -1 : 1;
   868→    }
   869→    return 0;
   870→  });
   871→
   872→  const toggleModule = (moduleId: string) => {
   873→    setExpandedModules(prev => {
   874→      const next = new Set(prev);
   875→      if (next.has(moduleId)) next.delete(moduleId);
   876→      else next.add(moduleId);
   877→      return next;
   878→    });
   879→  };
   880→
   881→  // Count questions per lesson
   882→  const questionCountsByLesson = new Map<string, number>();
   883→  const unresolvedCountsByLesson = new Map<string, number>();
   884→  qaTopLevel.forEach((q: any) => {
   885→    questionCountsByLesson.set(q.lessonId, (questionCountsByLesson.get(q.lessonId) || 0) + 1);
   886→    if (!(repliesMap[q.id]?.length > 0)) unresolvedCountsByLesson.set(q.lessonId, (unresolvedCountsByLesson.get(q.lessonId) || 0) + 1);
   887→  });
   888→
   889→  const unresolvedTotal = qaTopLevel.filter((q: any) => !(repliesMap[q.id]?.length > 0)).length;
   890→
   891→  // Render content with markdown
   892→  const renderContent = (content: string) => {
   893→    const parts = content.split(/(```[\s\S]*?```)/g);
   894→    return parts.map((part, i) => {
   895→      if (part.startsWith('```') && part.endsWith('```')) {
   896→        const codeContent = part.slice(3, -3).replace(/^\w+\n/, '');
   897→        return (
   898→          <pre key={i} className="bg-slate-900 dark:bg-slate-950 text-green-400 p-3 rounded-lg text-xs font-mono overflow-x-auto my-2 leading-relaxed">
   899→            <code>{codeContent}</code>
   900→          </pre>
   901→        );
   902→      }
   903→      const inlineParts = part.split(/(`[^`]+`)/g);
   904→      return (
   905→        <span key={i}>
   906→          {inlineParts.map((ip, j) => {
   907→            if (ip.startsWith('`') && ip.endsWith('`')) {
   908→              return <code key={j} className="bg-slate-100 dark:bg-slate-800 text-emerald-700 dark:text-emerald-400 px-1.5 py-0.5 rounded text-xs font-mono">{ip.slice(1, -1)}</code>;
   909→            }
   910→            const boldParts = ip.split(/(\*\*[^*]+\*\*)/g);
   911→            return boldParts.map((bp, k) => {
   912→              if (bp.startsWith('**') && bp.endsWith('**')) {
   913→                return <strong key={`${j}-${k}`} className="font-semibold text-foreground">{bp.slice(2, -2)}</strong>;
   914→              }
   915→              return <span key={`${j}-${k}`}>{bp}</span>;
   916→            });
   917→          })}
   918→        </span>
   919→      );
   920→    });
   921→  };
   922→
   923→  // Handle submit question
   924→  const handleSubmitQuestion = () => {
   925→    const errs = validateFields({
   926→      askContent: [required(askContent, 'Question'), minLength(askContent, 5, 'Question')],
   927→    });
   928→    if (!askLessonId) errs.askLessonId = 'Please select a lesson';
   929→    setQaErrors(errs);
   930→    if (Object.keys(errs).length > 0) return;
   931→    createDiscussion.mutate({
   932→      lessonId: askLessonId,
   933→      userId,
   934→      content: askContent,
   935→    }, {
   936→      onSuccess: () => {
   937→        setAskContent('');
   938→        setShowAskDialog(false);
   939→        setQaErrors({});
   940→      },
   941→    });
   942→  };
   943→
   944→  // Handle submit answer
   945→  const handleSubmitAnswer = () => {
   946→    const errs = validateFields({
   947→      answerContent: [required(answerContent, 'Answer'), minLength(answerContent, 5, 'Answer')],
   948→    });
   949→    setQaErrors(errs);
   950→    if (Object.keys(errs).length > 0) return;
   951→    if (!selectedQuestion) return;
   952→    createDiscussion.mutate({
   953→      lessonId: selectedQuestion.lessonId,
   954→      userId,
   955→      content: answerContent,
   956→      parentId: selectedQuestion.id,
   957→    }, {
   958→      onSuccess: () => {
   959→        setAnswerContent('');
   960→        setQaErrors({});
   961→      },
   962→    });
   963→  };
   964→
   965→  if (isLoading) {
   966→    return (
   967→      <div className="space-y-3">
   968→        {Array.from({ length: 3 }).map((_, i) => (
   969→          <div key={i} className="rounded-xl p-4 bg-muted/30 animate-pulse space-y-3">
   970→            <div className="h-4 bg-muted rounded w-1/2" />
   971→            <div className="h-3 bg-muted rounded w-3/4" />
   972→          </div>
   973→        ))}
   974→      </div>
   975→    );
   976→  }
   977→
   978→  // ─── Thread Detail View ─────────────────────────────────
   979→  if (selectedQuestion) {
   980→    const q = selectedQuestion;
   981→    const qReplies = repliesMap[q.id] || [];
   982→    const isResolved = qReplies.length > 0;
   983→

  1691→    if (!progressData || !course) return [];

  1697→      (mod.lessons || []).forEach((lesson: any) => {
  1698→        lessonLookup.set(lesson.id, { title: lesson.title, moduleName: mod.title });

  1705→

  1707→        ? new Date(p.completedAt).toLocaleDateString('en-US', { month: 'short', day: 'numeric', year: 'numeric' })
  1708→        : new Date(p.updatedAt || p.createdAt).toLocaleDateString('en-US', { month: 'short', day: 'numeric', year: 'numeric' });
  1709→
  1710→      events.push({
  1711→        id: p.id,
  1712→        lessonTitle: lessonInfo.title,
  1713→        moduleName: lessonInfo.moduleName,
  1714→        date,
  1715→        timeSpent: Math.round((p.timeSpent || 0) / 60),
  1716→        type: p.status === 'completed' ? 'completed' : 'started',
  1717→      });
  1718→    });

  1720→    // Sort by date (most recent first)
