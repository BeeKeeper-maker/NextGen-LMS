   224→// ---- Animated Counter ----
   225→function AnimatedCounter({ value, duration = 1200 }: { value: number; duration?: number }) {
   226→  const [display, setDisplay] = useState(0);
   227→  const startRef = useRef<number | null>(null);
   228→
   229→  useEffect(() => {
   230→    startRef.current = null;
   231→    const step = (timestamp: number) => {
   232→      if (!startRef.current) startRef.current = timestamp;
   233→      const progress = Math.min((timestamp - startRef.current) / duration, 1);
   234→      setDisplay(Math.floor(progress * value));
   235→      if (progress < 1) requestAnimationFrame(step);
   236→    };
   237→    requestAnimationFrame(step);
   238→  }, [value, duration]);
   239→
   240→  return <>{display}</>;
   241→}

   363→// ---- Per-Question Statistics ----
   364→function QuestionStatsPanel({ questions }: { questions: Question[] }) {
   365→  // Generate simulated stats for each question
   366→  const stats = useMemo(() => {
   367→    return questions.map((q, idx) => {
   368→      const seed = q.id.charCodeAt(q.id.length - 1);
   369→      const timesAnswered = 45 + (seed * 13) % 200;
   370→      const correctPercent = q.difficulty === 'easy' ? 70 + (seed % 25) : q.difficulty === 'medium' ? 45 + (seed % 30) : 20 + (seed % 35);
   371→      const avgTime = q.difficulty === 'easy' ? 15 + (seed % 20) : q.difficulty === 'medium' ? 30 + (seed % 40) : 50 + (seed % 60);
   372→      const discriminationIndex = +(0.3 + ((seed * 7) % 60) / 100).toFixed(2);
   373→      const distractorCounts: Record<string, number> = {};
   374→      if (q.type === 'multiple_choice' && q.options) {
   375→        q.options.forEach((_, oi) => {
   376→          if (String(oi) !== q.correctAnswer) {
   377→            distractorCounts[String(oi)] = Math.floor((100 - correctPercent) * (0.2 + ((oi * 13 + seed) % 50) / 100));
   378→          }
   379→        });
   380→      }
   381→      return { ...q, timesAnswered, correctPercent, avgTime, discriminationIndex, distractorCounts, idx };
   382→    });
   383→  }, [questions]);
   384→
   385→  const getColorClass = (pct: number) => {
   386→    if (pct > 80) return 'text-emerald-600 dark:text-emerald-400';
   387→    if (pct >= 50) return 'text-amber-600 dark:text-amber-400';
   388→    return 'text-red-600 dark:text-red-400';
   389→  };
   390→
   391→  const getBgClass = (pct: number) => {
   392→    if (pct > 80) return 'bg-emerald-50 dark:bg-emerald-950/20 border-emerald-200 dark:border-emerald-800';
   393→    if (pct >= 50) return 'bg-amber-50 dark:bg-amber-950/20 border-amber-200 dark:border-amber-800';
   394→    return 'bg-red-50 dark:bg-red-950/20 border-red-200 dark:border-red-800';
   395→  };
   396→
   397→  const getDiscriminationLabel = (di: number) => {
   398→    if (di >= 0.7) return { text: 'Excellent', cls: 'text-emerald-600 dark:text-emerald-400' };
   399→    if (di >= 0.4) return { text: 'Good', cls: 'text-amber-600 dark:text-amber-400' };
   400→    return { text: 'Poor', cls: 'text-red-600 dark:text-red-400' };
   401→  };
   402→
   403→  return (
   404→    <div className="space-y-3 max-h-96 overflow-y-auto pr-1 custom-scrollbar">
   405→      {stats.map((s, i) => {
   406→        const disc = getDiscriminationLabel(s.discriminationIndex);
   407→        return (
   408→          <motion.div
   409→            key={s.id}
   410→            initial={{ opacity: 0, y: 8 }}
   411→            animate={{ opacity: 1, y: 0 }}
   412→            transition={{ delay: i * 0.05 }}
   413→            className={`p-3 rounded-lg border ${getBgClass(s.correctPercent)} transition-all hover:shadow-md`}
   414→          >
   415→            <div className="flex items-start justify-between gap-2">
   416→              <div className="flex-1 min-w-0">
   417→                <div className="flex items-center gap-2 mb-1">
   418→                  <span className="text-xs font-mono text-muted-foreground">Q{s.idx + 1}</span>
   419→                  <QuestionTypeBadge type={s.type} />
   420→                  <DifficultyBadge difficulty={s.difficulty} />
   421→                </div>
   422→                <p className="text-sm font-medium truncate">{s.question}</p>
   423→              </div>
   424→              <div className="text-right shrink-0">
   425→                <div className={`text-lg font-bold ${getColorClass(s.correctPercent)}`}>{s.correctPercent}%</div>
   426→                <div className="text-xs text-muted-foreground">correct</div>
   427→              </div>
   428→            </div>
   429→
   430→            <div className="grid grid-cols-3 gap-3 mt-2 pt-2 border-t border-black/5 dark:border-white/5">
   431→              <div className="text-center">
   432→                <div className="text-sm font-semibold">{s.timesAnswered}</div>
   433→                <div className="text-xs text-muted-foreground flex items-center justify-center gap-1">
   434→                  <Users className="h-3 w-3" /> answered
   435→                </div>
   436→              </div>
   437→              <div className="text-center">
   438→                <div className="text-sm font-semibold">{s.avgTime}s</div>
   439→                <div className="text-xs text-muted-foreground flex items-center justify-center gap-1">
   440→                  <Clock className="h-3 w-3" /> avg time
   441→                </div>
   442→              </div>
   443→              <div className="text-center">
   444→                <div className={`text-sm font-semibold ${disc.cls}`}>{s.discriminationIndex}</div>
   445→                <div className="text-xs text-muted-foreground">{disc.text}</div>
   446→              </div>
   447→            </div>
   448→
   449→            {/* Distractor analysis */}
   450→            {s.type === 'multiple_choice' && s.options && Object.keys(s.distractorCounts).length > 0 && (
   451→              <div className="mt-2 pt-2 border-t border-black/5 dark:border-white/5">

   637→function AssessmentList({

   647→  courses: any[];
   648→  onEdit: (a: Assessment) => void;
   649→  onPreview: (a: Assessment) => void;
   650→  onCreateNew: () => void;

   653→}) {
   654→  const [search, setSearch] = useState('');
   655→  const [typeFilter, setTypeFilter] = useState<string>('all');
   656→  const [statusFilter, setStatusFilter] = useState<string>('all');
   657→  const [sortField, setSortField] = useState<SortField>('name');
   658→  const [sortAsc, setSortAsc] = useState(true);
   659→  const [quickPreviewId, setQuickPreviewId] = useState<string | null>(null);

   661→
   662→  const filtered = assessments.filter(a => {
   663→    const matchesSearch = a.title.toLowerCase().includes(search.toLowerCase()) ||
   664→      a.description?.toLowerCase().includes(search.toLowerCase());
   665→    const matchesType = typeFilter === 'all' || a.type === typeFilter;
   666→    const matchesStatus = statusFilter === 'all' ||
   667→      (statusFilter === 'published' && a.isPublished) ||
   668→      (statusFilter === 'draft' && !a.isPublished);
   669→    return matchesSearch && matchesType && matchesStatus;
   670→  });
   671→
   672→  // Simulated stats
   673→  const getSimStats = (a: Assessment) => {
   674→    const seed = a.id.charCodeAt(a.id.length - 1);

   676→      submissions: 12 + (seed * 7) % 156,
   677→      avgScore: 55 + (seed * 3) % 35,
   678→      passRate: 50 + (seed * 5) % 40,
   679→    };
   680→  };

   688→      case 'date': cmp = a.id.localeCompare(b.id); break;
   689→      case 'submissions': cmp = sa.submissions - sb.submissions; break;
   690→      case 'avgScore': cmp = sa.avgScore - sb.avgScore; break;
   691→    }
   692→    return sortAsc ? cmp : -cmp;

   703→  const avgScore = Math.round(assessments.reduce((s, a) => s + getSimStats(a).avgScore, 0) / (assessments.length || 1));

   712→            Assessments
   713→          </h2>
   714→          <p className="text-muted-foreground text-sm mt-1">Create and manage quizzes, assignments, and coding challenges</p>
   715→        </div>
   716→        <Button onClick={onCreateNew} className="bg-gradient-to-r from-violet-600 to-purple-600 hover:from-violet-700 hover:to-purple-700 text-white gap-2 shadow-lg shadow-violet-500/20">
   717→          <Plus className="h-4 w-4" /> Create Assessment

   719→      </div>
   720→
   721→      {/* Animated Stats Row */}
   722→      <div className="grid grid-cols-2 sm:grid-cols-4 gap-4">
   723→        {[
   724→          { label: 'Total Assessments', value: assessments.length, icon: ListChecks, gradient: 'from-violet-500 to-purple-500' },
   725→          { label: 'Total Submissions', value: totalSubmissions, icon: Send, gradient: 'from-emerald-500 to-teal-500' },
   726→          { label: 'Avg Score', value: avgScore, icon: Target, gradient: 'from-amber-500 to-orange-500', suffix: '%' },
   727→          { label: 'Pass Rate', value: passRate, icon: Award, gradient: 'from-rose-500 to-pink-500', suffix: '%' },
   728→        ].map((stat, i) => (
   729→          <motion.div
   730→            key={stat.label}
   731→            initial={{ opacity: 0, y: 12, scale: 0.95 }}
   732→            animate={{ opacity: 1, y: 0, scale: 1 }}
   733→            transition={{ delay: 0.1 + i * 0.1, type: 'spring', stiffness: 200 }}

   736→              <CardContent className="p-4 flex items-center gap-3">
   737→                <div className={`p-2.5 rounded-xl bg-gradient-to-br ${stat.gradient} shadow-lg group-hover:scale-110 transition-transform duration-300`}>
   738→                  <stat.icon className="h-5 w-5 text-white" />
