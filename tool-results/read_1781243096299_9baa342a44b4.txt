   125→function formatDuration(seconds: number): string {
   126→  const mins = Math.floor(seconds / 60);
   127→  if (mins < 60) return `${mins}m`;
   128→  const hrs = Math.floor(mins / 60);
   129→  const rem = mins % 60;
   130→  return rem > 0 ? `${hrs}h ${rem}m` : `${hrs}h`;
   131→}

   432→function CourseBuilderTab() {
   433→  const [selectedCourseId, setSelectedCourseId] = useState<string>(demoCourses[0].id);
   434→  const [selectedLessonId, setSelectedLessonId] = useState<string | null>(null);
   435→  const [expandedModules, setExpandedModules] = useState<Set<string>>(
   436→    new Set(demoCourses[0].modules?.map((m) => m.id) || [])
   437→  );
   438→
   439→  const selectedCourse = useMemo(
   440→    () => demoCourses.find((c) => c.id === selectedCourseId) || demoCourses[0],
   441→    [selectedCourseId]
   442→  );
   443→
   444→  const selectedLesson = useMemo(() => {
   445→    if (!selectedLessonId) return null;
   446→    for (const mod of selectedCourse.modules || []) {
   447→      const found = mod.lessons?.find((l) => l.id === selectedLessonId);
   448→      if (found) return found;
   449→    }
   450→    return null;
   451→  }, [selectedLessonId, selectedCourse]);
   452→
   453→  const toggleModule = (moduleId: string) => {
   454→    setExpandedModules((prev) => {
   455→      const next = new Set(prev);
   456→      if (next.has(moduleId)) next.delete(moduleId);
   457→      else next.add(moduleId);
   458→      return next;
   459→    });
   460→  };
   461→
   462→  const totalLessons = useMemo(
   463→    () =>
   464→      (selectedCourse.modules || []).reduce(
   465→        (acc, m) => acc + (m.lessons?.length || 0),
   466→        0
   467→      ),
   468→    [selectedCourse]
   469→  );
   470→
   471→  const totalDurationSec = useMemo(
   472→    () =>
   473→      (selectedCourse.modules || []).reduce(
   474→        (acc, m) =>
   475→          acc +
   476→          (m.lessons || []).reduce(
   477→            (a, l) => a + (l.videoDuration || 0),
   478→            0
   479→          ),
   480→        0
   481→      ),
   482→    [selectedCourse]
   483→  );
   484→
   485→  return (
   486→    <div className="space-y-4">
   487→      {/* Course selector */}
   488→      <div className="flex items-center gap-3">
   489→        <Label className="text-sm font-medium shrink-0">Course:</Label>
   490→        <Select value={selectedCourseId} onValueChange={(v) => { setSelectedCourseId(v); setSelectedLessonId(null); }}>
   491→          <SelectTrigger className="w-full sm:w-[320px]">
   492→            <SelectValue />
   493→          </SelectTrigger>
   494→          <SelectContent>
   495→            {demoCourses.map((c) => (
   496→              <SelectItem key={c.id} value={c.id}>
   497→                {c.title}
   498→              </SelectItem>
   499→            ))}
   500→          </SelectContent>
   501→        </Select>
   502→      </div>
   503→
   504→      {/* Two-panel layout */}
   505→      <div className="grid grid-cols-1 lg:grid-cols-[340px_1fr] gap-4 min-h-[600px]">
   506→        {/* Left panel: Module & Lesson Tree */}
   507→        <Card className="overflow-hidden">
   508→          <div className="p-4 border-b bg-muted/30">
   509→            <div className="flex items-center justify-between">
   510→              <h3 className="text-sm font-semibold text-foreground">Curriculum</h3>
   511→              <div className="flex items-center gap-1">
   512→                <Button variant="ghost" size="sm" className="h-7 text-xs gap-1 text-emerald-600 dark:text-emerald-400 hover:text-emerald-700 dark:hover:text-emerald-300">
   513→                  <Plus className="h-3 w-3" /> Module
   514→                </Button>
   515→              </div>
   516→            </div>
   517→            <div className="flex items-center gap-3 mt-2 text-xs text-muted-foreground">
   518→              <span>{(selectedCourse.modules || []).length} modules</span>
   519→              <span>{totalLessons} lessons</span>
   520→              <span>{formatDuration(totalDurationSec)}</span>
   521→            </div>
   522→          </div>
   523→
   524→          <ScrollArea className="h-[540px]">
   525→            <div className="p-2">
   526→              {(selectedCourse.modules || []).length === 0 ? (
   527→                <div className="flex flex-col items-center justify-center py-12 text-muted-foreground">
   528→                  <Layers className="h-8 w-8 mb-2 opacity-40" />
   529→                  <p className="text-sm">No modules yet</p>
   530→                  <p className="text-xs">Add a module to start building</p>
   531→                </div>
   532→              ) : (
   533→                (selectedCourse.modules || [])
   534→                  .sort((a, b) => a.orderIndex - b.orderIndex)
   535→                  .map((mod) => (
   536→                    <ModuleTreeItem
   537→                      key={mod.id}
   538→                      module={mod}
   539→                      isExpanded={expandedModules.has(mod.id)}
   540→                      onToggle={() => toggleModule(mod.id)}
   541→                      selectedLessonId={selectedLessonId}
   542→                      onSelectLesson={setSelectedLessonId}
   543→                    />
   544→                  ))
