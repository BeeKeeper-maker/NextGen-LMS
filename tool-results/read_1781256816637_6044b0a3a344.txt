   209→function formatDuration(seconds: number): string {
   210→  const mins = Math.floor(seconds / 60);
   211→  if (mins < 60) return `${mins}m`;
   212→  const hrs = Math.floor(mins / 60);
   213→  const rem = mins % 60;
   214→  return rem > 0 ? `${hrs}h ${rem}m` : `${hrs}h`;
   215→}

   239→function CourseCatalogTab() {
   240→  const { courses } = useCoursesData();
   241→  const [search, setSearch] = useState('');
   242→  const [categoryFilter, setCategoryFilter] = useState('all');
   243→  const [levelFilter, setLevelFilter] = useState('all');
   244→
   245→  const categories = useMemo(
   246→    () => ['all', ...Array.from(new Set(courses.map((c) => c.category).filter(Boolean)))],
   247→    [courses]
   248→  );
   249→
   250→  const levels = ['all', 'beginner', 'intermediate', 'advanced', 'expert'];
   251→
   252→  const filtered = courses.filter((c) => {
   253→    const matchSearch =
   254→      !search ||
   255→      c.title.toLowerCase().includes(search.toLowerCase()) ||
   256→      c.description?.toLowerCase().includes(search.toLowerCase());
   257→    const matchCategory = categoryFilter === 'all' || c.category === categoryFilter;
   258→    const matchLevel = levelFilter === 'all' || c.level === levelFilter;
   259→    return matchSearch && matchCategory && matchLevel;
   260→  });
   261→
   262→  return (
   263→    <div className="space-y-6">
   264→      {/* Top bar */}

   584→function CourseBuilderTab() {
   585→  const { courses } = useCoursesData();
   586→  const [selectedCourseId, setSelectedCourseId] = useState<string>(courses[0]?.id || '');
   587→  const [selectedLessonId, setSelectedLessonId] = useState<string | null>(null);
   588→  const [expandedModules, setExpandedModules] = useState<Set<string>>(
   589→    new Set(courses[0]?.modules?.map((m) => m.id) || [])
   590→  );
   591→
   592→  const selectedCourse = useMemo(
   593→    () => courses.find((c) => c.id === selectedCourseId) || courses[0],
   594→    [selectedCourseId, courses]
   595→  );
   596→
   597→  const selectedLesson = useMemo(() => {
   598→    if (!selectedLessonId) return null;
   599→    for (const mod of selectedCourse.modules || []) {
   600→      const found = mod.lessons?.find((l) => l.id === selectedLessonId);
   601→      if (found) return found;
   602→    }
   603→    return null;
   604→  }, [selectedLessonId, selectedCourse]);
   605→
   606→  const toggleModule = (moduleId: string) => {
   607→    setExpandedModules((prev) => {
   608→      const next = new Set(prev);
   609→      if (next.has(moduleId)) next.delete(moduleId);
   610→      else next.add(moduleId);
   611→      return next;
   612→    });
   613→  };
   614→
   615→  const totalLessons = useMemo(
   616→    () =>
   617→      (selectedCourse.modules || []).reduce(
   618→        (acc, m) => acc + (m.lessons?.length || 0),
   619→        0
   620→      ),
   621→    [selectedCourse]
   622→  );
   623→
   624→  const totalDurationSec = useMemo(
   625→    () =>
   626→      (selectedCourse.modules || []).reduce(
   627→        (acc, m) =>
   628→          acc +
   629→          (m.lessons || []).reduce(
   630→            (a, l) => a + (l.videoDuration || 0),
   631→            0
   632→          ),
   633→        0
   634→      ),
   635→    [selectedCourse]
   636→  );
   637→
   638→  return (
   639→    <div className="space-y-4">
   640→      {/* Course selector */}
   641→      <div className="flex items-center gap-3">
   642→        <Label className="text-sm font-medium shrink-0">Course:</Label>
   643→        <Select value={selectedCourseId} onValueChange={(v) => { setSelectedCourseId(v); setSelectedLessonId(null); }}>
   644→          <SelectTrigger className="w-full sm:w-[320px]">
   645→            <SelectValue />
   646→          </SelectTrigger>
   647→          <SelectContent>
   648→            {courses.map((c) => (
   649→              <SelectItem key={c.id} value={c.id}>
   650→                {c.title}
   651→              </SelectItem>
   652→            ))}
   653→          </SelectContent>
   654→        </Select>
   655→      </div>
   656→
   657→      {/* Two-panel layout */}
   658→      <div className="grid grid-cols-1 lg:grid-cols-[340px_1fr] gap-4 min-h-[600px]">
   659→        {/* Left panel: Module & Lesson Tree */}
   660→        <Card className="overflow-hidden">
   661→          <div className="p-4 border-b bg-muted/30">
   662→            <div className="flex items-center justify-between">
   663→              <h3 className="text-sm font-semibold text-foreground">Curriculum</h3>
   664→              <div className="flex items-center gap-1">
   665→                <Button variant="ghost" size="sm" className="h-7 text-xs gap-1 text-emerald-600 dark:text-emerald-400 hover:text-emerald-700 dark:hover:text-emerald-300">
   666→                  <Plus className="h-3 w-3" /> Module
   667→                </Button>
   668→              </div>
   669→            </div>
   670→            <div className="flex items-center gap-3 mt-2 text-xs text-muted-foreground">
   671→              <span>{(selectedCourse.modules || []).length} modules</span>
   672→              <span>{totalLessons} lessons</span>
   673→              <span>{formatDuration(totalDurationSec)}</span>
   674→            </div>
   675→          </div>
   676→
   677→          <ScrollArea className="h-[540px]">
   678→            <div className="p-2">
   679→              {(selectedCourse.modules || []).length === 0 ? (
   680→                <div className="flex flex-col items-center justify-center py-12 text-muted-foreground">
   681→                  <Layers className="h-8 w-8 mb-2 opacity-40" />
   682→                  <p className="text-sm">No modules yet</p>

   691→                      module={mod}
   692→                      isExpanded={expandedModules.has(mod.id)}
   693→                      onToggle={() => toggleModule(mod.id)}

   697→                  ))
   698→              )}
   699→            </div>
   700→          </ScrollArea>
   701→        </Card>
