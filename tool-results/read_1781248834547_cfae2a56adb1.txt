   110→// ─── Format duration ───────────────────────────────────────
   111→function formatDuration(seconds?: number): string {
   112→  if (!seconds) return '';
   113→  const mins = Math.floor(seconds / 60);
   114→  if (mins < 60) return `${mins} min`;
   115→  const hours = Math.floor(mins / 60);
   116→  const remainMins = mins % 60;
   117→  return `${hours}h ${remainMins}m`;
   118→}

   723→// ─── Time ago helper ──────────────────────────────────────
   724→function timeAgo(dateStr: string): string {
   725→  const seconds = Math.floor((Date.now() - new Date(dateStr).getTime()) / 1000);
   726→  if (seconds < 60) return 'just now';
   727→  const minutes = Math.floor(seconds / 60);
   728→  if (minutes < 60) return `${minutes}m ago`;
   729→  const hours = Math.floor(minutes / 60);
   730→  if (hours < 24) return `${hours}h ago`;
   731→  const days = Math.floor(hours / 24);
   732→  if (days < 30) return `${days}d ago`;
   733→  const months = Math.floor(days / 30);
   734→  return `${months}mo ago`;
   735→}
