   478→// ─── Trend Line Data (Linear Regression) ─────────────────────
   479→function computeLinearRegression(data: { duration: number; completion: number }[]): { x1: number; y1: number; x2: number; y2: number } {
   480→  const n = data.length;
   481→  let sumX = 0, sumY = 0, sumXY = 0, sumXX = 0;
   482→  data.forEach(d => {
   483→    sumX += d.duration;
   484→    sumY += d.completion;
   485→    sumXY += d.duration * d.completion;
   486→    sumXX += d.duration * d.duration;
   487→  });
   488→  const slope = (n * sumXY - sumX * sumY) / (n * sumXX - sumX * sumX);
   489→  const intercept = (sumY - slope * sumX) / n;
   490→  const minX = Math.min(...data.map(d => d.duration)) - 2;
   491→  const maxX = Math.max(...data.map(d => d.duration)) + 2;
   492→  return {
   493→    x1: minX,
   494→    y1: Math.max(0, Math.min(100, slope * minX + intercept)),
   495→    x2: maxX,
   496→    y2: Math.max(0, Math.min(100, slope * maxX + intercept)),
   497→  };
   498→}

   565→// ─── Animated Number Counter ────────────────────────────────
   566→function AnimatedCounter({ value, prefix = '', suffix = '', duration = 1500 }: {
   567→  value: number;
   568→  prefix?: string;
   569→  suffix?: string;
   570→  duration?: number;
   571→}) {
   572→  const [displayed, setDisplayed] = useState(0);
   573→  const startTime = useRef<number | null>(null);
   574→  const frameRef = useRef<number>(0);
   575→  const animateRef = useRef<((timestamp: number) => void) | null>(null);
   576→
   577→  useEffect(() => {
   578→    animateRef.current = (timestamp: number) => {
   579→      if (!startTime.current) startTime.current = timestamp;
   580→      const progress = Math.min((timestamp - startTime.current) / duration, 1);
   581→      const eased = 1 - Math.pow(1 - progress, 3);
   582→      setDisplayed(Math.floor(eased * value));
   583→      if (progress < 1) {
   584→        frameRef.current = requestAnimationFrame((ts) => animateRef.current?.(ts));
   585→      }
   586→    };
   587→    startTime.current = null;
   588→    frameRef.current = requestAnimationFrame((ts) => animateRef.current?.(ts));
   589→    return () => cancelAnimationFrame(frameRef.current);
   590→  }, [value, duration]);
   591→
   592→  return (
   593→    <span>
   594→      {prefix}{displayed.toLocaleString()}{suffix}
   595→    </span>
   596→  );
   597→}
