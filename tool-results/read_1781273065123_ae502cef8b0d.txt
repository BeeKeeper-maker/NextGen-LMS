    84→    84→  const { currentTenant } = useAppStore();

    94→    94→    email: award.user?.email || '',

    96→    96→    issuedDate: new Date(award.issuedAt).toISOString().split('T')[0],

   100→   100→  // Issue state
   101→   101→  const [selectedCourseId, setSelectedCourseId] = useState('');
   102→   102→  const [selectedTemplateId, setSelectedTemplateId] = useState('');
   103→   103→  const [recipientCriteria, setRecipientCriteria] = useState<RecipientCriteria>('all_completers');
   104→   104→  const [scoreThreshold, setScoreThreshold] = useState('70');
   105→   105→  const [sendNotification, setSendNotification] = useState(true);
   106→   106→  const [issueStep, setIssueStep] = useState<IssueStep>('configure');

   118→   118→  const [actionComplete, setActionComplete] = useState('');

   127→   127→    const completionRate = selectedCourse.completionRate || 0;
   128→   128→    const completers = Math.round(enrollmentCount * (completionRate / 100));
   129→   129→    if (recipientCriteria === 'all_completers') return completers;
   130→   130→    if (recipientCriteria === 'score_threshold') return Math.round(completers * 0.65);

   140→   140→        cert.email.toLowerCase().includes(searchQuery.toLowerCase()) ||
   141→   141→        cert.courseName.toLowerCase().includes(searchQuery.toLowerCase());
   142→   142→      const matchesStatus = statusFilter === 'all' || cert.status === statusFilter;
   143→   143→      return matchesSearch && matchesStatus;
   144→   144→    });
   145→   145→  }, [searchQuery, statusFilter]);

   150→   150→      if (next.has(id)) next.delete(id);
   151→   151→      else next.add(id);

   158→   158→      setSelectedCertIds(new Set());

   175→   175→        setIssuedCount(Math.round(total * 0.95));
   176→   176→        setIssueFailCount(total - Math.round(total * 0.95));

   180→   180→  }, [estimatedCount]);

   184→   184→    setSelectedTemplateId('');

   187→   187→    setIssueStep('configure');
   188→   188→    setIssueProgress(0);
   189→   189→  }, []);
   190→   190→

   193→   193→    setActionComplete('');

   195→   195→      setActionProgress(false);
   196→   196→      setActionComplete(action);
   197→   197→      setSelectedCertIds(new Set());

   200→   200→  }, []);
   201→   201→

   203→   203→    const headers = 'Name,Email,Course,Issued Date,Verification Code,Status\n';

   205→   205→      .map((c) => `${c.userName},${c.email},${c.courseName},${c.issuedDate},${c.verificationCode},${c.status}`)
   206→   206→      .join('\n');

   208→   208→    const url = URL.createObjectURL(blob);
   209→   209→    const a = document.createElement('a');
   210→   210→    a.href = url;
   211→   211→    a.download = 'certificates_export.csv';
   212→   212→    a.click();
   213→   213→    URL.revokeObjectURL(url);
   214→   214→  }, [filteredCerts]);
   215→   215→
   216→   216→  return (
   217→   217→    <div className="space-y-4">
   218→   218→      <Tabs value={subTab} onValueChange={(v) => setSubTab(v as CertSubTab)}>

   220→   220→          <TabsTrigger value="issue" className="gap-1.5">
   221→   221→            <Award className="h-3.5 w-3.5" />
   222→   222→            Issue Certificates
   223→   223→          </TabsTrigger>
   224→   224→          <TabsTrigger value="manage" className="gap-1.5">
   225→   225→            <FileText className="h-3.5 w-3.5" />
   226→   226→            Manage Certificates

   229→   229→
   230→   230→        {/* ─── ISSUE TAB ───────────────────────────────── */}

   234→   234→              <motion.div
   235→   235→                key="configure"
   236→   236→                initial={{ opacity: 0, x: 20 }}
   237→   237→                animate={{ opacity: 1, x: 0 }}

   239→   239→                className="space-y-5"
   240→   240→              >
   241→   241→                {/* Course Selector */}
   242→   242→                <div className="space-y-2">
   243→   243→                  <Label className="text-sm font-semibold flex items-center gap-2">
   244→   244→                    <FileText className="h-4 w-4 text-emerald-500" />
   245→   245→                    Select Course
   246→   246→                  </Label>
   247→   247→                  <Select value={selectedCourseId} onValueChange={setSelectedCourseId}>
   248→   248→                    <SelectTrigger>
   249→   249→                      <SelectValue placeholder="Choose a course..." />
   250→   250→                    </SelectTrigger>
   251→   251→                    <SelectContent>
   252→   252→                      {demoCourses.map((course: any) => (
   253→   253→                        <SelectItem key={course.id} value={course.id}>
   254→   254→                          <div className="flex items-center gap-2">
   255→   255→                            <span>{course.title}</span>
   256→   256→                            <Badge variant="secondary" className="text-[10px] px-1.5 py-0">
   257→   257→                              {course.completionRate || 0}% complete
   258→   258→                            </Badge>
   259→   259→                          </div>
   260→   260→                        </SelectItem>
   261→   261→                      ))}
   262→   262→                    </SelectContent>
   263→   263→                  </Select>
   264→   264→                </div>
   265→   265→
   266→   266→                {/* Template Selector */}

   269→   269→                    <Award className="h-4 w-4 text-emerald-500" />

   276→   276→                        onClick={() => setSelectedTemplateId(tpl.id)}

   278→   278→                          selectedTemplateId === tpl.id
   279→   279→                            ? 'border-emerald-400 bg-emerald-50/50 dark:bg-emerald-950/20 shadow-sm'

   281→   281→                        }`}
   282→   282→                      >

   290→   290→                <Separator />
   291→   291→
   292→   292→                {/* Recipient Criteria */}
   293→   293→                <div className="space-y-3">
   294→   294→                  <Label className="text-sm font-semibold flex items-center gap-2">
   295→   295→                    <Users className="h-4 w-4 text-emerald-500" />
   296→   296→                    Recipient Criteria

   300→   300→                      { key: 'all_completers' as RecipientCriteria, label: 'All Completers', desc: 'Everyone who completed the course' },
   301→   301→                      { key: 'score_threshold' as RecipientCriteria, label: 'Score > X%', desc: 'Only those above a score threshold' },
   302→   302→                      { key: 'custom' as RecipientCriteria, label: 'Custom Selection', desc: 'Manually select recipients' },
   303→   303→                    ].map((opt) => (
   304→   304→                      <button
   305→   305→                        key={opt.key}
   306→   306→                        onClick={() => setRecipientCriteria(opt.key)}
   307→   307→                        className={`flex flex-col items-start gap-1 rounded-lg border p-3 text-left transition-all duration-200 ${
   308→   308→                          recipientCriteria === opt.key
   309→   309→                            ? 'border-emerald-400 bg-emerald-50/50 dark:bg-emerald-950/20 shadow-sm'
   310→   310→                            : 'hover:border-emerald-300'
   311→   311→                        }`}
   312→   312→                      >
   313→   313→                        <p className="text-sm font-medium">{opt.label}</p>
   314→   314→                        <p className="text-xs text-muted-foreground">{opt.desc}</p>
   315→   315→                      </button>
   316→   316→                    ))}
   317→   317→                  </div>
   318→   318→
   319→   319→                  {recipientCriteria === 'score_threshold' && (
   320→   320→                    <div className="flex items-center gap-2">
   321→   321→                      <Label className="text-sm">Minimum Score:</Label>
   322→   322→                      <Input

   324→   324→                        value={scoreThreshold}
   325→   325→                        onChange={(e) => setScoreThreshold(e.target.value)}
   326→   326→                        className="w-20 h-8 text-sm"
   327→   327→                        min={0}
   328→   328→                        max={100}
   329→   329→                      />
   330→   330→                      <span className="text-sm text-muted-foreground">%</span>
   331→   331→                    </div>
   332→   332→                  )}
   333→   333→
   334→   334→                  {/* Count Preview */}
   335→   335→                  {selectedCourseId && (
   336→   336→                    <motion.div
   337→   337→                      initial={{ opacity: 0, y: 5 }}
   338→   338→                      animate={{ opacity: 1, y: 0 }}
   339→   339→                      className="rounded-lg border bg-emerald-50/50 dark:bg-emerald-950/20 p-3 flex items-center gap-3"
   340→   340→                    >
   341→   341→                      <Award className="h-8 w-8 text-emerald-500" />
   342→   342→                      <div>
   343→   343→                        <p className="text-lg font-bold text-emerald-600 dark:text-emerald-400">
   344→   344→                          {estimatedCount} certificates
   345→   345→                        </p>
   346→   346→                        <p className="text-xs text-muted-foreground">
   347→   347→                          will be issued based on your criteria
   348→   348→                        </p>
   349→   349→                      </div>
   350→   350→                    </motion.div>
   351→   351→                  )}
   352→   352→                </div>
   353→   353→
   354→   354→                <Separator />

   357→   357→                <div className="flex items-center justify-between rounded-lg border p-3">
   358→   358→                  <div className="flex items-center gap-3">
   359→   359→                    <Mail className="h-4 w-4 text-emerald-500" />

   361→   361→                      <p className="text-sm font-medium">Send Notification Email</p>

   364→   364→                  </div>
   365→   365→                  <Switch checked={sendNotification} onCheckedChange={setSendNotification} />
   366→   366→                </div>

   368→   368→                {/* Action */}

   371→   371→                    onClick={() => setShowIssueConfirm(true)}
   372→   372→                    disabled={!selectedCourseId || !selectedTemplateId || estimatedCount === 0}
   373→   373→                    className="bg-gradient-to-r from-emerald-500 to-emerald-600 hover:from-emerald-600 hover:to-emerald-700 text-white shadow-lg"
   374→   374→                  >
   375→   375→                    <Award className="h-4 w-4 mr-1.5" />
   376→   376→                    Issue {estimatedCount} Certificates
   377→   377→                  </Button>
   378→   378→                </div>
   379→   379→              </motion.div>
   380→   380→            )}
   381→   381→
   382→   382→            {issueStep === 'processing' && (
   383→   383→              <motion.div
   384→   384→                key="processing"
   385→   385→                initial={{ opacity: 0, scale: 0.95 }}
   386→   386→                animate={{ opacity: 1, scale: 1 }}
   387→   387→                className="flex flex-col items-center py-12 space-y-6"
   388→   388→              >
   389→   389→                <motion.div
   390→   390→                  animate={{ rotate: 360 }}
   391→   391→                  transition={{ duration: 2, repeat: Infinity, ease: 'linear' }}
   392→   392→                >
   393→   393→                  <Loader2 className="h-12 w-12 text-emerald-500" />
   394→   394→                </motion.div>
   395→   395→                <div className="text-center">
   396→   396→                  <h3 className="text-lg font-semibold">Issuing Certificates...</h3>

   399→   399→                  </p>
   400→   400→                </div>
   401→   401→                <div className="w-full max-w-sm">
   402→   402→                  <Progress value={issueProgress} className="h-3" />

   407→   407→
   408→   408→            {issueStep === 'complete' && (
   409→   409→              <motion.div
   410→   410→                key="complete"

   418→   418→                  transition={{ type: 'spring', stiffness: 200, delay: 0.2 }}
   419→   419→                >
   420→   420→                  <CheckCircle2 className="h-16 w-16 text-emerald-500" />
   421→   421→                </motion.div>
   422→   422→                <div className="text-center">
   423→   423→                  <h3 className="text-lg font-semibold">Certificates Issued!</h3>
   424→   424→                  <p className="text-sm text-muted-foreground mt-1">
   425→   425→                    {sendNotification ? 'Notifications have been sent.' : 'Notifications were skipped.'}
   426→   426→                  </p>
   427→   427→                </div>
   428→   428→                <div className="grid grid-cols-2 gap-4 w-full max-w-xs">
   429→   429→                  <div className="rounded-xl border bg-emerald-50 dark:bg-emerald-950/20 p-4 text-center">

   432→   432→                      <CheckCircle2 className="h-3 w-3 text-emerald-500" />
   433→   433→                      Issued

   436→   436→                  {issueFailCount > 0 && (

   439→   439→                      <p className="text-xs text-muted-foreground mt-1 flex items-center justify-center gap-1">

   443→   443→                    </div>
