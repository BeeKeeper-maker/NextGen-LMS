   400тЖТ   400тЖТfunction AnimatedCounter({ value, duration = 1500 }: { value: string; duration?: number }) {
   401тЖТ   401тЖТ  const [displayed, setDisplayed] = useState('0');
   402тЖТ   402тЖТ  const hasAnimated = useRef(false);
   403тЖТ   403тЖТ
   404тЖТ   404тЖТ  useEffect(() => {
   405тЖТ   405тЖТ    if (hasAnimated.current) return;
   406тЖТ   406тЖТ    hasAnimated.current = true;
   407тЖТ   407тЖТ
   408тЖТ   408тЖТ    const numericMatch = value.match(/[\d,.]+/);
   409тЖТ   409тЖТ    if (!numericMatch) {
   410тЖТ   410тЖТ      return;
   411тЖТ   411тЖТ    }
   412тЖТ   412тЖТ
   413тЖТ   413тЖТ    const numericStr = numericMatch[0];
   414тЖТ   414тЖТ    const prefix = value.slice(0, value.indexOf(numericStr));
   415тЖТ   415тЖТ    const suffix = value.slice(value.indexOf(numericStr) + numericStr.length);
   416тЖТ   416тЖТ    const cleanNumber = parseFloat(numericStr.replace(/,/g, ''));
   417тЖТ   417тЖТ    const hasDecimals = numericStr.includes('.');
   418тЖТ   418тЖТ    const decimalPlaces = hasDecimals ? numericStr.split('.')[1].length : 0;
   419тЖТ   419тЖТ
   420тЖТ   420тЖТ    const startTime = Date.now();
   421тЖТ   421тЖТ    const animate = () => {
   422тЖТ   422тЖТ      const elapsed = Date.now() - startTime;
   423тЖТ   423тЖТ      const progress = Math.min(elapsed / duration, 1);
   424тЖТ   424тЖТ      const eased = 1 - Math.pow(1 - progress, 3);
   425тЖТ   425тЖТ      const current = cleanNumber * eased;
   426тЖТ   426тЖТ
   427тЖТ   427тЖТ      let formatted: string;
   428тЖТ   428тЖТ      if (hasDecimals) {
   429тЖТ   429тЖТ        formatted = current.toFixed(decimalPlaces);
   430тЖТ   430тЖТ      } else {
   431тЖТ   431тЖТ        formatted = Math.round(current).toLocaleString('en-US');
   432тЖТ   432тЖТ      }
   433тЖТ   433тЖТ
   434тЖТ   434тЖТ      setDisplayed(`${prefix}${formatted}${suffix}`);
   435тЖТ   435тЖТ
   436тЖТ   436тЖТ      if (progress < 1) {
   437тЖТ   437тЖТ        requestAnimationFrame(animate);
   438тЖТ   438тЖТ      }
   439тЖТ   439тЖТ    };
   440тЖТ   440тЖТ
   441тЖТ   441тЖТ    requestAnimationFrame(animate);
   442тЖТ   442тЖТ  }, [value, duration]);
   443тЖТ   443тЖТ
   444тЖТ   444тЖТ  const numericMatch = value.match(/[\d,.]+/);
   445тЖТ   445тЖТ  if (!numericMatch) {
   446тЖТ   446тЖТ    return <span>{value}</span>;
   447тЖТ   447тЖТ  }
   448тЖТ   448тЖТ
   449тЖТ   449тЖТ  return <span>{displayed}</span>;
   450тЖТ   450тЖТ}
