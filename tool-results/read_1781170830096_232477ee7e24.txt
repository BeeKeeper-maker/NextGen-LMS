    58→// ─── Format duration ───────────────────────────────────────
    59→function formatDuration(seconds?: number): string {
    60→  if (!seconds) return '';
    61→  const mins = Math.floor(seconds / 60);
    62→  if (mins < 60) return `${mins} min`;
    63→  const hours = Math.floor(mins / 60);
    64→  const remainMins = mins % 60;
    65→  return `${hours}h ${remainMins}m`;
    66→}
