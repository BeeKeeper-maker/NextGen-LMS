   137→function AnimatedCounter({ target, duration = 1.5 }: { target: number; duration?: number }) {
   138→  const [count, setCount] = useState(0);
   139→
   140→  useEffect(() => {
   141→    let startTime: number;
   142→    let rafId: number;
   143→    const animate = (timestamp: number) => {
   144→      if (!startTime) startTime = timestamp;
   145→      const progress = Math.min((timestamp - startTime) / (duration * 1000), 1);
   146→      setCount(Math.floor(progress * target));
   147→      if (progress < 1) {
   148→        rafId = requestAnimationFrame(animate);
   149→      }
   150→    };
   151→    rafId = requestAnimationFrame(animate);
   152→    return () => cancelAnimationFrame(rafId);
   153→  }, [target, duration]);
   154→
   155→  return <>{count.toLocaleString()}</>;
   156→}

   162→function ProgressRing({
   163→  size = 120,
   164→  strokeWidth = 8,
   165→  progress = 0,
   166→  gradientId = 'ring-gradient',
   167→  animate = true,
   168→  label,
   169→  sublabel,
   170→}: {
   171→  size?: number;
   172→  strokeWidth?: number;
   173→  progress?: number;
   174→  gradientId?: string;
   175→  animate?: boolean;
   176→  label?: string;
   177→  sublabel?: string;
   178→}) {
   179→  const [animatedProgress, setAnimatedProgress] = useState(animate ? 0 : progress);
   180→  const radius = (size - strokeWidth) / 2;
   181→  const circumference = 2 * Math.PI * radius;
   182→  const offset = circumference - (animatedProgress / 100) * circumference;
   183→
   184→  useEffect(() => {
   185→    if (!animate) return;
   186→    const timer = setTimeout(() => setAnimatedProgress(progress), 200);
   187→    return () => clearTimeout(timer);
   188→  }, [progress, animate]);
   189→
   190→  return (
   191→    <div className="relative inline-flex items-center justify-center" style={{ width: size, height: size }}>
   192→      <svg width={size} height={size} className="-rotate-90">
   193→        <defs>
   194→          <linearGradient id={gradientId} x1="0%" y1="0%" x2="100%" y2="100%">
   195→            <stop offset="0%" stopColor="#10b981" />
   196→            <stop offset="50%" stopColor="#14b8a6" />
   197→            <stop offset="100%" stopColor="#06b6d4" />
   198→          </linearGradient>
   199→        </defs>
   200→        {/* Background ring */}
   201→        <circle
   202→          cx={size / 2}
   203→          cy={size / 2}
   204→          r={radius}
   205→          fill="none"
   206→          stroke="currentColor"
   207→          strokeWidth={strokeWidth}
   208→          className="text-slate-200 dark:text-slate-700"
   209→        />
   210→        {/* Progress ring */}
   211→        <motion.circle
   212→          cx={size / 2}
   213→          cy={size / 2}
   214→          r={radius}
   215→          fill="none"
   216→          stroke={`url(#${gradientId})`}
   217→          strokeWidth={strokeWidth}
   218→          strokeLinecap="round"
   219→          strokeDasharray={circumference}
   220→          initial={{ strokeDashoffset: circumference }}
   221→          animate={{ strokeDashoffset: offset }}
   222→          transition={{ duration: 1.5, ease: 'easeOut', delay: 0.3 }}
   223→        />
   224→      </svg>
   225→      {/* Center text */}
   226→      <div className="absolute inset-0 flex flex-col items-center justify-center">
   227→        {label && (
   228→          <span className="text-lg font-bold text-slate-900 dark:text-slate-50">
   229→            {Math.round(animatedProgress)}%
   230→          </span>
   231→        )}
   232→        {sublabel && (
   233→          <span className="text-[10px] text-muted-foreground leading-tight text-center">
   234→            {sublabel}
   235→          </span>
   236→        )}
   237→      </div>
   238→    </div>
   239→  );
   240→}

   496→            </div>

   498→        )}
   499→      </AnimatePresence>
   500→    </motion.div>

   508→function MilestoneTracker({ completedCount }: { completedCount: number }) {

   515→          const isUpcoming = i > completedCount;
   516→          const Icon = milestone.icon;
   517→
   518→          return (
   519→            <div key={milestone.id} className="flex items-center flex-1">

   522→                <motion.div

   524→                  animate={{ scale: 1 }}
   525→                  transition={{ duration: 0.4, delay: 0.1 * i }}
   526→                  className={cn(
   527→                    'relative flex h-10 w-10 sm:h-12 sm:w-12 items-center justify-center rounded-full border-2 transition-all',
   528→                    isCompleted && 'border-emerald-500 bg-emerald-500 text-white',
   529→                    isCurrent && 'border-amber-500 bg-amber-50 dark:bg-amber-950/30 text-amber-600 dark:text-amber-400',
   530→                    isUpcoming && 'border-slate-300 dark:border-slate-700 bg-white dark:bg-slate-900 text-slate-400'
   531→                  )}

   535→                  ) : (
   536→                    <Icon className="h-4 w-4 sm:h-5 sm:w-5" />
   537→                  )}
   538→
   539→                  {/* Pulsing ring for current */}
   540→                  {isCurrent && (
