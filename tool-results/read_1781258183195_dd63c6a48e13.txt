   258→// ============================================================
   259→// Typing Animation Component
   260→// ============================================================
   261→function TypingText({ text, speed = 35, delay = 0 }: { text: string; speed?: number; delay?: number }) {
   262→  const [displayed, setDisplayed] = useState('');
   263→  const [started, setStarted] = useState(false);
   264→
   265→  useEffect(() => {
   266→    const timer = setTimeout(() => setStarted(true), delay);
   267→    return () => clearTimeout(timer);
   268→  }, [delay]);
   269→
   270→  useEffect(() => {
   271→    if (!started) return;
   272→    if (displayed.length < text.length) {
   273→      const timer = setTimeout(() => {
   274→        setDisplayed(text.slice(0, displayed.length + 1));
   275→      }, speed);
   276→      return () => clearTimeout(timer);
   277→    }
   278→  }, [displayed, text, speed, started]);
   279→
   280→  return (
   281→    <span>
   282→      {displayed}
   283→      {displayed.length < text.length && started && (
   284→        <span className="inline-block w-0.5 h-5 bg-emerald-500 ml-0.5 animate-pulse align-middle" />
   285→      )}
   286→    </span>
   287→  );
   288→}

   293→function RotatingText({ phrases, interval = 3000 }: { phrases: string[]; interval?: number }) {
   294→  const [index, setIndex] = useState(0);
   295→  const [direction, setDirection] = useState(1);
   296→
   297→  useEffect(() => {
   298→    const timer = setInterval(() => {
   299→      setDirection(1);
   300→      setIndex((prev) => (prev + 1) % phrases.length);
   301→    }, interval);
   302→    return () => clearInterval(timer);
   303→  }, [phrases.length, interval]);
   304→
   305→  return (
   306→    <AnimatePresence mode="wait" custom={direction}>
   307→      <motion.span
   308→        key={phrases[index]}
   309→        custom={direction}
   310→        initial={{ y: 20, opacity: 0 }}
   311→        animate={{ y: 0, opacity: 1 }}
   312→        exit={{ y: -20, opacity: 0 }}
   313→        transition={{ duration: 0.4, ease: 'easeInOut' }}
   314→        className="inline-block bg-gradient-to-r from-emerald-600 via-emerald-500 to-teal-500 bg-clip-text text-transparent font-semibold"
   315→      >
   316→        {phrases[index]}
   317→      </motion.span>
   318→    </AnimatePresence>
   319→  );
   320→}

   359→const constellationData = Array.from({ length: 18 }, (_, i) => ({
   360→  id: i,
   361→  x: ((i * 17 + 5) % 95) + 2,
   362→  y: ((i * 23 + 11) % 90) + 5,
   363→  size: ((i * 3 + 2) % 3) + 2,
   364→  duration: 8 + ((i * 7 + 3) % 8),
   365→  delay: ((i * 5 + 1) % 6),
   366→  driftX: ((i * 13 % 7) - 3) * 8,
   367→  driftY: ((i * 11 % 5) - 2) * 6,
   368→}));
   369→
   370→// Pre-compute constellation lines (deterministic, no state needed)
   371→const constellationLines = constellationData.flatMap((p1, i) =>
   372→  constellationData.slice(i + 1).flatMap((p2) => {
   373→    const dx = p1.x - p2.x;
   374→    const dy = p1.y - p2.y;
   375→    const dist = Math.sqrt(dx * dx + dy * dy);
   376→    if (dist < 25) {
   377→      return [{ x1: p1.x, y1: p1.y, x2: p2.x, y2: p2.y, opacity: Math.max(0, 1 - dist / 25) * 0.15 }];
   378→    }
   379→    return [];
   380→  })
   381→);

   706→// ============================================================
   707→// Animated Counter Hook
   708→// ============================================================
   709→function useAnimatedCounter(target: number, duration: number = 800) {
   710→  const [count, setCount] = useState(0);
   711→  const prevTarget = useRef(0);
   712→
   713→  useEffect(() => {
   714→    if (target === prevTarget.current) return;
   715→    prevTarget.current = target;
   716→
   717→    const start = count;
   718→    const diff = target - start;
   719→    const startTime = performance.now();
   720→
   721→    const animate = (now: number) => {
   722→      const elapsed = now - startTime;
   723→      const progress = Math.min(elapsed / duration, 1);
   724→      // easeOutCubic
   725→      const eased = 1 - Math.pow(1 - progress, 3);
   726→      setCount(Math.round(start + diff * eased));
   727→      if (progress < 1) {
   728→        requestAnimationFrame(animate);
   729→      }
   730→    };
   731→
   732→    requestAnimationFrame(animate);
   733→  }, [target, duration]);
   734→
   735→  return count;
   736→}

  1006→  function StatCounter({ target, suffix, label, gradient, isDecimal = false }: {
  1007→    target: number; suffix: string; label: string; gradient: string; isDecimal?: boolean;

  1009→    const [count, setCount] = useState(0);

  1018→      const startTime = performance.now();

  1021→        const elapsed = now - startTime;
  1022→        const progress = Math.min(elapsed / duration, 1);
  1023→        // easeOutExpo
  1024→        const eased = progress === 1 ? 1 : 1 - Math.pow(2, -10 * progress);
  1025→        const current = target * eased;

  1028→          requestAnimationFrame(animate);

  1829→function AnimatedPrice({ target, isInView }: { target: number; isInView: boolean }) {
  1830→  const [count, setCount] = useState(0);
  1831→  const hasAnimated = useRef(false);
  1832→
  1833→  useEffect(() => {
  1834→    if (!isInView || hasAnimated.current) return;
  1835→    hasAnimated.current = true;
  1836→
  1837→    const duration = 1200;
  1838→    const startTime = performance.now();
  1839→
  1840→    const animate = (now: number) => {
  1841→      const elapsed = now - startTime;
  1842→      const progress = Math.min(elapsed / duration, 1);
  1843→      const eased = 1 - Math.pow(1 - progress, 3);
  1844→      setCount(Math.round(target * eased));
  1845→      if (progress < 1) {
  1846→        requestAnimationFrame(animate);
  1847→      }
  1848→    };
  1849→
  1850→    requestAnimationFrame(animate);
  1851→  }, [target, isInView]);
  1852→
  1853→  return <span>{count}</span>;
  1854→}
