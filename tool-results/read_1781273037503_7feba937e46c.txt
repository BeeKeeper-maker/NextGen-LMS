   262тЖТ   262тЖТfunction TypingText({ text, speed = 35, delay = 0 }: { text: string; speed?: number; delay?: number }) {
   263тЖТ   263тЖТ  const [displayed, setDisplayed] = useState('');
   264тЖТ   264тЖТ  const [started, setStarted] = useState(false);
   265тЖТ   265тЖТ
   266тЖТ   266тЖТ  useEffect(() => {
   267тЖТ   267тЖТ    const timer = setTimeout(() => setStarted(true), delay);
   268тЖТ   268тЖТ    return () => clearTimeout(timer);
   269тЖТ   269тЖТ  }, [delay]);
   270тЖТ   270тЖТ
   271тЖТ   271тЖТ  useEffect(() => {
   272тЖТ   272тЖТ    if (!started) return;
   273тЖТ   273тЖТ    if (displayed.length < text.length) {
   274тЖТ   274тЖТ      const timer = setTimeout(() => {
   275тЖТ   275тЖТ        setDisplayed(text.slice(0, displayed.length + 1));
   276тЖТ   276тЖТ      }, speed);
   277тЖТ   277тЖТ      return () => clearTimeout(timer);
   278тЖТ   278тЖТ    }
   279тЖТ   279тЖТ  }, [displayed, text, speed, started]);
   280тЖТ   280тЖТ
   281тЖТ   281тЖТ  return (
   282тЖТ   282тЖТ    <span>
   283тЖТ   283тЖТ      {displayed}
   284тЖТ   284тЖТ      {displayed.length < text.length && started && (
   285тЖТ   285тЖТ        <span className="inline-block w-0.5 h-5 bg-emerald-500 ml-0.5 animate-pulse align-middle" />
   286тЖТ   286тЖТ      )}

   289тЖТ   289тЖТ}

   708тЖТ   708тЖТ// Animated Counter Hook
   709тЖТ   709тЖТ// ============================================================
   710тЖТ   710тЖТfunction useAnimatedCounter(target: number, duration: number = 800) {
   711тЖТ   711тЖТ  const [count, setCount] = useState(0);
   712тЖТ   712тЖТ  const prevTarget = useRef(0);
   713тЖТ   713тЖТ
   714тЖТ   714тЖТ  useEffect(() => {
   715тЖТ   715тЖТ    if (target === prevTarget.current) return;
   716тЖТ   716тЖТ    prevTarget.current = target;
   717тЖТ   717тЖТ
   718тЖТ   718тЖТ    const start = count;
   719тЖТ   719тЖТ    const diff = target - start;
   720тЖТ   720тЖТ    const startTime = performance.now();
   721тЖТ   721тЖТ
   722тЖТ   722тЖТ    const animate = (now: number) => {
   723тЖТ   723тЖТ      const elapsed = now - startTime;
   724тЖТ   724тЖТ      const progress = Math.min(elapsed / duration, 1);
   725тЖТ   725тЖТ      // easeOutCubic
   726тЖТ   726тЖТ      const eased = 1 - Math.pow(1 - progress, 3);
   727тЖТ   727тЖТ      setCount(Math.round(start + diff * eased));
   728тЖТ   728тЖТ      if (progress < 1) {
   729тЖТ   729тЖТ        requestAnimationFrame(animate);
   730тЖТ   730тЖТ      }
   731тЖТ   731тЖТ    };
   732тЖТ   732тЖТ
   733тЖТ   733тЖТ    requestAnimationFrame(animate);
   734тЖТ   734тЖТ  }, [target, duration]);
   735тЖТ   735тЖТ
   736тЖТ   736тЖТ  return count;
   737тЖТ   737тЖТ}
