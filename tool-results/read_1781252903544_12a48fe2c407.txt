   480→// ─── Trend Line Data (Linear Regression) ─────────────────────
   481→function computeLinearRegression(data: { duration: number; completion: number }[]): { x1: number; y1: number; x2: number; y2: number } {
   482→  const n = data.length;
   483→  let sumX = 0, sumY = 0, sumXY = 0, sumXX = 0;
   484→  data.forEach(d => {
   485→    sumX += d.duration;
   486→    sumY += d.completion;
   487→    sumXY += d.duration * d.completion;
   488→    sumXX += d.duration * d.duration;
   489→  });
   490→  const slope = (n * sumXY - sumX * sumY) / (n * sumXX - sumX * sumX);
   491→  const intercept = (sumY - slope * sumX) / n;
   492→  const minX = Math.min(...data.map(d => d.duration)) - 2;
   493→  const maxX = Math.max(...data.map(d => d.duration)) + 2;
   494→  return {
   495→    x1: minX,
   496→    y1: Math.max(0, Math.min(100, slope * minX + intercept)),
   497→    x2: maxX,
   498→    y2: Math.max(0, Math.min(100, slope * maxX + intercept)),
   499→  };
   500→}

   556→// ─── Animated Number Counter ────────────────────────────────
   557→function AnimatedCounter({ value, prefix = '', suffix = '', duration = 1500 }: {
   558→  value: number;
   559→  prefix?: string;
   560→  suffix?: string;
   561→  duration?: number;
   562→}) {
   563→  const [displayed, setDisplayed] = useState(0);
   564→  const startTime = useRef<number | null>(null);
   565→  const frameRef = useRef<number>(0);
   566→  const animateRef = useRef<((timestamp: number) => void) | null>(null);
   567→
   568→  useEffect(() => {
   569→    animateRef.current = (timestamp: number) => {
   570→      if (!startTime.current) startTime.current = timestamp;
   571→      const progress = Math.min((timestamp - startTime.current) / duration, 1);
   572→      const eased = 1 - Math.pow(1 - progress, 3);
   573→      setDisplayed(Math.floor(eased * value));
   574→      if (progress < 1) {
   575→        frameRef.current = requestAnimationFrame((ts) => animateRef.current?.(ts));
   576→      }
   577→    };
   578→    startTime.current = null;
   579→    frameRef.current = requestAnimationFrame((ts) => animateRef.current?.(ts));
   580→    return () => cancelAnimationFrame(frameRef.current);
   581→  }, [value, duration]);
   582→
   583→  return (
   584→    <span>
   585→      {prefix}{displayed.toLocaleString()}{suffix}
   586→    </span>
   587→  );
   588→}
