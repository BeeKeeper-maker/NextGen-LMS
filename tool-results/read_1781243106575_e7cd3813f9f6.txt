   109→// ─── Format duration ───────────────────────────────────────
   110→function formatDuration(seconds?: number): string {
   111→  if (!seconds) return '';
   112→  const mins = Math.floor(seconds / 60);
   113→  if (mins < 60) return `${mins} min`;
   114→  const hours = Math.floor(mins / 60);
   115→  const remainMins = mins % 60;
   116→  return `${hours}h ${remainMins}m`;
   117→}

   722→// ─── Time ago helper ──────────────────────────────────────
   723→function timeAgo(dateStr: string): string {
   724→  const seconds = Math.floor((Date.now() - new Date(dateStr).getTime()) / 1000);
   725→  if (seconds < 60) return 'just now';
   726→  const minutes = Math.floor(seconds / 60);
   727→  if (minutes < 60) return `${minutes}m ago`;
   728→  const hours = Math.floor(minutes / 60);
   729→  if (hours < 24) return `${hours}h ago`;
   730→  const days = Math.floor(hours / 24);
   731→  if (days < 30) return `${days}d ago`;
   732→  const months = Math.floor(days / 30);
   733→  return `${months}mo ago`;
   734→}
