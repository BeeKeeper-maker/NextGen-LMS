   179→// ---- Animated Counter ----
   180→function AnimatedCounter({ value, duration = 1200 }: { value: number; duration?: number }) {
   181→  const [display, setDisplay] = useState(0);
   182→  const startRef = useRef<number | null>(null);
   183→
   184→  useEffect(() => {
   185→    startRef.current = null;
   186→    const step = (timestamp: number) => {
   187→      if (!startRef.current) startRef.current = timestamp;
   188→      const progress = Math.min((timestamp - startRef.current) / duration, 1);
   189→      setDisplay(Math.floor(progress * value));
   190→      if (progress < 1) requestAnimationFrame(step);
   191→    };
   192→    requestAnimationFrame(step);
   193→  }, [value, duration]);
   194→
   195→  return <>{display}</>;
   196→}

   316→// ---- Per-Question Statistics ----
   317→function QuestionStatsPanel({ questions }: { questions: Question[] }) {
   318→  // Generate simulated stats for each question
   319→  const stats = useMemo(() => {
   320→    return questions.map((q, idx) => {
   321→      const seed = q.id.charCodeAt(q.id.length - 1);
   322→      const timesAnswered = 45 + (seed * 13) % 200;
   323→      const correctPercent = q.difficulty === 'easy' ? 70 + (seed % 25) : q.difficulty === 'medium' ? 45 + (seed % 30) : 20 + (seed % 35);
   324→      const avgTime = q.difficulty === 'easy' ? 15 + (seed % 20) : q.difficulty === 'medium' ? 30 + (seed % 40) : 50 + (seed % 60);
   325→      const discriminationIndex = +(0.3 + ((seed * 7) % 60) / 100).toFixed(2);
   326→      const distractorCounts: Record<string, number> = {};
   327→      if (q.type === 'multiple_choice' && q.options) {
   328→        q.options.forEach((_, oi) => {
   329→          if (String(oi) !== q.correctAnswer) {
   330→            distractorCounts[String(oi)] = Math.floor((100 - correctPercent) * (0.2 + ((oi * 13 + seed) % 50) / 100));
   331→          }
   332→        });
   333→      }
   334→      return { ...q, timesAnswered, correctPercent, avgTime, discriminationIndex, distractorCounts, idx };
   335→    });
   336→  }, [questions]);
   337→
   338→  const getColorClass = (pct: number) => {
   339→    if (pct > 80) return 'text-emerald-600 dark:text-emerald-400';
   340→    if (pct >= 50) return 'text-amber-600 dark:text-amber-400';
   341→    return 'text-red-600 dark:text-red-400';
   342→  };
   343→
   344→  const getBgClass = (pct: number) => {
   345→    if (pct > 80) return 'bg-emerald-50 dark:bg-emerald-950/20 border-emerald-200 dark:border-emerald-800';
   346→    if (pct >= 50) return 'bg-amber-50 dark:bg-amber-950/20 border-amber-200 dark:border-amber-800';
   347→    return 'bg-red-50 dark:bg-red-950/20 border-red-200 dark:border-red-800';
   348→  };
   349→
   350→  const getDiscriminationLabel = (di: number) => {
   351→    if (di >= 0.7) return { text: 'Excellent', cls: 'text-emerald-600 dark:text-emerald-400' };
   352→    if (di >= 0.4) return { text: 'Good', cls: 'text-amber-600 dark:text-amber-400' };
   353→    return { text: 'Poor', cls: 'text-red-600 dark:text-red-400' };
   354→  };
   355→
   356→  return (
   357→    <div className="space-y-3 max-h-96 overflow-y-auto pr-1 custom-scrollbar">
   358→      {stats.map((s, i) => {
   359→        const disc = getDiscriminationLabel(s.discriminationIndex);
   360→        return (
   361→          <motion.div
   362→            key={s.id}
   363→            initial={{ opacity: 0, y: 8 }}
   364→            animate={{ opacity: 1, y: 0 }}
   365→            transition={{ delay: i * 0.05 }}
   366→            className={`p-3 rounded-lg border ${getBgClass(s.correctPercent)} transition-all hover:shadow-md`}
   367→          >
   368→            <div className="flex items-start justify-between gap-2">
   369→              <div className="flex-1 min-w-0">
   370→                <div className="flex items-center gap-2 mb-1">
   371→                  <span className="text-xs font-mono text-muted-foreground">Q{s.idx + 1}</span>
   372→                  <QuestionTypeBadge type={s.type} />
   373→                  <DifficultyBadge difficulty={s.difficulty} />
   374→                </div>
   375→                <p className="text-sm font-medium truncate">{s.question}</p>
   376→              </div>
   377→              <div className="text-right shrink-0">
   378→                <div className={`text-lg font-bold ${getColorClass(s.correctPercent)}`}>{s.correctPercent}%</div>
   379→                <div className="text-xs text-muted-foreground">correct</div>
   380→              </div>
   381→            </div>
   382→
   383→            <div className="grid grid-cols-3 gap-3 mt-2 pt-2 border-t border-black/5 dark:border-white/5">
   384→              <div className="text-center">
   385→                <div className="text-sm font-semibold">{s.timesAnswered}</div>
   386→                <div className="text-xs text-muted-foreground flex items-center justify-center gap-1">
   387→                  <Users className="h-3 w-3" /> answered
   388→                </div>
   389→              </div>
   390→              <div className="text-center">
   391→                <div className="text-sm font-semibold">{s.avgTime}s</div>
   392→                <div className="text-xs text-muted-foreground flex items-center justify-center gap-1">
   393→                  <Clock className="h-3 w-3" /> avg time
   394→                </div>
   395→              </div>
   396→              <div className="text-center">
   397→                <div className={`text-sm font-semibold ${disc.cls}`}>{s.discriminationIndex}</div>
   398→                <div className="text-xs text-muted-foreground">{disc.text}</div>
   399→              </div>
   400→            </div>
   401→
   402→            {/* Distractor analysis */}
   403→            {s.type === 'multiple_choice' && s.options && Object.keys(s.distractorCounts).length > 0 && (
   404→              <div className="mt-2 pt-2 border-t border-black/5 dark:border-white/5">
   405→                <div className="text-xs font-medium text-muted-foreground mb-1 flex items-center gap-1">
   406→                  <AlertCircle className="h-3 w-3" /> Distractor Analysis
   407→                </div>
   408→                <div className="space-y-1">
   409→                  {Object.entries(s.distractorCounts)
   410→                    .sort(([, a], [, b]) => b - a)
   411→                    .slice(0, 3)
   412→                    .map(([optIdx, count]) => (
   413→                      <div key={optIdx} className="flex items-center gap-2 text-xs">
   414→                        <div className="w-2 h-2 rounded-full bg-red-400" />
   415→                        <span className="truncate max-w-[180px] text-muted-foreground">
   416→                          {s.options?.[Number(optIdx)] ?? `Option ${Number(optIdx) + 1}`}
   417→                        </span>
   418→                        <div className="flex-1 h-1.5 bg-muted rounded-full overflow-hidden">
   419→                          <div className="h-full bg-red-400/60 rounded-full" style={{ width: `${Math.min(100, (count / s.timesAnswered) * 200)}%` }} />
   420→                        </div>
   421→                        <span className="text-muted-foreground shrink-0">{count}</span>
   422→                      </div>
   423→                    ))}
   424→                </div>
   425→              </div>
   426→            )}
   427→          </motion.div>
   428→        );
   429→      })}
   430→    </div>
   431→  );
   432→}

   605→  const [statusFilter, setStatusFilter] = useState<string>('all');

   613→      a.description?.toLowerCase().includes(search.toLowerCase());

   616→      (statusFilter === 'published' && a.isPublished) ||
   617→      (statusFilter === 'draft' && !a.isPublished);

   625→      submissions: 12 + (seed * 7) % 156,
   626→      avgScore: 55 + (seed * 3) % 35,

   674→          { label: 'Total Submissions', value: totalSubmissions, icon: Send, gradient: 'from-emerald-500 to-teal-500' },

   676→          { label: 'Pass Rate', value: passRate, icon: Award, gradient: 'from-rose-500 to-pink-500', suffix: '%' },
