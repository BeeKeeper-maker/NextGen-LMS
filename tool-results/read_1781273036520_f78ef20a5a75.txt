    36→    36→// Markdown Renderer (simplified - handles code, bold, lists)
    37→    37→// ============================================================
    38→    38→function MarkdownRenderer({ content }: { content: string }) {
    39→    39→  const lines = content.split('\n');
    40→    40→  const elements: React.ReactNode[] = [];
    41→    41→  let inCodeBlock = false;
    42→    42→  let codeContent = '';
    43→    43→  let codeLang = '';
    44→    44→  let inList = false;
    45→    45→
    46→    46→  for (let i = 0; i < lines.length; i++) {
    47→    47→    const line = lines[i];
    48→    48→
    49→    49→    if (line.startsWith('```')) {
    50→    50→      if (inCodeBlock) {
    51→    51→        elements.push(
    52→    52→          <pre
    53→    53→            key={`code-${i}`}
    54→    54→            className="my-2 rounded-lg bg-slate-900 p-3 text-sm text-emerald-300 overflow-x-auto font-mono"
    55→    55→          >
    56→    56→            {codeLang && (
    57→    57→              <div className="text-xs text-slate-500 mb-1">{codeLang}</div>
    58→    58→            )}
    59→    59→            <code>{codeContent.trim()}</code>
    60→    60→          </pre>
    61→    61→        );
    62→    62→        codeContent = '';
    63→    63→        codeLang = '';
    64→    64→        inCodeBlock = false;
    65→    65→      } else {
    66→    66→        inCodeBlock = true;
    67→    67→        codeLang = line.slice(3).trim();
    68→    68→      }
    69→    69→      continue;
    70→    70→    }
    71→    71→
    72→    72→    if (inCodeBlock) {
    73→    73→      codeContent += line + '\n';
    74→    74→      continue;
    75→    75→    }
    76→    76→
    77→    77→    // Inline formatting
    78→    78→    const formatInline = (text: string): React.ReactNode => {
    79→    79→      // Bold
    80→    80→      const parts = text.split(/(\*\*[^*]+\*\*)/g);
    81→    81→      return parts.map((part, idx) => {
    82→    82→        if (part.startsWith('**') && part.endsWith('**')) {
    83→    83→          return (
    84→    84→            <strong key={idx} className="font-semibold text-foreground">
    85→    85→              {part.slice(2, -2)}
    86→    86→            </strong>
    87→    87→          );
    88→    88→        }
    89→    89→        // Inline code
    90→    90→        const codeParts = part.split(/(`[^`]+`)/g);
    91→    91→        return codeParts.map((cp, cidx) => {
    92→    92→          if (cp.startsWith('`') && cp.endsWith('`')) {
    93→    93→            return (
    94→    94→              <code
    95→    95→                key={`${idx}-${cidx}`}
    96→    96→                className="rounded bg-slate-100 px-1.5 py-0.5 text-xs font-mono text-violet-700 dark:bg-slate-800 dark:text-violet-300"
    97→    97→              >
    98→    98→                {cp.slice(1, -1)}
    99→    99→              </code>
   100→   100→            );
   101→   101→          }
   102→   102→          return cp;
   103→   103→        });
   104→   104→      });
   105→   105→    };
   106→   106→
   107→   107→    // Headers
   108→   108→    if (line.startsWith('### ')) {
   109→   109→      if (inList) { inList = false; }
   110→   110→      elements.push(
   111→   111→        <h4 key={`h3-${i}`} className="font-semibold text-foreground mt-3 mb-1">
   112→   112→          {formatInline(line.slice(4))}
   113→   113→        </h4>
   114→   114→      );
   115→   115→      continue;
   116→   116→    }
   117→   117→    if (line.startsWith('## ')) {
   118→   118→      if (inList) { inList = false; }
   119→   119→      elements.push(
   120→   120→        <h3 key={`h2-${i}`} className="text-base font-semibold text-foreground mt-3 mb-1">
   121→   121→          {formatInline(line.slice(3))}
   122→   122→        </h3>
   123→   123→      );
   124→   124→      continue;
   125→   125→    }
   126→   126→    if (line.startsWith('# ')) {
   127→   127→      if (inList) { inList = false; }
   128→   128→      elements.push(
   129→   129→        <h2 key={`h1-${i}`} className="text-lg font-bold text-foreground mt-3 mb-1">
   130→   130→          {formatInline(line.slice(2))}
   131→   131→        </h2>
   132→   132→      );
   133→   133→      continue;
   134→   134→    }
   135→   135→
   136→   136→    // Unordered list
   137→   137→    if (line.match(/^[-*]\s/)) {
   138→   138→      inList = true;
   139→   139→      elements.push(
   140→   140→        <li key={`li-${i}`} className="ml-4 list-disc text-sm text-muted-foreground">
   141→   141→          {formatInline(line.replace(/^[-*]\s/, ''))}
   142→   142→        </li>
   143→   143→      );
   144→   144→      continue;
   145→   145→    }
   146→   146→
   147→   147→    // Ordered list
   148→   148→    if (line.match(/^\d+\.\s/)) {
   149→   149→      inList = true;
   150→   150→      elements.push(
   151→   151→        <li key={`oli-${i}`} className="ml-4 list-decimal text-sm text-muted-foreground">
   152→   152→          {formatInline(line.replace(/^\d+\.\s/, ''))}
   153→   153→        </li>
   154→   154→      );
   155→   155→      continue;
   156→   156→    }
   157→   157→
   158→   158→    if (inList) { inList = false; }
   159→   159→
   160→   160→    // Empty line
   161→   161→    if (line.trim() === '') {
   162→   162→      elements.push(<div key={`br-${i}`} className="h-2" />);
   163→   163→      continue;
   164→   164→    }
   165→   165→
   166→   166→    // Paragraph
   167→   167→    elements.push(
   168→   168→      <p key={`p-${i}`} className="text-sm text-muted-foreground leading-relaxed">
   169→   169→        {formatInline(line)}
   170→   170→      </p>
   171→   171→    );
   172→   172→  }
   173→   173→
   174→   174→  return <div className="space-y-0.5">{elements}</div>;
   175→   175→}

   178→   178→// Persistent Chat Hook - localStorage persistence for chat history

   180→   180→const WELCOME_MESSAGE: ChatMessage = {
   181→   181→  id: 'welcome',
   182→   182→  role: 'assistant',
   183→   183→  content:
   184→   184→    "Hi! I'm your AI Learning Assistant. I can help you understand concepts, debug code, create study plans, and more. What would you like to learn today?",
   185→   185→  timestamp: new Date().toISOString(),
   186→   186→};
   187→   187→
   188→   188→function loadMessagesFromStorage(storageKey: string): ChatMessage[] {
   189→   189→  if (typeof window === 'undefined') return [WELCOME_MESSAGE];
   190→   190→  try {
   191→   191→    const saved = localStorage.getItem(storageKey);
   192→   192→    if (saved) {
   193→   193→      const data = JSON.parse(saved);
   194→   194→      if (Array.isArray(data.messages) && data.messages.length > 0) {
   195→   195→        return data.messages;
   196→   196→      }
   197→   197→    }
   198→   198→  } catch {
   199→   199→    // ignore parse errors
   200→   200→  }
   201→   201→  return [WELCOME_MESSAGE];
   202→   202→}
   203→   203→
   204→   204→function usePersistentChat(storageKey: string) {
   205→   205→  const [messages, setMessages] = useState<ChatMessage[]>(() => loadMessagesFromStorage(storageKey));
   206→   206→
   207→   207→  // Save to localStorage on changes
   208→   208→  useEffect(() => {
   209→   209→    if (messages.length > 0) {
   210→   210→      try {
   211→   211→        localStorage.setItem(storageKey, JSON.stringify({ messages }));
   212→   212→      } catch {
   213→   213→        // ignore storage errors
   214→   214→      }
   215→   215→    }
   216→   216→  }, [storageKey, messages]);
   217→   217→
   218→   218→  return { messages, setMessages };
   219→   219→}

   246→   246→function loadConversationsFromStorage(storageKey: string): Conversation[] {
   247→   247→  if (typeof window === 'undefined') return DEFAULT_CONVERSATIONS;

   253→   253→        return data.conversations;
   254→   254→      }

   262→   262→function usePersistentConversations(storageKey: string) {

   265→   265→  // Save to localStorage on changes

   267→   267→    if (conversations.length > 0) {
   268→   268→      try {
   269→   269→        localStorage.setItem(storageKey, JSON.stringify({ conversations }));
   270→   270→      } catch {
   271→   271→        // ignore storage errors
   272→   272→      }
   273→   273→    }
   274→   274→  }, [storageKey, conversations]);
   275→   275→
   276→   276→  return { conversations, setConversations };
   277→   277→}
