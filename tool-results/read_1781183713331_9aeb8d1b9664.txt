    52→function formatPrice(amount: number, currency: CurrencyOption): string {
    53→  const converted = amount * currency.rate;
    54→  if (currency.code === 'JPY') {
    55→    return `${currency.symbol}${Math.round(converted).toLocaleString()}`;
    56→  }
    57→  return `${currency.symbol}${converted.toFixed(2)}`;
    58→}
    59→
    60→function detectCardBrand(number: string): 'visa' | 'mastercard' | 'amex' | 'unknown' {
    61→  const cleaned = number.replace(/\s/g, '');
    62→  if (/^4/.test(cleaned)) return 'visa';
    63→  if (/^5[1-5]/.test(cleaned) || /^2[2-7]/.test(cleaned)) return 'mastercard';
    64→  if (/^3[47]/.test(cleaned)) return 'amex';
    65→  return 'unknown';
    66→}
    67→
    68→function formatCardNumber(value: string): string {
    69→  const cleaned = value.replace(/\D/g, '').slice(0, 16);
    70→  const groups = cleaned.match(/.{1,4}/g);
    71→  return groups ? groups.join(' ') : cleaned;
    72→}
    73→
    74→function formatExpiry(value: string): string {
    75→  const cleaned = value.replace(/\D/g, '').slice(0, 4);
    76→  if (cleaned.length > 2) {
    77→    return `${cleaned.slice(0, 2)}/${cleaned.slice(2)}`;
    78→  }
    79→  return cleaned;
    80→}

    91→function AnimatedPrice({ value, currency }: { value: number; currency: CurrencyOption }) {
    92→  const motionVal = useMotionValue(0);
    93→  const display = useTransform(motionVal, (v) => {
    94→    if (currency.code === 'JPY') {
    95→      return `${currency.symbol}${Math.round(v).toLocaleString()}`;
    96→    }
    97→    return `${currency.symbol}${v.toFixed(2)}`;
    98→  });
    99→  const [displayStr, setDisplayStr] = useState(formatPrice(value, currency));
   100→  const prevValue = useRef(value);
   101→
   102→  useEffect(() => {
   103→    const controls = animate(motionVal, value, {
   104→      duration: 0.6,
   105→      ease: 'easeOut',
   106→      onUpdate: (v) => {
   107→        if (currency.code === 'JPY') {
   108→          setDisplayStr(`${currency.symbol}${Math.round(v).toLocaleString()}`);
   109→        } else {
   110→          setDisplayStr(`${currency.symbol}${v.toFixed(2)}`);
   111→        }
   112→      },
   113→    });
   114→    prevValue.current = value;
   115→    return () => controls.stop();
   116→  }, [value, currency.code, currency.symbol, motionVal]);
   117→
   118→  return <span>{displayStr}</span>;
   119→}

   123→function ConfettiParticle({ delay }: { delay: number }) {
   124→  const colors = ['#10B981', '#6366F1', '#F59E0B', '#EF4444', '#8B5CF6', '#EC4899', '#14B8A6'];
   125→  const color = colors[Math.floor(Math.random() * colors.length)];

   127→  const rotation = Math.random() * 360;
   128→  const size = 6 + Math.random() * 8;
   129→
   130→  return (
   131→    <motion.div
   132→      initial={{ opacity: 1, y: -20, x: `${x}vw`, rotate: 0 }}
   133→      animate={{
   134→        opacity: [1, 1, 0],
   135→        y: ['0vh', '100vh'],
   136→        x: [`${x}vw`, `${x + (Math.random() - 0.5) * 20}vw`],
   137→        rotate: [0, rotation + 720],
   138→      }}
   139→      transition={{ duration: 2.5 + Math.random(), delay, ease: 'easeIn' }}
   140→      className="fixed top-0 pointer-events-none z-50"
   141→      style={{ width: size, height: size, backgroundColor: color, borderRadius: Math.random() > 0.5 ? '50%' : '2px' }}
   142→    />
   143→  );
