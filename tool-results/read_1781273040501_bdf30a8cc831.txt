   175тЖТ   175тЖТfunction useLessonNotes(courseId: string | undefined) {
   176тЖТ   176тЖТ  const storageKey = `lms-notes-${courseId || 'default'}`;
   177тЖТ   177тЖТ
   178тЖТ   178тЖТ  const [notes, setNotes] = useState<Note[]>(() => {
   179тЖТ   179тЖТ    if (typeof window === 'undefined') return [];
   180тЖТ   180тЖТ    try {
   181тЖТ   181тЖТ      const stored = localStorage.getItem(storageKey);
   182тЖТ   182тЖТ      return stored ? JSON.parse(stored) : [];
   183тЖТ   183тЖТ    } catch {
   184тЖТ   184тЖТ      return [];
   185тЖТ   185тЖТ    }
   186тЖТ   186тЖТ  });
   187тЖТ   187тЖТ
   188тЖТ   188тЖТ  const persistNotes = useCallback((updated: Note[]) => {
   189тЖТ   189тЖТ    setNotes(updated);
   190тЖТ   190тЖТ    if (typeof window !== 'undefined') {
   191тЖТ   191тЖТ      try {
   192тЖТ   192тЖТ        localStorage.setItem(storageKey, JSON.stringify(updated));
   193тЖТ   193тЖТ      } catch {
   194тЖТ   194тЖТ        // ignore quota errors
   195тЖТ   195тЖТ      }
   196тЖТ   196тЖТ    }
   197тЖТ   197тЖТ  }, [storageKey]);
   198тЖТ   198тЖТ
   199тЖТ   199тЖТ  const addNote = useCallback((note: Omit<Note, 'id' | 'createdAt' | 'updatedAt'>) => {
   200тЖТ   200тЖТ    const newNote: Note = {
   201тЖТ   201тЖТ      ...note,
   202тЖТ   202тЖТ      id: `note-${Date.now()}-${Math.random().toString(36).slice(2, 7)}`,
   203тЖТ   203тЖТ      createdAt: new Date().toISOString(),
   204тЖТ   204тЖТ      updatedAt: new Date().toISOString(),
   205тЖТ   205тЖТ    };
   206тЖТ   206тЖТ    persistNotes([...notes, newNote]);
   207тЖТ   207тЖТ    return newNote;
   208тЖТ   208тЖТ  }, [notes, persistNotes]);
   209тЖТ   209тЖТ
   210тЖТ   210тЖТ  const updateNote = useCallback((id: string, updates: Partial<Note>) => {
   211тЖТ   211тЖТ    const updated = notes.map(n =>
   212тЖТ   212тЖТ      n.id === id ? { ...n, ...updates, updatedAt: new Date().toISOString() } : n
   213тЖТ   213тЖТ    );
   214тЖТ   214тЖТ    persistNotes(updated);
   215тЖТ   215тЖТ  }, [notes, persistNotes]);
   216тЖТ   216тЖТ
   217тЖТ   217тЖТ  const deleteNote = useCallback((id: string) => {
   218тЖТ   218тЖТ    persistNotes(notes.filter(n => n.id !== id));
   219тЖТ   219тЖТ  }, [notes, persistNotes]);
   220тЖТ   220тЖТ
   221тЖТ   221тЖТ  return { notes, addNote, updateNote, deleteNote };
   222тЖТ   222тЖТ}
